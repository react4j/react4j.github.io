001package react4j;
002
003import org.realityforge.braincheck.BrainCheckConfig;
004
005/**
006 * Location of all compile time configuration settings for framework.
007 */
008public final class ReactConfig
009{
010  private static final ConfigProvider PROVIDER = new ConfigProvider();
011  private static final boolean ENABLE_NAMES = PROVIDER.enableComponentNames();
012  private static final boolean SHOULD_MINIMIZE_PROP_KEYS = PROVIDER.shouldMinimizePropKeys();
013  private static final boolean SHOULD_VALIDATE_PROP_VALUES = PROVIDER.shouldValidatePropValues();
014  private static final boolean SHOULD_STORE_DEBUG_DATA_AS_STATE = PROVIDER.shouldStoreDebugDataAsState();
015  private static final boolean SHOULD_FREEZE_PROPS = PROVIDER.shouldFreezeProps();
016  private static final boolean CHECK_INVARIANTS = PROVIDER.shouldCheckInvariants();
017
018  private ReactConfig()
019  {
020  }
021
022  /**
023   * Return true if components should have human readable names specified.
024   * Useful if you want to interact via DevTools or other tool chains.
025   *
026   * @return true to enable human readable names for components.
027   */
028  public static boolean enableComponentNames()
029  {
030    return ENABLE_NAMES;
031  }
032
033  /**
034   * Return true if the prop keys should be minimized.
035   * This will significantly reduce the size of the compiled output but will make inspecting the props
036   * in DevTools difficult if not impossible.
037   *
038   * @return true to minimize prop keys.
039   */
040  public static boolean shouldMinimizePropKeys()
041  {
042    return SHOULD_MINIMIZE_PROP_KEYS;
043  }
044
045  /**
046   * Return true if the prop value should be validated when initially set or when changed.
047   *
048   * @return true to validate prop values.
049   */
050  public static boolean shouldValidatePropValues()
051  {
052    return SHOULD_VALIDATE_PROP_VALUES;
053  }
054
055  /**
056   * Return true if react state should be used to store debug data.
057   * Useful if you want to inspect the debug data via DevTools. This feature is resource intensive
058   * and should not be enabled in production.
059   *
060   * @return true if react state should be used to store debug data.
061   */
062  public static boolean shouldStoreDebugDataAsState()
063  {
064    return SHOULD_STORE_DEBUG_DATA_AS_STATE;
065  }
066
067  /**
068   * Return true if invariants will be checked.
069   *
070   * @return true if invariants will be checked.
071   */
072  public static boolean shouldCheckInvariants()
073  {
074    return CHECK_INVARIANTS && BrainCheckConfig.checkInvariants();
075  }
076
077  /**
078   * Return true if props should be frozen before being passed to react.
079   *
080   * @return true if props should be frozen before being passed to react.
081   */
082  public static boolean shouldFreezeProps()
083  {
084    return SHOULD_FREEZE_PROPS;
085  }
086
087  private static final class ConfigProvider
088    extends AbstractConfigProvider
089  {
090    @GwtIncompatible
091    @Override
092    boolean enableComponentNames()
093    {
094      return "true".equals( System.getProperty( "react4j.enable_component_names", "false" ) );
095    }
096
097    @GwtIncompatible
098    @Override
099    boolean shouldMinimizePropKeys()
100    {
101      return "true".equals( System.getProperty( "react4j.minimize_prop_keys", "true" ) );
102    }
103
104    @GwtIncompatible
105    @Override
106    boolean shouldValidatePropValues()
107    {
108      return "true".equals( System.getProperty( "react4j.validate_prop_values", "false" ) );
109    }
110
111    @GwtIncompatible
112    @Override
113    boolean shouldStoreDebugDataAsState()
114    {
115      return "true".equals( System.getProperty( "react4j.store_debug_data_as_state", "false" ) );
116    }
117
118    @GwtIncompatible
119    @Override
120    boolean shouldCheckInvariants()
121    {
122      return "true".equals( System.getProperty( "react4j.check_invariants", "false" ) );
123    }
124
125    @GwtIncompatible
126    @Override
127    boolean shouldFreezeProps()
128    {
129      return "true".equals( System.getProperty( "react4j.freeze_props", "false" ) );
130    }
131  }
132
133  @SuppressWarnings( { "unused", "StringEquality" } )
134  private static abstract class AbstractConfigProvider
135  {
136    boolean enableComponentNames()
137    {
138      return "true" == System.getProperty( "react4j.enable_component_names" );
139    }
140
141    boolean shouldMinimizePropKeys()
142    {
143      return "true" == System.getProperty( "react4j.minimize_prop_keys" );
144    }
145
146    boolean shouldValidatePropValues()
147    {
148      return "true" == System.getProperty( "react4j.validate_prop_values" );
149    }
150
151    boolean shouldStoreDebugDataAsState()
152    {
153      return "true" == System.getProperty( "react4j.store_debug_data_as_state" );
154    }
155
156    boolean shouldCheckInvariants()
157    {
158      return "true" == System.getProperty( "react4j.check_invariants" );
159    }
160
161    boolean shouldFreezeProps()
162    {
163      return "true" == System.getProperty( "react4j.freeze_props" );
164    }
165  }
166}
