001package react4j;
002
003import org.realityforge.braincheck.BrainCheckConfig;
004
005/**
006 * Location of all compile time configuration settings for framework.
007 */
008public final class ReactConfig
009{
010  private static final ConfigProvider PROVIDER = new ConfigProvider();
011  private static final boolean PRODUCTION_MODE = PROVIDER.isProductionMode();
012  private static boolean ENABLE_NAMES = PROVIDER.enableComponentNames();
013  private static final boolean CHECK_COMPONENT_STATE_INVARIANTS = PROVIDER.checkComponentStateInvariants();
014  private static final boolean CHECK_INVARIANTS = PROVIDER.shouldCheckInvariants();
015
016  private ReactConfig()
017  {
018  }
019
020  /**
021   * Return true if in production mode.
022   * Production mode sets the default values for other compile time constants to the variant
023   * that assumes your code is correct and does not generate additional assert or debug statements.
024   * The individual configuration settings can still be specified to override this value.
025   *
026   * @return true if in production mode.
027   */
028  public static boolean isProductionMode()
029  {
030    return PRODUCTION_MODE;
031  }
032
033  /**
034   * Return true if components should have human readable names specified.
035   * Useful if you want to interact via DevTools or other tool chains.
036   *
037   * @return to enable human readable names for components.
038   */
039  public static boolean enableComponentNames()
040  {
041    return ENABLE_NAMES;
042  }
043
044  /**
045   * Return true if we should check that the user interacts with React component in a way compatible with the state.
046   * i.e. setInitialState should only be invoked at the start, setState should not be invoked while render
047   * is running etc.
048   *
049   * @return true to enable invariant checking about how we interact with native react component.
050   */
051  public static boolean checkComponentStateInvariants()
052  {
053    return CHECK_COMPONENT_STATE_INVARIANTS;
054  }
055
056  /**
057   * Return true if invariants will be checked.
058   *
059   * @return true if invariants will be checked.
060   */
061  public static boolean shouldCheckInvariants()
062  {
063    return CHECK_INVARIANTS && BrainCheckConfig.checkInvariants();
064  }
065
066  private static final class ConfigProvider
067    extends AbstractConfigProvider
068  {
069    @GwtIncompatible
070    @Override
071    boolean isProductionMode()
072    {
073      return "production".equals( System.getProperty( "react4j.environment", "production" ) );
074    }
075
076    @GwtIncompatible
077    @Override
078    boolean enableComponentNames()
079    {
080      return "true".equals( System.getProperty( "react4j.enable_component_names",
081                                                isProductionMode() ? "false" : "true" ) );
082    }
083
084    @GwtIncompatible
085    @Override
086    boolean checkComponentStateInvariants()
087    {
088      return "true".equals( System.getProperty( "react4j.check_component_state_invariants",
089                                                isProductionMode() ? "false" : "true" ) );
090    }
091
092    @GwtIncompatible
093    @Override
094    boolean shouldCheckInvariants()
095    {
096      return "true".equals( System.getProperty( "react4j.check_invariants", "true" ) );
097    }
098  }
099
100  @SuppressWarnings( { "unused", "StringEquality" } )
101  private static abstract class AbstractConfigProvider
102  {
103    boolean isProductionMode()
104    {
105      return "production" == System.getProperty( "react4j.environment" );
106    }
107
108    boolean enableComponentNames()
109    {
110      return "true" == System.getProperty( "react4j.enable_component_names" );
111    }
112
113    boolean checkComponentStateInvariants()
114    {
115      return "true" == System.getProperty( "react4j.check_component_state_invariants" );
116    }
117
118    boolean shouldCheckInvariants()
119    {
120      return "true" == System.getProperty( "react4j.check_invariants" );
121    }
122  }
123}
