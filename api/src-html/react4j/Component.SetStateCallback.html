012/**
013 * The base java class that mirrors the react component.
014 */
015public abstract class Component
016{
017  /**
018   * Callback function for updating state.
019   * Useful if the state update is based on current state.
020   */
021  @JsFunction
022  @FunctionalInterface
023  public interface SetStateCallback
024  {
025    /**
026     * Callback used to update state.
027     * Result is merged into existing state.
028     * Return null to abort state update.
029     *
030     * @param previousState the preiovus state.
031     * @param currentProps  the current props.
032     * @return the state to shallow merge or null to abort state update.
033     */
034    @Nullable
035    JsPropertyMap<Object> onSetState( @Nullable JsPropertyMap<Object> previousState,
036                                      @Nullable JsPropertyMap<Object> currentProps );
037  }
038
039  @Nonnull
040  private ComponentPhase _phase = ComponentPhase.INITIALIZING;
041  @Nonnull
042  private LifecycleMethod _lifecycleMethod = LifecycleMethod.UNKNOWN;
043  @Nullable
044  private NativeComponent _nativeComponent;
045
046  /**
047   * Set the phase of the component. Only used for invariant checking.
048   */
049  final void setPhase( @Nonnull final ComponentPhase phase )
050  {
051    if ( ReactConfig.shouldCheckInvariants() )
052    {
053      invariant( ReactConfig::checkComponentStateInvariants,
054                 () -> "Component.setComponentPhase() invoked on " + this +
055                       " when ReactConfig.checkComponentStateInvariants() is false" );
056    }
057    _phase = Objects.requireNonNull( phase );
058  }
059
060  /**
061   * Set the current lifecycle method of the component. Only used for invariant checking.
062   */
063  final void setLifecycleMethod( @Nonnull final LifecycleMethod lifecycleMethod )
064  {
065    if ( ReactConfig.shouldCheckInvariants() )
066    {
067      invariant( ReactConfig::checkComponentStateInvariants,
068                 () -> "Component.setLifecycleMethod() invoked on " + this +
069                       " when ReactConfig.checkComponentStateInvariants() is false" );
070    }
071    _lifecycleMethod = Objects.requireNonNull( lifecycleMethod );
072  }
073
074  final void bindComponent( @Nonnull final NativeComponent nativeComponent )
075  {
076    _nativeComponent = Objects.requireNonNull( nativeComponent );
077  }
078
079  /**
080   * Set the initial state of the component.
081   * This should only be invoked when the component is initializing.
082   * Calling this at any other time is an error.
083   *
084   * @param state the state.
085   */
086  protected final void setInitialState( @Nonnull final JsPropertyMap<Object> state )
087  {
088    if ( ReactConfig.shouldCheckInvariants() && ReactConfig.checkComponentStateInvariants() )
089    {
090      apiInvariant( () -> ComponentPhase.INITIALIZING == _phase,
091                    () -> "Attempted to invoke setInitialState on " + this + " when component is " +
092                          "not in INITIALIZING phase but in phase " + _phase + " and state " + _lifecycleMethod );
093    }
094    component().setInitialState( state );
095  }
096
097  /**
098   * Return the native react component.
099   */
100  @Nonnull
101  private NativeComponent component()
102  {
103    if ( ReactConfig.shouldCheckInvariants() )
104    {
105      invariant( () -> null != _nativeComponent,
106                 () -> "Invoked component() on " + this + " before a component has been bound." );
107    }
108    assert null != _nativeComponent;
109    return _nativeComponent;
110  }
111
112  /**
113   * Return the component state from the native component.
114   * This may be null if initial state was never set.
115   *
116   * @return the component state.
117   */
118  protected final JsPropertyMap<Object> state()
119  {
120    return component().state();
121  }
122
123  /**
124   * Return the component props from the native component.
125   * This may be null if no props were supplied.
126   *
127   * @return the component state.
128   */
129  protected final JsPropertyMap<Object> props()
130  {
131    return component().props();
132  }
133
134  /**
135   * Return the key associated with the component if any.
136   * The key is used by the react reconcilliation process as a means to identify components so that they
137   * can be moved rather than recreated or reconfigured. Typically, it should not be used by the component
138   * itself except for generating keys of sub-components. This value may not be specified in which case null
139   * will be returned.
140   *
141   * @return the key if specified.
142   */
143  @Nullable
144  protected final String getKey()
145  {
146    return Js.asPropertyMap( props() ).getAny( "key" ).asString();
147  }

149  /**
150   * Schedule a shallow merge of supplied state into current state.
151   * This will trigger an update cycle and is the primary method you
152   * use to trigger UI updates from event handlers and server request callbacks.
153   *
154   * @param state the object literal representing state.
155   */
156  protected final void scheduleStateUpdate( @Nonnull final JsPropertyMap<Object> state )
157  {
158    scheduleStateUpdate( ( p, s ) -> state );
159  }
160
161  /**
162   * Schedule a shallow merge of supplied state into current state.
163   * This will trigger an update cycle and is the primary method you
164   * use to trigger UI updates from event handlers and server request callbacks.
165   *
166   * @param state                 the object literal representing state.
167   * @param onStateUpdateComplete a callback that will be invoked after state has been updated.
168   */
169  protected final void scheduleStateUpdate( @Nonnull final JsPropertyMap<Object> state,
170                                            @Nullable final Procedure onStateUpdateComplete )
171  {
172    scheduleStateUpdate( ( p, s ) -> state, onStateUpdateComplete );
173  }
174
175  /**
176   * Schedule a shallow merge of supplied state into current state.
177   * This will trigger an update cycle and is the primary method you
178   * use to trigger UI updates from event handlers and server request callbacks.
179   *
180   * @param callback the callback that will be invoked to update state.
181   */
182  protected final void scheduleStateUpdate( @Nonnull final SetStateCallback callback )
183  {
184    scheduleStateUpdate( callback, null );
185  }
186
187  /**
188   * Schedule a shallow merge of supplied state into current state.
189   * This will trigger an update cycle and is the primary method you
190   * use to trigger UI updates from event handlers and server request callbacks.
191   *
192   * @param callback              the callback that will be invoked to update state.
193   * @param onStateUpdateComplete a callback that will be invoked after state has been updated.
194   */
195  protected void scheduleStateUpdate( @Nonnull final SetStateCallback callback,
196                                      @Nullable final Procedure onStateUpdateComplete )
197  {
198    invariantsSetState();
199    component().setState( callback, onStateUpdateComplete );
200  }
201
202  /**
203   * Check invariants that should be true when invoking setState()
204   */
205  private void invariantsSetState()
206  {
207    if ( ReactConfig.shouldCheckInvariants() && ReactConfig.checkComponentStateInvariants() )
208    {
209      apiInvariant( () -> LifecycleMethod.RENDER != _lifecycleMethod,
210                    () -> "Incorrectly invoked scheduleStateUpdate() on " + this + " in scope of render()." );
211      apiInvariant( () -> ComponentPhase.UNMOUNTING != _phase,
212                    () -> "Incorrectly invoked scheduleStateUpdate() on " + this + " when component is " +
213                          "unmounting or has unmounted." );
214    }
215  }
216
217  /**
218   * Schedule this component for re-rendering.
219   * The component re-renders when state or props change but calling this method is another way to
220   * schedule the component to be re-rendered.
221   *
222   * <p>If the force parameter is true then the {@link #shouldComponentUpdate(JsPropertyMap, JsPropertyMap)} will be skipped
223   * and it is equivalent to calling forceUpdate() on the native react component. See the
224   * <a href="https://reactjs.org/docs/react-component.html#forceupdate">React Component documentation</a> for more
225   * details.</p>
226   *
227   * <p>If the force parameter is true then the {@link #shouldComponentUpdate(JsPropertyMap, JsPropertyMap)} will be
228   * invoked. This is equivalent to calling setState({}) on the native react component.</p>
229   *
230   * @param force true to skip shouldComponentUpdate during re-render, false otherwise.
231   */
232  protected final void scheduleRender( final boolean force )
233  {
234    if ( ReactConfig.shouldCheckInvariants() && ReactConfig.checkComponentStateInvariants() )
235    {
236      apiInvariant( () -> ComponentPhase.UNMOUNTING != _phase,
237                    () -> "Incorrectly invoked scheduleRender() on " + this + " when component is " +
238                          "unmounting or has unmounted." );
239    }
240    if ( force )
241    {
242      component().forceUpdate();
243    }
244    else
245    {
246      // This schedules a re-render but will not skip shouldComponentUpdate
247      scheduleStateUpdate( JsPropertyMap.of() );
248    }
249  }

272  /**
273   * Wrapper method that delegates to the {@link #postConstruct()} method.
274   * This method exists to give middleware a mechanism to hook into component lifecycle step.
275   */
276  protected void performPostConstruct()
277  {
278    final JsPropertyMap<Object> props = props();
279    if ( ReactConfig.shouldValidatePropValues() && null != props )
280    {
281      validatePropValues( props );
282    }
283    postConstruct();
284  }

362  /**
363   * The componentDidCatch() method works like a JavaScript catch {} block, but for components.
364   * Only class components can be error boundaries. In practice, most of the time you’ll want to
365   * declare an error boundary component once and use it throughout your application.
366   *
367   * <p>Note that error boundaries only catch errors in the components below them in the tree. An
368   * error boundary can’t catch an error within itself. If an error boundary fails trying to render
369   * the error message, the error will propagate to the closest error boundary above it. This, too,
370   * is similar to how catch {} block works in JavaScript.</p>
371   *
372   * @param error the error that has been thrown.
373   * @param info  information about component stack during thrown error.
374   */

376  {

391  /**
392   * Use this method to let React know if a component's output is not affected
393   * by the current change in state or props. The default behavior is to re-render on
394   * every state change, and in the vast majority of cases you should rely on the default behavior.
395   * See the <a href="https://reactjs.org/docs/react-component.html#shouldcomponentupdate">React Component documentation</a> for more details.
396   *
397   * <p>This method is invoked before rendering when new props or state are being received.
398   * This method is not called for the initial render or when {@link #scheduleRender(boolean)}} is called
399   * with the force parameter set to true.</p>
400   *
401   * <p>Returning false does not prevent child components from re-rendering when their state changes.</p>
402   *
403   * <p>If this method returns false, then {@link #render()}, and {@link #componentDidUpdate(JsPropertyMap, JsPropertyMap)}
404   * will not be invoked. In the future React may treat this method  as a hint rather than a strict directive, and
405   * returning false may still result in a re-rendering of the component.</p>
406   *
407   * @param nextProps the new properties of the component.
408   * @param nextState the new state of the component.
409   * @return true if the component should be updated.
410   */
411  protected boolean shouldComponentUpdate( @Nullable final JsPropertyMap<Object> nextProps,
412                                           @Nullable final JsPropertyMap<Object> nextState )
413  {
414    return true;
415  }
416
417  /**
418   * Wrapper method that delegates to the {@link #shouldComponentUpdate(JsPropertyMap, JsPropertyMap)} method.
419   * This method exists to give middleware a mechanism to hook into component lifecycle step.
420   *
421   * @param nextProps the new properties of the component.
422   * @param nextState the new state of the component.
423   * @return true if the component should be updated.
424   */
425  protected boolean performShouldComponentUpdate( @Nullable final JsPropertyMap<Object> nextProps,
426                                                  @Nullable final JsPropertyMap<Object> nextState )
427  {
428    if ( ReactConfig.shouldValidatePropValues() && null != nextProps )
429    {
430      validatePropValues( nextProps );
431    }
432    return shouldComponentUpdate( nextProps, nextState );
433  }
