047/**
048 * Annotation processor that analyzes React annotated source and generates models from the annotations.
049 */
050@SuppressWarnings( "Duplicates" )
051@AutoService( Processor.class )
052@SupportedAnnotationTypes( { "react4j.annotations.*" } )
053@SupportedSourceVersion( SourceVersion.RELEASE_8 )
054public final class ReactProcessor
055  extends AbstractProcessor
056{
057  /**
058   * Cache of lifecycle names to methods as defined by Component.
059   */
060  private final HashMap<String, ExecutableElement> _componentLifecycleMethods = new HashMap<>();
061  /**
062   * Cache of render method method as defined by Component.
063   */
064  private ExecutableElement _componentRenderMethod;
065  /**
066   * Elements that were unresolved and have been deferred to a later compilation cycle.
067   */
068  @Nonnull
069  private HashSet<TypeElement> _deferred = new HashSet<>();

071  /**
072   * {@inheritDoc}
073   */
074  @Override
075  public boolean process( final Set<? extends TypeElement> annotations, final RoundEnvironment env )
076  {
077    // Clear method caches to avoid potential inter-run problems
078    _componentLifecycleMethods.clear();
079    _componentRenderMethod = null;
080
081    final TypeElement annotation =
082      processingEnv.getElementUtils().getTypeElement( Constants.REACT_COMPONENT_ANNOTATION_CLASSNAME );
083    final Set<? extends Element> elements = env.getElementsAnnotatedWith( annotation );
084    final Collection<Element> elementsToProcess = getElementsToProcess( elements );
085    processElements( elementsToProcess );
086    if ( env.getRootElements().isEmpty() && !_deferred.isEmpty() )
087    {
088      _deferred.forEach( this::processingErrorMessage );
089      _deferred.clear();
090    }
091    return true;
092  }
093
094  private void processingErrorMessage( @Nonnull final TypeElement target )
095  {
096    processingEnv
097      .getMessager()
098      .printMessage( ERROR,
099                     "ReactProcessor unable to process " + target.getQualifiedName() +
100                     " because not all of its dependencies could be resolved. Check for " +
101                     "compilation errors or a circular dependency with generated code.",
102                     target );
103  }

105  @Nonnull
106  private Collection<Element> getElementsToProcess( @Nonnull final Set<? extends Element> elements )
107  {
108    final List<TypeElement> deferred = _deferred
109      .stream()
110      .map( e -> processingEnv.getElementUtils().getTypeElement( e.getQualifiedName() ) )
111      .collect( Collectors.toList() );
112    _deferred = new HashSet<>();
113
114    final ArrayList<Element> elementsToProcess = new ArrayList<>();
115    collectElementsToProcess( elements, elementsToProcess );
116    collectElementsToProcess( deferred, elementsToProcess );
117    return elementsToProcess;
118  }
119
120  private void collectElementsToProcess( @Nonnull final Collection<? extends Element> elements,
121                                         @Nonnull final ArrayList<Element> elementsToProcess )
122  {
123    for ( final Element element : elements )
124    {
125      if ( SuperficialValidation.validateElement( element ) )
126      {
127        elementsToProcess.add( element );
128      }
129      else
130      {
131        _deferred.add( (TypeElement) element );
132      }
133    }
134  }

136  private void processElements( @Nonnull final Collection<? extends Element> elements )
137  {
138    for ( final Element element : elements )
139    {
140      try
141      {
142        process( (TypeElement) element );
143      }
144      catch ( final IOException ioe )
145      {
146        processingEnv.getMessager().printMessage( ERROR, ioe.getMessage(), element );
147      }
148      catch ( final ReactProcessorException e )
149      {
150        processingEnv.getMessager().printMessage( ERROR, e.getMessage(), e.getElement() );
151      }
152      catch ( final Throwable e )
153      {
154        final StringWriter sw = new StringWriter();
155        e.printStackTrace( new PrintWriter( sw ) );
156        sw.flush();
157
158        final String message =
159          "Unexpected error will running the " + getClass().getName() + " processor. This has " +
160          "resulted in a failure to process the code and has left the compiler in an invalid " +
161          "state. Please report the failure to the developers so that it can be fixed.\n" +
162          " Report the error at: https://github.com/react4j/react4j/issues\n" +
163          "\n\n" +
164          sw.toString();
165        processingEnv.getMessager().printMessage( ERROR, message, element );
166      }
167    }
168  }

170  private void process( @Nonnull final TypeElement element )
171    throws IOException, ReactProcessorException
172  {
173    final ComponentDescriptor descriptor = parse( element );
174    if ( descriptor.needsHelper() )
175    {
176      emitTypeSpec( descriptor.getPackageName(), Generator.buildComponentHelper( descriptor ) );
177    }
178    emitTypeSpec( descriptor.getPackageName(), Generator.buildEnhancedComponent( descriptor ) );
179    emitTypeSpec( descriptor.getPackageName(), Generator.buildComponentBuilder( descriptor ) );
180    if ( descriptor.needsDaggerIntegration() )
181    {
182      emitTypeSpec( descriptor.getPackageName(), Generator.buildDaggerFactory( descriptor ) );
183    }
184  }
185
186  private void emitTypeSpec( @Nonnull final String packageName, @Nonnull final TypeSpec typeSpec )
187    throws IOException
188  {
189    JavaFile.builder( packageName, typeSpec ).
190      skipJavaLangImports( true ).
191      build().
192      writeTo( processingEnv.getFiler() );
193  }

195  @Nonnull
196  private ComponentDescriptor parse( @Nonnull final TypeElement typeElement )
197  {
198    final String name = deriveComponentName( typeElement );
199    final PackageElement packageElement = processingEnv.getElementUtils().getPackageOf( typeElement );
200    final ComponentDescriptor descriptor =
201      new ComponentDescriptor( processingEnv.getElementUtils(),
202                               processingEnv.getSourceVersion(),
203                               name,
204                               packageElement,
205                               typeElement );
206
207    determineComponentType( descriptor, typeElement );
208    determineRenderMethod( typeElement, descriptor );
209    determineCallbacks( descriptor );
210    determineProps( descriptor );
211    determinePropValidatesMethods( descriptor );
212    determineDefaultPropsMethods( descriptor );
213    determineDefaultPropsFields( descriptor );
214    determineStateValues( descriptor );
215    determineLifecycleMethods( typeElement, descriptor );
216
217    for ( final PropDescriptor prop : descriptor.getProps() )
218    {
219      if ( !isPropRequired( prop ) )
220      {
221        prop.markAsOptional();
222      }
223    }
224
225    /*
226     * Sorting must occur after @PropDefault has been processed to ensure the sorting
227     * correctly sorts optional props after required props.
228     */
229    descriptor.sortProps();
230
231    verifyNoUnexpectedAbstractMethod( descriptor );
232    verifyStateNotAnnotatedWithArezAnnotations( descriptor );
233    verifyPropsNotAnnotatedWithArezAnnotations( descriptor );
234    verifyPropsNotCollectionOfArezComponents( descriptor );
235
236    return descriptor;
237  }

239  private void verifyPropsNotCollectionOfArezComponents( @Nonnull final ComponentDescriptor descriptor )
240  {
241    for ( final PropDescriptor prop : descriptor.getProps() )
242    {
243      final ExecutableElement method = prop.getMethod();
244      final TypeMirror returnType = method.getReturnType();
245      if ( TypeKind.DECLARED == returnType.getKind() )
246      {
247        final DeclaredType declaredType = (DeclaredType) returnType;
248        final List<? extends TypeMirror> typeArguments = declaredType.getTypeArguments();
249        if ( isCollection( declaredType ) )
250        {
251          if ( 1 == typeArguments.size() && isArezComponent( typeArguments.get( 0 ) ) )
252          {
253            throw new ReactProcessorException( "@Prop target is a collection that contains Arez components. " +
254                                               "This is not a safe pattern when the arez components can be disposed.",
255                                               method );
256          }
257        }
258        else if ( isMap( declaredType ) )
259        {
260          if ( 2 == typeArguments.size() &&
261               ( isArezComponent( typeArguments.get( 0 ) ) ||
262                 isArezComponent( typeArguments.get( 1 ) ) ) )
263          {
264            throw new ReactProcessorException( "@Prop target is a collection that contains Arez components. " +
265                                               "This is not a safe pattern when the arez components can be disposed.",
266                                               method );
267          }
268        }
269      }
270      else if ( TypeKind.ARRAY == returnType.getKind() )
271      {
272        final ArrayType arrayType = (ArrayType) returnType;
273        if ( isArezComponent( arrayType.getComponentType() ) )
274        {
275          throw new ReactProcessorException( "@Prop target is an array that contains Arez components. " +
276                                             "This is not a safe pattern when the arez components can be disposed.",
277                                             method );
278        }
279      }
280    }
281  }

283  private boolean isCollection( @Nonnull final DeclaredType declaredType )
284  {
285    final TypeElement returnType = (TypeElement) processingEnv.getTypeUtils().asElement( declaredType );
286    final String classname = returnType.getQualifiedName().toString();
287    /*
288     * For the time being lets just list out a bunch of collections. We can ge more specific when/if
289     * it is ever required
290     */
291    return Collection.class.getName().equals( classname ) ||
292           Set.class.getName().equals( classname ) ||
293           List.class.getName().equals( classname ) ||
294           HashSet.class.getName().equals( classname ) ||
295           ArrayList.class.getName().equals( classname );
296  }

309  private boolean isArezComponent( @Nonnull final TypeMirror typeMirror )
310  {
311    return typeMirror instanceof DeclaredType &&
312           processingEnv.getTypeUtils()
313             .asElement( typeMirror )
314             .getAnnotationMirrors()
315             .stream()
316             .anyMatch( a -> a.getAnnotationType().toString().equals( Constants.AREZ_COMPONENT_ANNOTATION_CLASSNAME ) );

319  private void verifyPropsNotAnnotatedWithArezAnnotations( @Nonnull final ComponentDescriptor descriptor )
320  {
321    for ( final PropDescriptor prop : descriptor.getProps() )
322    {
323      final ExecutableElement method = prop.getMethod();
324      for ( final AnnotationMirror mirror : method.getAnnotationMirrors() )
325      {
326        final String classname = mirror.getAnnotationType().toString();
327        if ( isArezAnnotation( classname ) )
328        {
329          throw new ReactProcessorException( "@Prop target must not be annotated with any arez annotations but " +
330                                             "is annotated by '" + classname + "'.", method );
331        }
332      }
333    }

363  private void linkStateMethods( @Nonnull final ComponentDescriptor descriptor )
364  {
365    final List<ExecutableElement> candidates =
366      getMethods( descriptor.getElement() )
367        .stream()
368        .filter( m -> m.getModifiers().contains( Modifier.ABSTRACT ) )
369        .filter( m -> null == ProcessorUtil.findAnnotationByType( descriptor.getElement(),
370                                                                  Constants.PROP_ANNOTATION_CLASSNAME ) &&
371                      null == ProcessorUtil.findAnnotationByType( descriptor.getElement(),
372                                                                  Constants.STATE_ANNOTATION_CLASSNAME ) )
373        .collect( Collectors.toList() );
374    for ( final ExecutableElement method : candidates )
375    {
376      final ExecutableType methodType =
377        (ExecutableType) processingEnv.getTypeUtils().asMemberOf( descriptor.getDeclaredType(), method );
378
379      if ( method.getReturnType().getKind() == TypeKind.VOID && 1 == method.getParameters().size() )
380      {
381        final String stateName =
382          ProcessorUtil.getPropertyMutatorName( method, ProcessorUtil.SENTINEL_NAME );
383        final StateValueDescriptor stateValueNamed = descriptor.findStateValueNamed( stateName );
384        if ( null != stateValueNamed && !stateValueNamed.hasSetter() )
385        {
386          stateValueNamed.setSetter( method, methodType );
387        }
388      }
389      else if ( method.getReturnType().getKind() != TypeKind.VOID && 0 == method.getParameters().size() )
390      {
391        final String stateName =
392          ProcessorUtil.getPropertyAccessorName( method, ProcessorUtil.SENTINEL_NAME );
393        final StateValueDescriptor stateValueNamed = descriptor.findStateValueNamed( stateName );
394        if ( null != stateValueNamed && !stateValueNamed.hasGetter() )
395        {
396          stateValueNamed.setGetter( method, methodType );
397        }
398      }
399    }
400  }

402  private void verifyNoUnexpectedAbstractMethod( @Nonnull final ComponentDescriptor descriptor )
403  {
404    if ( !descriptor.isArezComponent() )
405    {
406      final ExecutableElement abstractMethod =
407        getMethods( descriptor.getElement() )
408          .stream()
409          .filter( m -> m.getModifiers().contains( Modifier.ABSTRACT ) )
410          // @Props and @State methods are expected to be null
411          .filter( m -> descriptor.getProps().stream().noneMatch( p -> p.getMethod() == m ) &&
412                        descriptor.getStateValues().stream()
413                          .noneMatch( p -> p.getGetter() == m || p.getSetter() == m ) )
414          .findAny()
415          .orElse( null );
416      if ( null != abstractMethod )
417      {
418        throw new ReactProcessorException( "@ReactComponent target has an unexpected abstract method",
419                                           abstractMethod );
420      }
421    }
422  }

424  private void determinePropValidatesMethods( @Nonnull final ComponentDescriptor descriptor )
425  {
426    final List<ExecutableElement> methods =
427      getMethods( descriptor.getElement() ).stream()
428        .filter( m -> null != ProcessorUtil.findAnnotationByType( m, Constants.PROP_VALIDATE_ANNOTATION_CLASSNAME ) )
429        .collect( Collectors.toList() );
430
431    for ( final ExecutableElement method : methods )
432    {
433      final String name = derivePropValidateName( method );
434      final PropDescriptor prop = descriptor.findPropNamed( name );
435      if ( null == prop )
436      {
437        throw new ReactProcessorException( "@PropValidate target for prop named '" + name + "' has no corresponding " +
438                                           "@Prop annotated method.", method );
439      }
440      if ( 1 != method.getParameters().size() )
441      {
442        throw new ReactProcessorException( "@PropValidate target must have exactly 1 parameter", method );
443      }
444      final ExecutableType methodType =
445        (ExecutableType) processingEnv.getTypeUtils().asMemberOf( descriptor.getDeclaredType(), method );
446      if ( !processingEnv.getTypeUtils().isAssignable( methodType.getParameterTypes().get( 0 ),
447                                                       prop.getMethodType().getReturnType() ) )
448      {
449        throw new ReactProcessorException( "@PropValidate target has a parameter type that is not assignable to the " +
450                                           "return type of the associated @Prop annotated method.", method );
451      }
452      prop.setValidateMethod( method );
453    }
454  }

456  @Nonnull
457  private String derivePropValidateName( @Nonnull final Element element )
458    throws ReactProcessorException
459  {
460    final String name =
461      (String) ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
462                                                 element,
463                                                 Constants.PROP_VALIDATE_ANNOTATION_CLASSNAME,
464                                                 "name" ).getValue();
465
466    if ( ProcessorUtil.isSentinelName( name ) )
467    {
468      final String deriveName = ProcessorUtil.deriveName( element, ProcessorUtil.VALIDATE_PROP_PATTERN, name );
469      if ( null == deriveName )
470      {
471        throw new ReactProcessorException( "@PropValidate target has not specified name nor is it named according " +
472                                           "to the convention 'validate[Name]Prop'.", element );
473      }
474      return deriveName;
475    }
476    else
477    {
478      if ( !SourceVersion.isIdentifier( name ) )
479      {
480        throw new ReactProcessorException( "@PropValidate target specified an invalid name '" + name + "'. The " +
481                                           "name must be a valid java identifier.", element );
482      }
483      else if ( SourceVersion.isKeyword( name ) )
484      {
485        throw new ReactProcessorException( "@PropValidate target specified an invalid name '" + name + "'. The " +
486                                           "name must not be a java keyword.", element );
487      }
488      return name;
489    }
490  }

492  private void determineDefaultPropsMethods( @Nonnull final ComponentDescriptor descriptor )
493  {
494    final List<ExecutableElement> defaultPropsMethods =
495      getMethods( descriptor.getElement() ).stream()
496        .filter( m -> null != ProcessorUtil.findAnnotationByType( m, Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME ) )
497        .collect( Collectors.toList() );
498
499    for ( final ExecutableElement method : defaultPropsMethods )
500    {
501      final String name = derivePropDefaultName( method );
502      final PropDescriptor prop = descriptor.findPropNamed( name );
503      if ( null == prop )
504      {
505        throw new ReactProcessorException( "@PropDefault target for prop named '" + name + "' has no corresponding " +
506                                           "@Prop annotated method.", method );
507      }
508      final ExecutableType methodType = (ExecutableType) processingEnv.getTypeUtils()
509        .asMemberOf( descriptor.getDeclaredType(), method );
510      if ( !processingEnv.getTypeUtils().isAssignable( methodType.getReturnType(),
511                                                       prop.getMethodType().getReturnType() ) )
512      {
513        throw new ReactProcessorException( "@PropDefault target has a return type that is not assignable to the " +
514                                           "return type of the associated @Prop annotated method.", method );
515      }
516      MethodChecks.mustBeStaticallySubclassCallable( descriptor.getElement(),
517                                                     Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME,
518                                                     method );
519      prop.setDefaultMethod( method );
520    }
521  }

551  @Nonnull
552  private String derivePropDefaultName( @Nonnull final Element element )
553    throws ReactProcessorException
554  {
555    final String name =
556      (String) ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
557                                                 element,
558                                                 Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME,
559                                                 "name" ).getValue();
560
561    if ( ProcessorUtil.isSentinelName( name ) )
562    {
563      if ( element instanceof ExecutableElement )
564      {
565        final String deriveName = ProcessorUtil.deriveName( element, ProcessorUtil.DEFAULT_GETTER_PATTERN, name );
566        if ( null == deriveName )
567        {
568          throw new ReactProcessorException( "@PropDefault target has not specified name nor is it named according " +
569                                             "to the convention 'get[Name]Default'.", element );
570        }
571        return deriveName;
572      }
573      else
574      {
575        final String fieldName = element.getSimpleName().toString();
576        boolean matched = true;
577        final int lengthPrefix = "DEFAULT_".length();
578        final int length = fieldName.length();
579        if ( fieldName.startsWith( "DEFAULT_" ) && length > lengthPrefix )
580        {
581          for ( int i = lengthPrefix; i < length; i++ )
582          {
583            final char ch = fieldName.charAt( i );
584            if ( Character.isLowerCase( ch ) ||
585                 (
586                   ( i != lengthPrefix || !Character.isJavaIdentifierStart( ch ) ) &&
587                   ( i == lengthPrefix || !Character.isJavaIdentifierPart( ch ) )
588                 ) )
589            {
590              matched = false;
591              break;
592            }
593          }
594        }
595        else
596        {
597          matched = false;
598        }
599        if ( matched )
600        {
601          return uppercaseConstantToPascalCase( fieldName.substring( lengthPrefix ) );
602        }
603        else
604        {
605          throw new ReactProcessorException( "@PropDefault target has not specified name nor is it named according " +
606                                             "to the convention 'DEFAULT_[NAME]'.", element );
607        }
608      }
609    }
610    else
611    {
612      if ( !SourceVersion.isIdentifier( name ) )
613      {
614        throw new ReactProcessorException( "@PropDefault target specified an invalid name '" + name + "'. The " +
615                                           "name must be a valid java identifier.", element );
616      }
617      else if ( SourceVersion.isKeyword( name ) )
618      {
619        throw new ReactProcessorException( "@PropDefault target specified an invalid name '" + name + "'. The " +
620                                           "name must not be a java keyword.", element );
621      }
622      return name;
623    }
624  }

626  @Nonnull
627  private String uppercaseConstantToPascalCase( @Nonnull final String candidate )
628  {
629    final String s = candidate.toLowerCase();
630    final StringBuilder sb = new StringBuilder();
631    boolean uppercase = false;
632    for ( int i = 0; i < s.length(); i++ )
633    {
634      final char ch = s.charAt( i );
635      if ( '_' == ch )
636      {
637        uppercase = true;
638      }
639      else if ( uppercase )
640      {
641        sb.append( Character.toUpperCase( ch ) );
642        uppercase = false;
643      }
644      else
645      {
646        sb.append( ch );
647      }
648    }
649    return sb.toString();
650  }

652  private void determineCallbacks( @Nonnull final ComponentDescriptor descriptor )
653  {
654    final List<CallbackDescriptor> callbacks =
655      getMethods( descriptor.getElement() ).stream()
656        .filter( m -> null != ProcessorUtil.findAnnotationByType( m, Constants.CALLBACK_ANNOTATION_CLASSNAME ) )
657        .map( m -> createCallbackDescriptor( descriptor, m ) )
658        .collect( Collectors.toList() );
659    for ( final CallbackDescriptor callback : callbacks )
660    {
661      final ExecutableElement method = callback.getMethod();
662      final TypeElement callbackType = getCallbackType( method );
663      if ( ElementKind.INTERFACE != callbackType.getKind() )
664      {
665        throw new ReactProcessorException( "The @Callback specified an invalid type that is not an interface.",
666                                           callback.getMethod() );
667      }
668      final CallbackDescriptor matched = callbacks.stream()
669        .filter( h -> h != callback && h.getName().equals( callback.getName() ) )
670        .findAny().orElse( null );
671      if ( null != matched )
672      {
673        throw new ReactProcessorException( "The @Callback has the same name as the callback defined by " +
674                                           matched.getMethod() + ".", callback.getMethod() );
675      }
676      final CallbackDescriptor matched2 = callbacks.stream()
677        .filter( h -> h != callback &&
678                      h.getMethod().getSimpleName().equals( callback.getMethod().getSimpleName() ) )
679        .findAny().orElse( null );
680      if ( null != matched2 )
681      {
682        throw new ReactProcessorException( "The @Callback has the same method name as the callback defined " +
683                                           "by " + matched2.getMethod() + ".", callback.getMethod() );
684      }
685      final ExecutableType methodType = callback.getMethodType();
686      if ( !processingEnv.getTypeUtils().isAssignable( methodType.getReturnType(),
687                                                       callback.getCallbackMethod().getReturnType() ) )
688      {
689        throw new ReactProcessorException( "@Callback target has a return type that is not assignable to the return " +
690                                           "type defined in the callback type " + callbackType.getQualifiedName() + ".",
691                                           callback.getMethod() );
692      }
693
694      final List<? extends TypeMirror> parameters = methodType.getParameterTypes();
695      if ( !parameters.isEmpty() )
696      {
697        // Our annotated callback method has parameters so they should exactly align
698        // in count and type with the parameters in the callback method
699        final ExecutableElement target = callback.getCallbackMethod();
700        final List<? extends VariableElement> targetParameters = target.getParameters();
701        if ( targetParameters.size() != parameters.size() )
702        {
703          throw new ReactProcessorException( "The @Callback target has " + parameters.size() + " parameters " +
704                                             "but the type parameter specified a callback with method type " +
705                                             callback.getCallbackType().getQualifiedName() + " that has a " +
706                                             "callback method with " + targetParameters.size() + " parameters. The " +
707                                             "@Callback target should have zero parameters or match the number " +
708                                             "of parameter in the target method " + target.getSimpleName() + ".",
709                                             callback.getMethod() );
710        }
711        for ( int i = 0; i < parameters.size(); i++ )
712        {
713          final TypeMirror parameterType = parameters.get( i );
714          final VariableElement element = targetParameters.get( i );
715          final TypeMirror targetParameterType = element.asType();
716          final TypeMirror targetErased = processingEnv.getTypeUtils().erasure( targetParameterType );
717          final TypeMirror parameterErased = processingEnv.getTypeUtils().erasure( parameterType );
718          if ( !processingEnv.getTypeUtils().isAssignable( targetErased, parameterErased ) )
719          {
720            throw new ReactProcessorException( "The @Callback target parameter named " +
721                                               callback.getMethod().getParameters().get( i ).getSimpleName() +
722                                               " of type " + parameterType + " is not assignable from target type " +
723                                               targetParameterType + " of parameter " + element.getSimpleName() +
724                                               " in method " + callback.getCallbackType().getQualifiedName() +
725                                               "." + target.getSimpleName() + ".",
726                                               callback.getMethod() );
727          }
728        }
729      }
730    }
731
732    descriptor.setCallbacks( callbacks );
733  }

895  private PropDescriptor createPropDescriptor( @Nonnull final ComponentDescriptor descriptor,
896                                               @Nonnull final ExecutableElement method )
897  {
898    final String name = derivePropName( method );
899    final ExecutableType methodType =
900      (ExecutableType) processingEnv.getTypeUtils().asMemberOf( descriptor.getDeclaredType(), method );
901
902    verifyNoDuplicateAnnotations( method );
903    MethodChecks.mustBeAbstract( Constants.PROP_ANNOTATION_CLASSNAME, method );
904    MethodChecks.mustNotBePublic( Constants.PROP_ANNOTATION_CLASSNAME, method );
905    MethodChecks.mustNotHaveAnyParameters( Constants.PROP_ANNOTATION_CLASSNAME, method );
906    MethodChecks.mustReturnAValue( Constants.PROP_ANNOTATION_CLASSNAME, method );
907    MethodChecks.mustNotThrowAnyExceptions( Constants.PROP_ANNOTATION_CLASSNAME, method );
908    MethodChecks.mustNotBePackageAccessInDifferentPackage( descriptor.getElement(),
909                                                           Constants.PROP_ANNOTATION_CLASSNAME,
910                                                           method );
911    final TypeMirror returnType = method.getReturnType();
912    if ( "key".equals( name ) )
913    {
914      throw new ReactProcessorException( "@Prop named 'key' is invalid as the name references value used in the " +
915                                         "reconciliation process. This value can be accessed via Component.getKey()",
916                                         method );
917    }
918    else if ( "build".equals( name ) )
919    {
920      throw new ReactProcessorException( "@Prop named 'build' is invalid as it conflicts with the method named " +
921                                         "build() that is used in the generated Builder classes",
922                                         method );
923    }
924    else if ( "child".equals( name ) &&
925              ( returnType.getKind() != TypeKind.DECLARED && !"react4j.ReactNode".equals( returnType.toString() ) ) )
926    {
927      throw new ReactProcessorException( "@Prop named 'child' should be of type react4j.ReactNode", method );
928    }
929    else if ( "children".equals( name ) &&
930              ( returnType.getKind() != TypeKind.DECLARED && !"react4j.ReactNode[]".equals( returnType.toString() ) ) )
931    {
932      throw new ReactProcessorException( "@Prop named 'children' should be of type react4j.ReactNode[]", method );
933    }
934
935    if ( returnType instanceof TypeVariable )
936    {
937      final TypeVariable typeVariable = (TypeVariable) returnType;
938      final String typeVariableName = typeVariable.asElement().getSimpleName().toString();
939      List<? extends TypeParameterElement> typeParameters = method.getTypeParameters();
940      if ( typeParameters.stream().anyMatch( p -> p.getSimpleName().toString().equals( typeVariableName ) ) )
941      {
942        throw new ReactProcessorException( "@Prop named '" + name + "' is has a type variable as a return type " +
943                                           "that is declared on the method.", method );
944      }
945    }
946
947    final Element propType = processingEnv.getTypeUtils().asElement( returnType );
948    final boolean shouldUpdateOnChange = shouldUpdateOnChange( method );
949    final boolean observable = isPropObservable( descriptor, method, shouldUpdateOnChange );
950    final boolean disposable = null != propType && isPropDisposable( descriptor, method, propType );
951    if ( observable && !descriptor.isArezComponent() )
952    {
953      throw new ReactProcessorException( "@Prop named '" + name + "' is marked as observable but the host component " +
954                                         "is not a subclass of react4j.arez.ReactArezComponent", method );
955    }
956    if ( disposable && !descriptor.isArezComponent() )
957    {
958      throw new ReactProcessorException( "@Prop named '" + name + "' is marked as disposable but the host component " +
959                                         "is not a subclass of react4j.arez.ReactArezComponent", method );
960    }
961    if ( TypeName.get( returnType ).isBoxedPrimitive() &&
962         null != ProcessorUtil.findAnnotationByType( method, Constants.NONNULL_ANNOTATION_CLASSNAME ) )
963    {
964      throw new ReactProcessorException( "@Prop named '" + name + "' is a boxed primitive annotated with a " +
965                                         "@Nonnull annotation. The return type should be the primitive type.",
966                                         method );
967    }
968    return new PropDescriptor( descriptor, name, method, methodType, shouldUpdateOnChange, observable, disposable );
969  }

1047  private void parseStateValueMethod( @Nonnull final ComponentDescriptor descriptor,
1048                                      @Nonnull final ExecutableElement method,
1049                                      @Nonnull final ExecutableType methodType,
1050                                      @Nonnull final Map<String, StateValueDescriptor> values )
1051  {
1052    final String declaredName = getAnnotationParameter( method, Constants.STATE_ANNOTATION_CLASSNAME, "name" );
1053    final TypeMirror returnType = method.getReturnType();
1054
1055    final boolean setter = TypeKind.VOID == returnType.getKind();
1056
1057    verifyNoDuplicateAnnotations( method );
1058    MethodChecks.mustBeAbstract( Constants.STATE_ANNOTATION_CLASSNAME, method );
1059    MethodChecks.mustNotThrowAnyExceptions( Constants.STATE_ANNOTATION_CLASSNAME, method );
1060    MethodChecks.mustNotBePackageAccessInDifferentPackage( descriptor.getElement(),
1061                                                           Constants.STATE_ANNOTATION_CLASSNAME,
1062                                                           method );
1063
1064    final String name;
1065    if ( setter )
1066    {
1067      if ( 1 != method.getParameters().size() )
1068      {
1069        throw new ReactProcessorException( "@State target must have a single parameter", method );
1070      }
1071
1072      MethodChecks.mustNotReturnAValue( Constants.STATE_ANNOTATION_CLASSNAME, method );
1073
1074      name = ProcessorUtil.getPropertyMutatorName( method, declaredName );
1075    }
1076    else
1077    {
1078      MethodChecks.mustNotHaveAnyParameters( Constants.STATE_ANNOTATION_CLASSNAME, method );
1079      MethodChecks.mustReturnAValue( Constants.STATE_ANNOTATION_CLASSNAME, method );
1080
1081      name = ProcessorUtil.getPropertyAccessorName( method, declaredName );
1082    }
1083
1084    if ( !SourceVersion.isIdentifier( name ) )
1085    {
1086      throw new ReactProcessorException( "@State target specified an invalid name '" + name + "'. The " +
1087                                         "name must be a valid java identifier.", method );
1088    }
1089    else if ( SourceVersion.isKeyword( name ) )
1090    {
1091      throw new ReactProcessorException( "@State target specified an invalid name '" + name + "'. The " +
1092                                         "name must not be a java keyword.", method );
1093    }
1094
1095    //Find or create descriptor
1096    final StateValueDescriptor stateValueDescriptor = values.computeIfAbsent( name, StateValueDescriptor::new );
1097    if ( setter )
1098    {
1099      if ( stateValueDescriptor.hasSetter() )
1100      {
1101        throw new ReactProcessorException( "@State target defines duplicate state property mutator for " +
1102                                           "property named '" + name + "'. Existing mutator: " +
1103                                           stateValueDescriptor.getSetter().getSimpleName(), method );
1104      }
1105      stateValueDescriptor.setSetter( method, methodType );
1106    }
1107    else
1108    {
1109      if ( stateValueDescriptor.hasGetter() )
1110      {
1111        throw new ReactProcessorException( "@State target defines duplicate state property accessor for " +
1112                                           "property named '" + name + "'. Existing accessor: " +
1113                                           stateValueDescriptor.getGetter().getSimpleName(), method );
1114      }
1115      stateValueDescriptor.setGetter( method, methodType );
1116    }

1274     * Our code should just gracefully handle this and perform none of the arez specific checks or generation.
1275     */
1276    @Nullable
1277    final TypeElement arezComponentType =
1278      processingEnv.getElementUtils().getTypeElement( "react4j.arez.ReactArezComponent" );
1279    @Nullable
1280    final TypeMirror rawArezComponentType =
1281      null == arezComponentType ? null : processingEnv.getTypeUtils().erasure( arezComponentType.asType() );
1282
1283    final TypeMirror type = descriptor.getDeclaredType();
1284
1285    final boolean isComponent = processingEnv.getTypeUtils().isSubtype( type, rawComponentType );
1286    final boolean isArezComponent =
1287      null != rawArezComponentType && processingEnv.getTypeUtils().isSubtype( type, rawArezComponentType );
1288
1289    if ( !isComponent )
1290    {
1291      throw new ReactProcessorException( "@ReactComponent target must be a subclass of react4j.Component",
1292                                         typeElement );
1293    }
1294    else if ( isArezComponent )
1295    {
1296      final AnnotationMirror arezAnnotation = typeElement.getAnnotationMirrors().stream().
1297        filter( m -> m.getAnnotationType().toString().equals( "arez.annotations.ArezComponent" ) ).
1298        findAny().orElse( null );
1299      if ( null != arezAnnotation )
1300      {
1301        throw new ReactProcessorException( "@ReactComponent target extends react4j.arez.ReactArezComponent and should " +
1302                                           "not be annotated with arez.annotations.ArezComponent as " +
1303                                           "React4j will add annotation", typeElement );
1304      }
1305    }
1306
1307    final boolean needsInjection = isInjectionRequired( typeElement );
1308    final boolean isDaggerPresent = needsInjection && isDaggerRequired( typeElement );
1309
1310    if ( isDaggerPresent && !descriptor.getDeclaredType().getTypeArguments().isEmpty() )
1311    {
1312      throw new ReactProcessorException( "@ReactComponent target has enabled dagger injection and the class " +
1313                                         "has type argument but type arguments on a component are not compatible " +
1314                                         "with dagger injected components", typeElement );
1315    }
1316
1317    descriptor.setNeedsInjection( needsInjection );
1318    descriptor.setNeedsDaggerIntegration( isDaggerPresent );
1319    descriptor.setArezComponent( isArezComponent );
1320
1321    if ( isArezComponent )
1322    {
1323      ensureComputedMatchesExpectations( typeElement );
1324      ensureMemoizeMatchesExpectations( typeElement );
1325      final boolean runArezScheduler =
1326        hasAnyArezScheduledObserverMethods( typeElement ) ||
1327        hasAnyKeepAliveComputedMethods( typeElement ) ||
1328        hasAnyDependencyMethods( typeElement );
1329      descriptor.setRunArezScheduler( runArezScheduler );
1330
1331      descriptor.setComputedMethods( getComputedMethods( typeElement ) );
1332      descriptor.setMemoizeMethods( getMemoizeMethods( typeElement ) );
1333    }
1334    else
1335    {
1336      for ( final ExecutableElement method : ProcessorUtil.getMethods( typeElement, processingEnv.getTypeUtils() ) )
1337      {
1338        for ( final AnnotationMirror mirror : method.getAnnotationMirrors() )
1339        {
1340          final String classname = mirror.getAnnotationType().toString();
1341          if ( isArezAnnotation( classname ) )
1342          {
1343            throw new ReactProcessorException( "@ReactComponent target has a method '" + method.getSimpleName() +
1344                                               "' with an arez annotation '" + classname + "' but is not an " +
1345                                               "arez component.", method );
1346          }
1347        }
1348      }
1349      for ( final VariableElement element : ProcessorUtil.getFieldElements( typeElement ) )
1350      {
1351        for ( final AnnotationMirror mirror : element.getAnnotationMirrors() )
1352        {
1353          final String classname = mirror.getAnnotationType().toString();
1354          if ( isArezAnnotation( classname ) )
1355          {
1356            throw new ReactProcessorException( "@ReactComponent target has a field '" + element.getSimpleName() +
1357                                               "' with an arez annotation '" + classname + "' but is not an " +
1358                                               "arez component.", element );
1359          }
1360        }
1361      }
1362    }
1363  }

1365  private void ensureComputedMatchesExpectations( @Nonnull final TypeElement typeElement )
1366  {
1367    final TypeElement computedElement =
1368      processingEnv.getElementUtils().getTypeElement( Constants.COMPUTED_ANNOTATION_CLASSNAME );
1369    final Set<String> parameters = computedElement.getEnclosedElements()
1370      .stream()
1371      .map( e -> e.getSimpleName().toString() )
1372      .collect( Collectors.toSet() );
1373    if ( !( parameters.contains( "name" ) &&
1374            parameters.contains( "priority" ) &&
1375            parameters.contains( "keepAlive" ) &&
1376            parameters.contains( "observeLowerPriorityDependencies" ) &&
1377            parameters.contains( "depType" ) &&
1378            parameters.contains( "requireEnvironment" ) &&
1379            6 == parameters.size() ) )
1380    {
1381      throw new ReactProcessorException( "The @" + Constants.COMPUTED_ANNOTATION_CLASSNAME + " annotation was " +
1382                                         "expected to have the parameters name, priority, keepAlive, " +
1383                                         "depType, requireEnvironment and observeLowerPriorityDependencies but has " +
1384                                         parameters + ". The react4j annotation processor needs to be updated to " +
1385                                         "handle the change in parameters.", typeElement );
1386
1387    }

1617  private void verifyNoDuplicateAnnotations( @Nonnull final ExecutableElement method )
1618    throws ReactProcessorException
1619  {
1620    final String[] annotationTypes =
1621      new String[]{ Constants.CALLBACK_ANNOTATION_CLASSNAME,
1622                    Constants.STATE_ANNOTATION_CLASSNAME,
1623                    Constants.PROP_ANNOTATION_CLASSNAME };
1624    for ( int i = 0; i < annotationTypes.length; i++ )
1625    {
1626      final String type1 = annotationTypes[ i ];
1627      final Object annotation1 = ProcessorUtil.findAnnotationByType( method, type1 );
1628      if ( null != annotation1 )
1629      {
1630        for ( int j = i + 1; j < annotationTypes.length; j++ )
1631        {
1632          final String type2 = annotationTypes[ j ];
1633          final Object annotation2 = ProcessorUtil.findAnnotationByType( method, type2 );
1634          if ( null != annotation2 )
1635          {
1636            final String message =
1637              "Method can not be annotated with both @" + ProcessorUtil.toSimpleName( type1 ) +
1638              " and @" + ProcessorUtil.toSimpleName( type2 );
1639            throw new ReactProcessorException( message, method );
1640          }
1641        }
1642      }
1643    }
1644  }
