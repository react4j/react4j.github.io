047/**
048 * Annotation processor that analyzes React annotated source and generates models from the annotations.
049 */
050@SuppressWarnings( "Duplicates" )
051@AutoService( Processor.class )
052@SupportedAnnotationTypes( { "react4j.annotations.*" } )
053@SupportedSourceVersion( SourceVersion.RELEASE_8 )
054public final class ReactProcessor
055  extends AbstractProcessor
056{
057  /**
058   * Cache of render method method as defined by Component.
059   */
060  private ExecutableElement _componentRenderMethod;
061  /**
062   * Elements that were unresolved and have been deferred to a later compilation cycle.
063   */
064  @Nonnull
065  private HashSet<TypeElement> _deferred = new HashSet<>();
066  private int _invalidTypeCount;
067  private RoundEnvironment _env;
068
069  @Override
070  public boolean process( final Set<? extends TypeElement> annotations, final RoundEnvironment env )
071  {
072    // Clear method caches to avoid potential inter-run problems
073    _componentRenderMethod = null;
074
075    _env = env;
076
077    final TypeElement annotation =
078      processingEnv.getElementUtils().getTypeElement( Constants.REACT_COMPONENT_ANNOTATION_CLASSNAME );
079    final Set<? extends Element> elements = env.getElementsAnnotatedWith( annotation );
080    final Collection<Element> elementsToProcess = getElementsToProcess( elements );
081    processElements( elementsToProcess );
082    if ( env.getRootElements().isEmpty() && !_deferred.isEmpty() )
083    {
084      _deferred.forEach( this::processingErrorMessage );
085      _deferred.clear();
086    }
087    if ( _env.processingOver() )
088    {
089      if ( 0 != _invalidTypeCount )
090      {
091        processingEnv
092          .getMessager()
093          .printMessage( ERROR, "ReactProcessor failed to process " + _invalidTypeCount +
094                                " types. See earlier warnings for further details." );
095      }
096      _invalidTypeCount = 0;
097    }
098    _env = null;
099    return true;
100  }
101
102  private void processingErrorMessage( @Nonnull final TypeElement target )
103  {
104    reportError( "ReactProcessor unable to process " + target.getQualifiedName() +
105                 " because not all of its dependencies could be resolved. Check for " +
106                 "compilation errors or a circular dependency with generated code.",
107                 target );
108  }
109
110  private void reportError( @Nonnull final String message, @Nullable final Element element )
111  {
112    _invalidTypeCount++;
113    if ( _env.errorRaised() || _env.processingOver() )
114    {
115      processingEnv.getMessager().printMessage( ERROR, message, element );
116    }
117    else
118    {
119      processingEnv.getMessager().printMessage( MANDATORY_WARNING, message, element );
120    }
121  }
122
123  @Nonnull
124  private Collection<Element> getElementsToProcess( @Nonnull final Set<? extends Element> elements )
125  {
126    final List<TypeElement> deferred = _deferred
127      .stream()
128      .map( e -> processingEnv.getElementUtils().getTypeElement( e.getQualifiedName() ) )
129      .collect( Collectors.toList() );
130    _deferred = new HashSet<>();
131
132    final ArrayList<Element> elementsToProcess = new ArrayList<>();
133    collectElementsToProcess( elements, elementsToProcess );
134    collectElementsToProcess( deferred, elementsToProcess );
135    return elementsToProcess;
136  }
137
138  private void collectElementsToProcess( @Nonnull final Collection<? extends Element> elements,
139                                         @Nonnull final ArrayList<Element> elementsToProcess )
140  {
141    for ( final Element element : elements )
142    {
143      if ( SuperficialValidation.validateElement( element ) )
144      {
145        elementsToProcess.add( element );
146      }
147      else
148      {
149        _deferred.add( (TypeElement) element );
150      }
151    }
152  }
153
154  private void processElements( @Nonnull final Collection<? extends Element> elements )
155  {
156    for ( final Element element : elements )
157    {
158      try
159      {
160        process( (TypeElement) element );
161      }
162      catch ( final IOException ioe )
163      {
164        reportError( ioe.getMessage(), element );
165      }
166      catch ( final ReactProcessorException e )
167      {
168        final Element errorLocation = e.getElement();
169        final Element outerElement = getOuterElement( errorLocation );
170        if ( !_env.getRootElements().contains( outerElement ) )
171        {
172          final String location;
173          if ( errorLocation instanceof ExecutableElement )
174          {
175            final ExecutableElement executableElement = (ExecutableElement) errorLocation;
176            final TypeElement typeElement = (TypeElement) executableElement.getEnclosingElement();
177            location = typeElement.getQualifiedName() + "." + executableElement.getSimpleName();
178          }
179          else if ( errorLocation instanceof VariableElement )
180          {
181            final VariableElement variableElement = (VariableElement) errorLocation;
182            final TypeElement typeElement = (TypeElement) variableElement.getEnclosingElement();
183            location = typeElement.getQualifiedName() + "." + variableElement.getSimpleName();
184          }
185          else
186          {
187            assert errorLocation instanceof TypeElement;
188            final TypeElement typeElement = (TypeElement) errorLocation;
189            location = typeElement.getQualifiedName().toString();
190          }
191
192          final StringWriter sw = new StringWriter();
193          processingEnv.getElementUtils().printElements( sw, errorLocation );
194          sw.flush();
195
196          final String message =
197            "An error was generated processing the element " + element.getSimpleName() +
198            " but the error was triggered by code not currently being compiled but inherited or " +
199            "implemented by the element and may not be highlighted by your tooling or IDE. The " +
200            "error occurred at " + location + " and may look like:\n" + sw.toString();
201
202          reportError( e.getMessage(), element );
203          reportError( message, null );
204        }
205        reportError( e.getMessage(), e.getElement() );
206      }
207      catch ( final Throwable e )
208      {
209        final StringWriter sw = new StringWriter();
210        e.printStackTrace( new PrintWriter( sw ) );
211        sw.flush();
212
213        final String message =
214          "Unexpected error running the " + getClass().getName() + " processor. This has " +
215          "resulted in a failure to process the code and has left the compiler in an invalid " +
216          "state. Please report the failure to the developers so that it can be fixed.\n" +
217          " Report the error at: https://github.com/react4j/react4j/issues\n" +
218          "\n\n" +
219          sw.toString();
220        reportError( message, element );
221      }
222    }
223  }

225  /**
226   * Return the outer enclosing element.
227   * This is either the top-level class, interface, enum, etc within a package.
228   * This helps identify the top level compilation units.
229   */
230  @Nonnull
231  private Element getOuterElement( @Nonnull final Element element )
232  {
233    Element result = element;
234    while ( !( result.getEnclosingElement() instanceof PackageElement ) )
235    {
236      result = result.getEnclosingElement();
237    }
238    return result;
239  }

241  private void process( @Nonnull final TypeElement element )
242    throws IOException, ReactProcessorException
243  {
244    final ComponentDescriptor descriptor = parse( element );
245    emitTypeSpec( descriptor.getPackageName(), Generator.buildEnhancedComponent( descriptor ) );
246    emitTypeSpec( descriptor.getPackageName(), Generator.buildComponentBuilder( descriptor ) );
247    if ( descriptor.needsInjection() )
248    {
249      emitTypeSpec( descriptor.getPackageName(), Generator.buildDaggerComponentExtension( descriptor ) );
250    }
251  }
252
253  private void emitTypeSpec( @Nonnull final String packageName, @Nonnull final TypeSpec typeSpec )
254    throws IOException
255  {
256    JavaFile.builder( packageName, typeSpec ).
257      skipJavaLangImports( true ).
258      build().
259      writeTo( processingEnv.getFiler() );
260  }

262  /**
263   * Return true if there is any injection points that are not through the constructor.
264   */
265  private boolean nonConstructorInjections( @Nonnull final TypeElement typeElement )
266  {
267    return getMethods( typeElement )
268             .stream()
269             .anyMatch( e -> ProcessorUtil.hasAnnotationOfType( e, Constants.INJECT_ANNOTATION_CLASSNAME ) ) ||
270           ProcessorUtil.getFieldElements( typeElement )
271             .stream()
272             .anyMatch( e -> ProcessorUtil.hasAnnotationOfType( e, Constants.INJECT_ANNOTATION_CLASSNAME ) );
273  }

275  @Nonnull
276  private ComponentDescriptor parse( @Nonnull final TypeElement typeElement )
277  {
278    final String name = deriveComponentName( typeElement );
279    final PackageElement packageElement = processingEnv.getElementUtils().getPackageOf( typeElement );
280    final ComponentType type = extractComponentType( typeElement );
281    final boolean nonConstructorInjections = nonConstructorInjections( typeElement );
282    final ComponentDescriptor descriptor =
283      new ComponentDescriptor( processingEnv.getElementUtils(),
284                               processingEnv.getSourceVersion(),
285                               name,
286                               packageElement,
287                               typeElement,
288                               type,
289                               nonConstructorInjections );
290
291    determineComponentCapabilities( descriptor, typeElement );
292    determineRenderMethod( typeElement, descriptor );
293    determineProps( descriptor );
294    determinePropValidatesMethods( descriptor );
295    determineOnPropChangeMethods( descriptor );
296    determineDefaultPropsMethods( descriptor );
297    determineDefaultPropsFields( descriptor );
298    determinePreUpdateMethod( typeElement, descriptor );
299    determinePostRenderMethod( typeElement, descriptor );
300    determinePostUpdateMethod( typeElement, descriptor );
301    determinePostMountMethod( typeElement, descriptor );
302    determineOnErrorMethod( typeElement, descriptor );
303
304    for ( final PropDescriptor prop : descriptor.getProps() )
305    {
306      if ( !isPropRequired( prop ) )
307      {
308        prop.markAsOptional();
309      }
310    }
311
312    /*
313     * Sorting must occur after @PropDefault has been processed to ensure the sorting
314     * correctly sorts optional props after required props.
315     */
316    descriptor.sortProps();
317
318    verifyPropsNotAnnotatedWithArezAnnotations( descriptor );
319    verifyPropsNotCollectionOfArezComponents( descriptor );
320
321    return descriptor;
322  }

324  private void verifyPropsNotCollectionOfArezComponents( @Nonnull final ComponentDescriptor descriptor )
325  {
326    for ( final PropDescriptor prop : descriptor.getProps() )
327    {
328      final ExecutableElement method = prop.getMethod();
329      final TypeMirror returnType = method.getReturnType();
330      if ( TypeKind.DECLARED == returnType.getKind() )
331      {
332        final DeclaredType declaredType = (DeclaredType) returnType;
333        final List<? extends TypeMirror> typeArguments = declaredType.getTypeArguments();
334        if ( isCollection( declaredType ) )
335        {
336          if ( 1 == typeArguments.size() && isArezComponent( typeArguments.get( 0 ) ) )
337          {
338            throw new ReactProcessorException( "@Prop target is a collection that contains Arez components. " +
339                                               "This is not a safe pattern when the arez components can be disposed.",
340                                               method );
341          }
342        }
343        else if ( isMap( declaredType ) )
344        {
345          if ( 2 == typeArguments.size() &&
346               ( isArezComponent( typeArguments.get( 0 ) ) ||
347                 isArezComponent( typeArguments.get( 1 ) ) ) )
348          {
349            throw new ReactProcessorException( "@Prop target is a collection that contains Arez components. " +
350                                               "This is not a safe pattern when the arez components can be disposed.",
351                                               method );
352          }
353        }
354      }
355      else if ( TypeKind.ARRAY == returnType.getKind() )
356      {
357        final ArrayType arrayType = (ArrayType) returnType;
358        if ( isArezComponent( arrayType.getComponentType() ) )
359        {
360          throw new ReactProcessorException( "@Prop target is an array that contains Arez components. " +
361                                             "This is not a safe pattern when the arez components can be disposed.",
362                                             method );
363        }
364      }
365    }
366  }

368  private boolean isCollection( @Nonnull final DeclaredType declaredType )
369  {
370    final TypeElement returnType = (TypeElement) processingEnv.getTypeUtils().asElement( declaredType );
371    final String classname = returnType.getQualifiedName().toString();
372    /*
373     * For the time being lets just list out a bunch of collections. We can ge more specific when/if
374     * it is ever required
375     */
376    return Collection.class.getName().equals( classname ) ||
377           Set.class.getName().equals( classname ) ||
378           List.class.getName().equals( classname ) ||
379           HashSet.class.getName().equals( classname ) ||
380           ArrayList.class.getName().equals( classname );
381  }
382
383  private boolean isMap( @Nonnull final DeclaredType declaredType )
384  {
385    final TypeElement returnType = (TypeElement) processingEnv.getTypeUtils().asElement( declaredType );
386    final String classname = returnType.getQualifiedName().toString();
387    /*
388     * For the time being lets just list out a bunch of collections. We can ge more specific when/if
389     * it is ever required
390     */
391    return Map.class.getName().equals( classname ) || HashMap.class.getName().equals( classname );
392  }

394  private boolean isArezComponent( @Nonnull final TypeMirror typeMirror )
395  {
396    return typeMirror instanceof DeclaredType &&
397           processingEnv.getTypeUtils()
398             .asElement( typeMirror )
399             .getAnnotationMirrors()
400             .stream()
401             .anyMatch( a -> a.getAnnotationType().toString().equals( Constants.AREZ_COMPONENT_ANNOTATION_CLASSNAME ) );
402  }

404  private void verifyPropsNotAnnotatedWithArezAnnotations( @Nonnull final ComponentDescriptor descriptor )
405  {
406    for ( final PropDescriptor prop : descriptor.getProps() )
407    {
408      final ExecutableElement method = prop.getMethod();
409      for ( final AnnotationMirror mirror : method.getAnnotationMirrors() )
410      {
411        final String classname = mirror.getAnnotationType().toString();
412        if ( isArezAnnotation( classname ) )
413        {
414          throw new ReactProcessorException( "@Prop target must not be annotated with any arez annotations but " +
415                                             "is annotated by '" + classname + "'.", method );
416        }
417      }
418    }
419  }

421  private void determineOnPropChangeMethods( @Nonnull final ComponentDescriptor descriptor )
422  {
423    final List<ExecutableElement> methods =
424      getMethods( descriptor.getElement() ).stream()
425        .filter( m -> ProcessorUtil.hasAnnotationOfType( m, Constants.ON_PROP_CHANGE_ANNOTATION_CLASSNAME ) )
426        .collect( Collectors.toList() );
427
428    final ArrayList<OnPropChangeDescriptor> onPropChangeDescriptors = new ArrayList<>();
429    for ( final ExecutableElement method : methods )
430    {
431      final VariableElement phase = (VariableElement)
432        ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
433                                          method,
434                                          Constants.ON_PROP_CHANGE_ANNOTATION_CLASSNAME,
435                                          "phase" ).getValue();
436      final boolean preUpdate = phase.getSimpleName().toString().equals( "PRE" );
437
438      final List<? extends VariableElement> parameters = method.getParameters();
439      final ExecutableType methodType = resolveMethodType( descriptor, method );
440      final List<? extends TypeMirror> parameterTypes = methodType.getParameterTypes();
441
442      MethodChecks.mustBeSubclassCallable( descriptor.getElement(),
443                                           Constants.ON_PROP_CHANGE_ANNOTATION_CLASSNAME,
444                                           method );
445      MethodChecks.mustNotThrowAnyExceptions( Constants.ON_PROP_CHANGE_ANNOTATION_CLASSNAME, method );
446      MethodChecks.mustNotReturnAValue( Constants.ON_PROP_CHANGE_ANNOTATION_CLASSNAME, method );
447      MethodChecks.mustNotBePublic( Constants.ON_PROP_CHANGE_ANNOTATION_CLASSNAME, method );
448
449      final int parameterCount = parameters.size();
450      if ( 0 == parameterCount )
451      {
452        throw new ReactProcessorException( "@OnPropChange target must have at least 1 parameter.", method );
453      }
454      final ArrayList<PropDescriptor> propDescriptors = new ArrayList<>( parameterCount );
455      for ( int i = 0; i < parameterCount; i++ )
456      {
457        final VariableElement parameter = parameters.get( i );
458        final String name = deriveOnPropChangeName( descriptor, parameter );
459        final PropDescriptor prop = descriptor.findPropNamed( name );
460        if ( null == prop )
461        {
462          throw new ReactProcessorException( "@OnPropChange target has a parameter named '" +
463                                             parameter.getSimpleName() + "' and the parameter is associated with a " +
464                                             "@Prop named '" + name + "' but there is no corresponding @Prop " +
465                                             "annotated method.", parameter );
466        }
467        final Types typeUtils = processingEnv.getTypeUtils();
468        if ( !typeUtils.isAssignable( parameterTypes.get( i ), prop.getMethodType().getReturnType() ) )
469        {
470          throw new ReactProcessorException( "@OnPropChange target has a parameter named '" +
471                                             parameter.getSimpleName() + "' and the parameter type is not " +
472                                             "assignable to the return type of the associated @Prop annotated method.",
473                                             method );
474        }
475        final boolean mismatchedNullability =
476          (
477            ProcessorUtil.hasAnnotationOfType( parameter, Constants.NONNULL_ANNOTATION_CLASSNAME ) &&
478            ProcessorUtil.hasAnnotationOfType( prop.getMethod(), Constants.NULLABLE_ANNOTATION_CLASSNAME )
479          ) ||
480          (
481            ProcessorUtil.hasAnnotationOfType( parameter, Constants.NULLABLE_ANNOTATION_CLASSNAME ) &&
482            ProcessorUtil.hasAnnotationOfType( prop.getMethod(), Constants.NONNULL_ANNOTATION_CLASSNAME ) );
483
484        if ( mismatchedNullability )
485        {
486          throw new ReactProcessorException( "@OnPropChange target has a parameter named '" +
487                                             parameter.getSimpleName() + "' that has a nullability annotation " +
488                                             "incompatible with the associated @Prop method named " +
489                                             method.getSimpleName(), method );
490        }
491        if ( prop.isImmutable() )
492        {
493          throw new ReactProcessorException( "@OnPropChange target has a parameter named '" +
494                                             parameter.getSimpleName() + "' that is associated with a @Prop " +
495                                             "annotated method and the prop is specified as immutable.", method );
496        }
497        propDescriptors.add( prop );
498      }
499      onPropChangeDescriptors.add( new OnPropChangeDescriptor( method, propDescriptors, preUpdate ) );
500    }
501    descriptor.setOnPropChangeDescriptors( onPropChangeDescriptors );
502  }

504  @Nonnull
505  private String deriveOnPropChangeName( @Nonnull final ComponentDescriptor descriptor,
506                                         @Nonnull final VariableElement parameter )
507  {
508    final AnnotationValue value =
509      ProcessorUtil.findAnnotationValue( descriptor.getElements(),
510                                         parameter,
511                                         Constants.PROP_REF_ANNOTATION_CLASSNAME,
512                                         "value" );
513
514    if ( null != value )
515    {
516      return (String) value.getValue();
517    }
518    else
519    {
520      final String parameterName = parameter.getSimpleName().toString();
521      if ( ProcessorUtil.LAST_PROP_PATTERN.matcher( parameterName ).matches() ||
522           ProcessorUtil.PREV_PROP_PATTERN.matcher( parameterName ).matches() )
523      {
524        return Character.toLowerCase( parameterName.charAt( 4 ) ) + parameterName.substring( 5 );
525      }
526      else if ( ProcessorUtil.PROP_PATTERN.matcher( parameterName ).matches() )
527      {
528        return parameterName;
529      }
530      else
531      {
532        throw new ReactProcessorException( "@OnPropChange target has a parameter named '" + parameterName +
533                                           "' is not explicitly associated with a prop using @PropRef nor does it " +
534                                           "follow required naming conventions 'prev[MyProp]', 'last[MyProp]' or " +
535                                           "'[myProp]'.", parameter );
536      }
537    }
538  }

540  private void determinePropValidatesMethods( @Nonnull final ComponentDescriptor descriptor )
541  {
542    final List<ExecutableElement> methods =
543      getMethods( descriptor.getElement() ).stream()
544        .filter( m -> ProcessorUtil.hasAnnotationOfType( m, Constants.PROP_VALIDATE_ANNOTATION_CLASSNAME ) )
545        .collect( Collectors.toList() );
546
547    for ( final ExecutableElement method : methods )
548    {
549      final String name = derivePropValidateName( method );
550      final PropDescriptor prop = descriptor.findPropNamed( name );
551      if ( null == prop )
552      {
553        throw new ReactProcessorException( "@PropValidate target for prop named '" + name + "' has no corresponding " +
554                                           "@Prop annotated method.", method );
555      }
556      if ( 1 != method.getParameters().size() )
557      {
558        throw new ReactProcessorException( "@PropValidate target must have exactly 1 parameter", method );
559      }
560      final ExecutableType methodType = resolveMethodType( descriptor, method );
561      if ( !processingEnv.getTypeUtils().isAssignable( methodType.getParameterTypes().get( 0 ),
562                                                       prop.getMethodType().getReturnType() ) )
563      {
564        throw new ReactProcessorException( "@PropValidate target has a parameter type that is not assignable to the " +
565                                           "return type of the associated @Prop annotated method.", method );
566      }
567      prop.setValidateMethod( method );
568    }
569  }

571  @Nonnull
572  private String derivePropValidateName( @Nonnull final Element element )
573    throws ReactProcessorException
574  {
575    final String name =
576      (String) ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
577                                                 element,
578                                                 Constants.PROP_VALIDATE_ANNOTATION_CLASSNAME,
579                                                 "name" ).getValue();
580
581    if ( ProcessorUtil.isSentinelName( name ) )
582    {
583      final String deriveName = ProcessorUtil.deriveName( element, ProcessorUtil.VALIDATE_PROP_PATTERN, name );
584      if ( null == deriveName )
585      {
586        throw new ReactProcessorException( "@PropValidate target has not specified name nor is it named according " +
587                                           "to the convention 'validate[Name]Prop'.", element );
588      }
589      return deriveName;
590    }
591    else
592    {
593      if ( !SourceVersion.isIdentifier( name ) )
594      {
595        throw new ReactProcessorException( "@PropValidate target specified an invalid name '" + name + "'. The " +
596                                           "name must be a valid java identifier.", element );
597      }
598      else if ( SourceVersion.isKeyword( name ) )
599      {
600        throw new ReactProcessorException( "@PropValidate target specified an invalid name '" + name + "'. The " +
601                                           "name must not be a java keyword.", element );
602      }
603      return name;
604    }
605  }

607  private void determineDefaultPropsMethods( @Nonnull final ComponentDescriptor descriptor )
608  {
609    final List<ExecutableElement> defaultPropsMethods =
610      getMethods( descriptor.getElement() ).stream()
611        .filter( m -> ProcessorUtil.hasAnnotationOfType( m, Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME ) )
612        .collect( Collectors.toList() );
613
614    for ( final ExecutableElement method : defaultPropsMethods )
615    {
616      final String name = derivePropDefaultName( method );
617      final PropDescriptor prop = descriptor.findPropNamed( name );
618      if ( null == prop )
619      {
620        throw new ReactProcessorException( "@PropDefault target for prop named '" + name + "' has no corresponding " +
621                                           "@Prop annotated method.", method );
622      }
623      final ExecutableType methodType = resolveMethodType( descriptor, method );
624      if ( !processingEnv.getTypeUtils().isAssignable( methodType.getReturnType(),
625                                                       prop.getMethodType().getReturnType() ) )
626      {
627        throw new ReactProcessorException( "@PropDefault target has a return type that is not assignable to the " +
628                                           "return type of the associated @Prop annotated method.", method );
629      }
630      MethodChecks.mustBeStaticallySubclassCallable( descriptor.getElement(),
631                                                     Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME,
632                                                     method );
633      prop.setDefaultMethod( method );
634    }
635  }

637  private void determineDefaultPropsFields( @Nonnull final ComponentDescriptor descriptor )
638  {
639    final List<VariableElement> defaultPropsFields =
640      ProcessorUtil.getFieldElements( descriptor.getElement() ).stream()
641        .filter( m -> ProcessorUtil.hasAnnotationOfType( m, Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME ) )
642        .collect( Collectors.toList() );
643
644    for ( final VariableElement field : defaultPropsFields )
645    {
646      final String name = derivePropDefaultName( field );
647      final PropDescriptor prop = descriptor.findPropNamed( name );
648      if ( null == prop )
649      {
650        throw new ReactProcessorException( "@PropDefault target for prop named '" + name + "' has no corresponding " +
651                                           "@Prop annotated method.", field );
652      }
653      if ( !processingEnv.getTypeUtils().isAssignable( field.asType(), prop.getMethodType().getReturnType() ) )
654      {
655        throw new ReactProcessorException( "@PropDefault target has a type that is not assignable to the " +
656                                           "return type of the associated @Prop annotated method.", field );
657      }
658      MethodChecks.mustBeStaticallySubclassCallable( descriptor.getElement(),
659                                                     Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME,
660                                                     field );
661      prop.setDefaultField( field );
662    }

665  @Nonnull
666  private String derivePropDefaultName( @Nonnull final Element element )
667    throws ReactProcessorException
668  {
669    final String name =
670      (String) ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
671                                                 element,
672                                                 Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME,
673                                                 "name" ).getValue();
674
675    if ( ProcessorUtil.isSentinelName( name ) )
676    {
677      if ( element instanceof ExecutableElement )
678      {
679        final String deriveName = ProcessorUtil.deriveName( element, ProcessorUtil.DEFAULT_GETTER_PATTERN, name );
680        if ( null == deriveName )
681        {
682          throw new ReactProcessorException( "@PropDefault target has not specified name nor is it named according " +
683                                             "to the convention 'get[Name]Default'.", element );
684        }
685        return deriveName;
686      }
687      else
688      {
689        final String fieldName = element.getSimpleName().toString();
690        boolean matched = true;
691        final int lengthPrefix = "DEFAULT_".length();
692        final int length = fieldName.length();
693        if ( fieldName.startsWith( "DEFAULT_" ) && length > lengthPrefix )
694        {
695          for ( int i = lengthPrefix; i < length; i++ )
696          {
697            final char ch = fieldName.charAt( i );
698            if ( Character.isLowerCase( ch ) ||
699                 (
700                   ( i != lengthPrefix || !Character.isJavaIdentifierStart( ch ) ) &&
701                   ( i == lengthPrefix || !Character.isJavaIdentifierPart( ch ) )
702                 ) )
703            {
704              matched = false;
705              break;
706            }
707          }
708        }
709        else
710        {
711          matched = false;
712        }
713        if ( matched )
714        {
715          return uppercaseConstantToPascalCase( fieldName.substring( lengthPrefix ) );
716        }
717        else
718        {
719          throw new ReactProcessorException( "@PropDefault target has not specified name nor is it named according " +
720                                             "to the convention 'DEFAULT_[NAME]'.", element );
721        }
722      }
723    }
724    else
725    {
726      if ( !SourceVersion.isIdentifier( name ) )
727      {
728        throw new ReactProcessorException( "@PropDefault target specified an invalid name '" + name + "'. The " +
729                                           "name must be a valid java identifier.", element );
730      }
731      else if ( SourceVersion.isKeyword( name ) )
732      {
733        throw new ReactProcessorException( "@PropDefault target specified an invalid name '" + name + "'. The " +
734                                           "name must not be a java keyword.", element );
735      }
736      return name;
737    }
738  }

740  @Nonnull
741  private String uppercaseConstantToPascalCase( @Nonnull final String candidate )
742  {
743    final String s = candidate.toLowerCase();
744    final StringBuilder sb = new StringBuilder();
745    boolean uppercase = false;
746    for ( int i = 0; i < s.length(); i++ )
747    {
748      final char ch = s.charAt( i );
749      if ( '_' == ch )
750      {
751        uppercase = true;
752      }
753      else if ( uppercase )
754      {
755        sb.append( Character.toUpperCase( ch ) );
756        uppercase = false;
757      }
758      else
759      {
760        sb.append( ch );
761      }
762    }
763    return sb.toString();
764  }

766  private void determineProps( @Nonnull final ComponentDescriptor descriptor )
767  {
768    final List<PropDescriptor> props =
769      getMethods( descriptor.getElement() ).stream()
770        .filter( m -> ProcessorUtil.hasAnnotationOfType( m, Constants.PROP_ANNOTATION_CLASSNAME ) )

779    {
780      throw new ReactProcessorException( "Multiple candidate children @Prop annotated methods: " +
781                                         childrenProp.getMethod().getSimpleName() + " and " +
782                                         childProp.getMethod().getSimpleName(),
783                                         childrenProp.getMethod() );
784    }
785
786    descriptor.setProps( props );

809  @Nonnull
810  private PropDescriptor createPropDescriptor( @Nonnull final ComponentDescriptor descriptor,
811                                               @Nonnull final ExecutableElement method )
812  {
813    final String name = derivePropName( method );
814    final ExecutableType methodType = resolveMethodType( descriptor, method );
815
816    verifyNoDuplicateAnnotations( method );
817    MethodChecks.mustBeAbstract( Constants.PROP_ANNOTATION_CLASSNAME, method );
818    MethodChecks.mustNotBePublic( Constants.PROP_ANNOTATION_CLASSNAME, method );
819    MethodChecks.mustNotHaveAnyParameters( Constants.PROP_ANNOTATION_CLASSNAME, method );
820    MethodChecks.mustReturnAValue( Constants.PROP_ANNOTATION_CLASSNAME, method );
821    MethodChecks.mustNotThrowAnyExceptions( Constants.PROP_ANNOTATION_CLASSNAME, method );
822    MethodChecks.mustNotBePackageAccessInDifferentPackage( descriptor.getElement(),
823                                                           Constants.PROP_ANNOTATION_CLASSNAME,
824                                                           method );
825    final TypeMirror returnType = method.getReturnType();
826    if ( "key".equals( name ) )
827    {
828      throw new ReactProcessorException( "@Prop named 'key' is invalid as the name references value used in the " +
829                                         "reconciliation process.", method );
830    }
831    else if ( "build".equals( name ) )
832    {
833      throw new ReactProcessorException( "@Prop named 'build' is invalid as it conflicts with the method named " +
834                                         "build() that is used in the generated Builder classes",
835                                         method );
836    }
837    else if ( "child".equals( name ) &&
838              ( returnType.getKind() != TypeKind.DECLARED && !"react4j.ReactNode".equals( returnType.toString() ) ) )
839    {
840      throw new ReactProcessorException( "@Prop named 'child' should be of type react4j.ReactNode", method );
841    }
842    else if ( "children".equals( name ) &&
843              ( returnType.getKind() != TypeKind.DECLARED && !"react4j.ReactNode[]".equals( returnType.toString() ) ) )
844    {
845      throw new ReactProcessorException( "@Prop named 'children' should be of type react4j.ReactNode[]", method );
846    }
847
848    if ( returnType instanceof TypeVariable )
849    {
850      final TypeVariable typeVariable = (TypeVariable) returnType;
851      final String typeVariableName = typeVariable.asElement().getSimpleName().toString();
852      List<? extends TypeParameterElement> typeParameters = method.getTypeParameters();
853      if ( typeParameters.stream().anyMatch( p -> p.getSimpleName().toString().equals( typeVariableName ) ) )
854      {
855        throw new ReactProcessorException( "@Prop named '" + name + "' is has a type variable as a return type " +
856                                           "that is declared on the method.", method );
857      }
858    }
859
860    final Element propType = processingEnv.getTypeUtils().asElement( returnType );
861    final boolean immutable = isPropImmutable( method );
862    final boolean shouldUpdateOnChange = shouldUpdateOnChange( method, immutable );
863    final boolean observable = isPropObservable( descriptor, method, shouldUpdateOnChange, immutable );
864    final boolean disposable = null != propType && isPropDisposable( method, propType );
865    final TypeName typeName = TypeName.get( returnType );
866    if ( typeName.isBoxedPrimitive() &&
867         ProcessorUtil.hasAnnotationOfType( method, Constants.NONNULL_ANNOTATION_CLASSNAME ) )
868    {
869      throw new ReactProcessorException( "@Prop named '" + name + "' is a boxed primitive annotated with a " +
870                                         "@Nonnull annotation. The return type should be the primitive type.",
871                                         method );
872    }
873    final ImmutablePropKeyStrategy strategy = immutable ? getImmutablePropKeyStrategy( typeName, propType ) : null;
874    if ( immutable && null == strategy )
875    {
876      throw new ReactProcessorException( "@Prop named '" + name + "' has specified the 'immutable' parameter as " +
877                                         "true but the annotation processor can not extract a key part from the " +
878                                         "type. This is because the type is not recognized as conforming to the " +
879                                         "rules as documented in the javadocs for the immutable parameter of " +
880                                         "the @Prop annotation.",
881                                         method );
882    }
883    return new PropDescriptor( descriptor,
884                               name,
885                               method,
886                               methodType,
887                               shouldUpdateOnChange,
888                               observable,
889                               disposable,
890                               strategy );
891  }

893  @Nullable
894  private ImmutablePropKeyStrategy getImmutablePropKeyStrategy( @Nonnull final TypeName typeName,
895                                                                @Nullable final Element element )
896  {
897    if ( typeName.toString().equals( "java.lang.String" ) )
898    {
899      return ImmutablePropKeyStrategy.IS_STRING;
900    }
901    else if ( typeName.isBoxedPrimitive() || typeName.isPrimitive() )
902    {
903      return ImmutablePropKeyStrategy.TO_STRING;
904    }
905    else if ( null != element )
906    {
907      if ( ElementKind.CLASS == element.getKind() &&
908           ProcessorUtil.hasAnnotationOfType( element, Constants.AREZ_COMPONENT_ANNOTATION_CLASSNAME ) &&
909           isIdRequired( (TypeElement) element ) )
910      {
911        return ImmutablePropKeyStrategy.AREZ_IDENTIFIABLE;
912      }
913      else if ( ElementKind.CLASS == element.getKind() || ElementKind.INTERFACE == element.getKind() )
914      {
915        final TypeElement keyedType = processingEnv.getElementUtils().getTypeElement( Constants.KEYED_CLASSNAME );
916        if ( processingEnv.getTypeUtils().isAssignable( element.asType(), keyedType.asType() ) )
917        {
918          return ImmutablePropKeyStrategy.KEYED;
919        }
920      }
921      else if ( ElementKind.ENUM == element.getKind() )
922      {
923        return ImmutablePropKeyStrategy.ENUM;
924      }
925    }
926    return null;
927  }

1055  private void determineRenderMethod( @Nonnull final TypeElement typeElement,
1056                                      @Nonnull final ComponentDescriptor descriptor )
1057  {
1058    /*
1059     * Get the render method that has been overridden by the typeElement, a parent class, or by a default
1060     * method method implemented by the typeElement or a parent class.
1061     */
1062    final ExecutableElement renderMethod = getComponentRenderMethod();
1063    final Elements elementUtils = processingEnv.getElementUtils();
1064    final Types typeUtils = processingEnv.getTypeUtils();
1065    final TypeElement componentType = elementUtils.getTypeElement( Constants.COMPONENT_CLASSNAME );
1066    final MethodDescriptor overriddenRenderMethod =
1067      // Get all methods on type parent classes, and default methods from interfaces
1068      getMethods( typeElement ).stream()
1069        // Only keep method if they override the render method
1070        .filter( m -> elementUtils.overrides( m, renderMethod, typeElement ) )
1071        //Remove those that come from the base classes
1072        .filter( m -> m.getEnclosingElement() != componentType )
1073        .map( m -> new MethodDescriptor( m, (ExecutableType) typeUtils.asMemberOf( descriptor.getDeclaredType(), m ) ) )
1074        .findAny().
1075        orElse( null );
1076
1077    if ( null == overriddenRenderMethod )
1078    {
1079      throw new ReactProcessorException( "The react component does not override the render method.", typeElement );
1080    }
1081  }

1135  private void determineComponentCapabilities( @Nonnull final ComponentDescriptor descriptor,
1136                                               @Nonnull final TypeElement typeElement )
1137  {
1138    final TypeElement componentType = processingEnv.getElementUtils().getTypeElement( Constants.COMPONENT_CLASSNAME );
1139    final TypeMirror rawComponentType = processingEnv.getTypeUtils().erasure( componentType.asType() );
1140
1141    final TypeMirror declaredType = descriptor.getDeclaredType();
1142
1143    final boolean isComponent = processingEnv.getTypeUtils().isSubtype( declaredType, rawComponentType );
1144
1145    if ( !isComponent )
1146    {
1147      throw new ReactProcessorException( "@ReactComponent target must be a subclass of react4j.Component",
1148                                         typeElement );
1149    }
1150    else
1151    {
1152      final AnnotationMirror arezAnnotation = typeElement.getAnnotationMirrors().stream().
1153        filter( m -> m.getAnnotationType().toString().equals( "arez.annotations.ArezComponent" ) ).
1154        findAny().orElse( null );
1155      if ( null != arezAnnotation )
1156      {
1157        throw new ReactProcessorException( "@ReactComponent target should not be annotated with the " +
1158                                           "arez.annotations.ArezComponent as React4j will add the annotation.",
1159                                           typeElement );
1160      }
1161    }
1162
1163    if ( descriptor.needsInjection() && !descriptor.getDeclaredType().getTypeArguments().isEmpty() )
1164    {
1165      throw new ReactProcessorException( "@ReactComponent target has enabled injection integration but the class " +
1166                                         "has type arguments which is incompatible with injection integration.",
1167                                         typeElement );
1168    }
1169
1170    final boolean hasArezElements =
1171      descriptor.trackRender() ||
1172      getMethods( typeElement ).stream().anyMatch( e -> e.getAnnotationMirrors()
1173        .stream()
1174        .map( a -> a.getAnnotationType().toString() )
1175        .anyMatch( n -> n.startsWith( "arez.annotations." ) &&
1176                        !(
1177                          // Ignore these annotations as they do not create disposable elements
1178                          n.endsWith( "PostConstruct" ) ||
1179                          n.endsWith( "ContextRef" ) ||
1180                          n.endsWith( "ComponentTypeNameRef" ) ||
1181                          n.endsWith( "ComponentNameRef" ) ||
1182                          n.endsWith( "ComponentIdRef" ) ||
1183                          n.endsWith( "ComponentId" ) ||
1184                          n.endsWith( "Action" )
1185                        )
1186        )
1187      ) ||
1188      ProcessorUtil
1189        .getFieldElements( typeElement )
1190        .stream()
1191        .anyMatch( e -> e.getAnnotationMirrors()
1192          .stream()
1193          .map( a -> a.getAnnotationType().toString() )
1194          .anyMatch( n -> n.equals( Constants.CASCADE_DISPOSE_ANNOTATION_CLASSNAME ) ||
1195                          n.equals( Constants.COMPONENT_DEPENDENCY_ANNOTATION_CLASSNAME ) )
1196        );
1197
1198    descriptor.setHasArezElements( hasArezElements );
1199
1200    ensureMemoizeMatchesExpectations( typeElement );
1201
1202    descriptor.setMemoizeMethods( getMemoizeMethods( typeElement ) );
1203  }

1309  private boolean isPropDisposable( @Nonnull final ExecutableElement method, @Nonnull final Element propType )
1310  {
1311    final VariableElement parameter = (VariableElement)
1312      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
1313                                        method,
1314                                        Constants.PROP_ANNOTATION_CLASSNAME,
1315                                        "disposable" ).getValue();
1316    switch ( parameter.getSimpleName().toString() )
1317    {
1318      case "ENABLE":
1319        return true;
1320      case "DISABLE":
1321        return false;
1322      default:
1323        return ElementKind.CLASS == propType.getKind() &&
1324               ProcessorUtil.hasAnnotationOfType( propType, Constants.AREZ_COMPONENT_ANNOTATION_CLASSNAME );
1325    }
1326  }

1328  /**
1329   * Return @Memoize that have not had the priority parameter explicitly set.
1330   */
1331  @Nonnull
1332  private List<MethodDescriptor> getMemoizeMethods( @Nonnull final TypeElement typeElement )
1333  {
1334    final DeclaredType type = (DeclaredType) typeElement.asType();
1335    return getMethods( typeElement )
1336      .stream()
1337      .filter( method -> !method.getModifiers().contains( Modifier.PRIVATE ) )
1338      .filter( method -> {
1339        final AnnotationMirror mirror =
1340          ProcessorUtil.findAnnotationByType( method, Constants.MEMOIZE_ANNOTATION_CLASSNAME );
1341        return null != mirror &&
1342               mirror.getElementValues().keySet().stream()
1343                 .noneMatch( v -> "priority".equals( v.getSimpleName().toString() ) );
1344      } )
1345      .map( m -> new MethodDescriptor( m, (ExecutableType) processingEnv.getTypeUtils().asMemberOf( type, m ) ) )
1346      .collect( Collectors.toList() );
1347  }

1349  private void verifyNoDuplicateAnnotations( @Nonnull final ExecutableElement method )
1350    throws ReactProcessorException
1351  {
1352    final String[] annotationTypes =
1353      new String[]{ Constants.PROP_DEFAULT_ANNOTATION_CLASSNAME,
1354                    Constants.PROP_VALIDATE_ANNOTATION_CLASSNAME,
1355                    Constants.ON_PROP_CHANGE_ANNOTATION_CLASSNAME,
1356                    Constants.PROP_ANNOTATION_CLASSNAME };
1357    for ( int i = 0; i < annotationTypes.length; i++ )
1358    {
1359      final String type1 = annotationTypes[ i ];
1360      final Object annotation1 = ProcessorUtil.findAnnotationByType( method, type1 );
1361      if ( null != annotation1 )
1362      {
1363        for ( int j = i + 1; j < annotationTypes.length; j++ )
1364        {
1365          final String type2 = annotationTypes[ j ];
1366          final Object annotation2 = ProcessorUtil.findAnnotationByType( method, type2 );
1367          if ( null != annotation2 )
1368          {
1369            final String message =
1370              "Method can not be annotated with both @" + ProcessorUtil.toSimpleName( type1 ) +
1371              " and @" + ProcessorUtil.toSimpleName( type2 );
1372            throw new ReactProcessorException( message, method );
1373          }
1374        }
1375      }
1376    }
1377  }
