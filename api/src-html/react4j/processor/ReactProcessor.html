040/**
041 * Annotation processor that analyzes React annotated source and generates models from the annotations.
042 */
043@SuppressWarnings( "Duplicates" )
044@AutoService( Processor.class )
045@SupportedAnnotationTypes( { "react4j.annotations.*" } )
046@SupportedSourceVersion( SourceVersion.RELEASE_8 )
047public final class ReactProcessor
048  extends AbstractProcessor
049{
050  private static final List<String> LIFECYCLE_METHODS =
051    Arrays.asList( "componentDidMount",
052                   "componentDidUpdate",
053                   "componentWillMount",
054                   "componentWillReceiveProps",
055                   "componentWillUnmount",
056                   "componentWillUpdate",
057                   "componentDidCatch",
058                   "getChildContext",
059                   "shouldComponentUpdate" );

061  /**
062   * Cache of lifecycle names to methods as defined by Component.
063   */
064  private final HashMap<String, ExecutableElement> _componentLifecycleMethods = new HashMap<>();
065  /**
066   * Cache of render method method as defined by Component.
067   */
068  private ExecutableElement _componentRenderMethod;
069
070  /**
071   * {@inheritDoc}
072   */
073  @Override
074  public boolean process( final Set<? extends TypeElement> annotations, final RoundEnvironment env )
075  {
076    // Clear method caches to avoid potential inter-run problems
077    _componentLifecycleMethods.clear();
078    _componentRenderMethod = null;
079
080    final TypeElement annotation =
081      processingEnv.getElementUtils().getTypeElement( Constants.REACT_COMPONENT_ANNOTATION_CLASSNAME );
082    final Set<? extends Element> elements = env.getElementsAnnotatedWith( annotation );
083    processElements( elements );
084    return false;
085  }
086
087  private void processElements( @Nonnull final Set<? extends Element> elements )
088  {
089    for ( final Element element : elements )
090    {
091      try
092      {
093        process( (TypeElement) element );
094      }
095      catch ( final IOException ioe )
096      {
097        processingEnv.getMessager().printMessage( ERROR, ioe.getMessage(), element );
098      }
099      catch ( final ReactProcessorException e )
100      {
101        processingEnv.getMessager().printMessage( ERROR, e.getMessage(), e.getElement() );
102      }
103      catch ( final Throwable e )
104      {
105        final StringWriter sw = new StringWriter();
106        e.printStackTrace( new PrintWriter( sw ) );
107        sw.flush();
108
109        final String message =
110          "Unexpected error will running the " + getClass().getName() + " processor. This has " +
111          "resulted in a failure to process the code and has left the compiler in an invalid " +
112          "state. Please report the failure to the developers so that it can be fixed.\n" +
113          " Report the error at: https://github.com/react4j/react4j/issues\n" +
114          "\n\n" +
115          sw.toString();
116        processingEnv.getMessager().printMessage( ERROR, message, element );
117      }
118    }
119  }
120
121  private void process( @Nonnull final TypeElement element )
122    throws IOException, ReactProcessorException
123  {
124    final ComponentDescriptor descriptor = parse( element );
125    emitTypeSpec( descriptor.getPackageName(), Generator.buildEnhancedComponent( descriptor ) );
126    if ( descriptor.needsDaggerIntegration() )
127    {
128      emitTypeSpec( descriptor.getPackageName(), Generator.buildDaggerFactory( descriptor ) );
129    }
130  }
131
132  private void emitTypeSpec( @Nonnull final String packageName, @Nonnull final TypeSpec typeSpec )
133    throws IOException
134  {
135    JavaFile.builder( packageName, typeSpec ).
136      skipJavaLangImports( true ).
137      build().
138      writeTo( processingEnv.getFiler() );
139  }

141  @Nonnull
142  private ComponentDescriptor parse( @Nonnull final TypeElement typeElement )
143  {
144    final String name = deriveComponentName( typeElement );
145    final PackageElement packageElement = processingEnv.getElementUtils().getPackageOf( typeElement );
146    final ComponentDescriptor descriptor = new ComponentDescriptor( name, packageElement, typeElement );
147
148    determineComponentType( descriptor, typeElement );
149    determinePropsAndStateTypes( descriptor );
150    determineLifecycleMethods( typeElement, descriptor );
151    determineChildContextTypes( descriptor );
152    determineRenderMethod( typeElement, descriptor );
153    determineEventHandlers( descriptor );
154    determineDefaultPropsMethod( descriptor );
155
156    return descriptor;
157  }

159  private void determineDefaultPropsMethod( @Nonnull final ComponentDescriptor descriptor )
160  {
161    final List<ExecutableElement> defaultPropsMethods =
162      ProcessorUtil.getMethods( descriptor.getElement(), processingEnv.getTypeUtils() ).stream()
163        .filter( m -> m.getSimpleName().toString().equals( "getInitialProps" ) )
164        .collect( Collectors.toList() );
165
166    if ( !defaultPropsMethods.isEmpty() )
167    {
168      for ( final ExecutableElement method : defaultPropsMethods )
169      {
170        final ExecutableType methodType =
171          (ExecutableType) processingEnv.getTypeUtils().asMemberOf( descriptor.getDeclaredType(), method );
172
173        if ( methodType.getThrownTypes().isEmpty() &&
174             methodType.getParameterTypes().isEmpty() &&
175             method.getModifiers().contains( Modifier.STATIC ) &&
176             !method.getModifiers().contains( Modifier.PRIVATE ) &&
177             processingEnv.getTypeUtils()
178               .isAssignable( methodType.getReturnType(), descriptor.getPropsType().asType() ) )
179        {
180          descriptor.setDefaultPropsMethod( method );
181          return;
182        }
183      }
184      throw new ReactProcessorException( "The getInitialProps method does not satisfy constraints. The method must " +
185                                         "be static, non-private, have no parameters, throw no exceptions and must " +
186                                         "return a value that is compatible with the prop type for the component.",
187                                         descriptor.getElement() );
188    }
189  }

191  private void determineEventHandlers( @Nonnull final ComponentDescriptor descriptor )
192  {
193    final List<EventHandlerDescriptor> eventHandlers =
194      ProcessorUtil.getMethods( descriptor.getElement(), processingEnv.getTypeUtils() ).stream()
195        .filter( m -> null != ProcessorUtil.findAnnotationByType( m, Constants.EVENT_HANDLER_ANNOTATION_CLASSNAME ) )
196        .map( m -> createEventHandlerDescriptor( descriptor, m ) )
197        .collect( Collectors.toList() );
198    for ( final EventHandlerDescriptor eventHandler : eventHandlers )
199    {
200      final ExecutableElement method = eventHandler.getMethod();
201      final TypeElement handlerType = getEventHandlerType( method );
202      if ( ElementKind.INTERFACE != handlerType.getKind() )
203      {
204        throw new ReactProcessorException( "The @EventHandler specified an invalid type that is not an interface.",
205                                           eventHandler.getMethod() );
206      }
207      if ( null == ProcessorUtil.findAnnotationByType( handlerType, Constants.JS_FUNCTION_CLASSNAME ) )
208      {
209        throw new ReactProcessorException( "The @EventHandler specified an invalid type that is not annotated " +
210                                           "with the annotation jsinterop.annotations.JsFunction.",
211                                           eventHandler.getMethod() );
212      }
213      final EventHandlerDescriptor matched = eventHandlers.stream()
214        .filter( h -> h != eventHandler && h.getName().equals( eventHandler.getName() ) )
215        .findAny().orElse( null );
216      if ( null != matched )
217      {
218        throw new ReactProcessorException( "The @EventHandler has the same name as the event handler defined by " +
219                                           matched.getMethod() + ".", eventHandler.getMethod() );
220      }
221      final EventHandlerDescriptor matched2 = eventHandlers.stream()
222        .filter( h -> h != eventHandler &&
223                      h.getMethod().getSimpleName().equals( eventHandler.getMethod().getSimpleName() ) )
224        .findAny().orElse( null );
225      if ( null != matched2 )
226      {
227        throw new ReactProcessorException( "The @EventHandler has the same method name as the event handler defined " +
228                                           "by " + matched2.getMethod() + ".", eventHandler.getMethod() );
229      }
230      final ExecutableType methodType = eventHandler.getMethodType();
231      final List<? extends TypeMirror> parameters = methodType.getParameterTypes();
232      if ( !parameters.isEmpty() )
233      {
234        // Our annotated handler method has parameters so they should exactly align
235        // in count and type with the parameters in the event handler method
236        final ExecutableElement target = eventHandler.getEventHandlerMethod();
237        final List<? extends VariableElement> targetParameters = target.getParameters();
238        if ( targetParameters.size() != parameters.size() )
239        {
240          throw new ReactProcessorException( "The @EventHandler target has " + parameters.size() + " parameters " +
241                                             "but the type parameter specified a handler with method type " +
242                                             eventHandler.getEventHandlerType().getQualifiedName() + " that has " +
243                                             "handler method with " + targetParameters.size() + " parameters. The " +
244                                             "@EventHandler target should have zero parameters or match the number " +
245                                             "of parameter in the target method " + target.getSimpleName() + ".",
246                                             eventHandler.getMethod() );
247        }
248        for ( int i = 0; i < parameters.size(); i++ )
249        {
250          final TypeMirror parameterType = parameters.get( i );
251          final VariableElement element = targetParameters.get( i );
252          final TypeMirror targetParameterType = element.asType();
253          final TypeMirror targetErased = processingEnv.getTypeUtils().erasure( targetParameterType );
254          final TypeMirror parameterErased = processingEnv.getTypeUtils().erasure( parameterType );
255          if ( !processingEnv.getTypeUtils().isAssignable( targetErased, parameterErased ) )
256          {
257            throw new ReactProcessorException( "The @EventHandler target parameter named " +
258                                               eventHandler.getMethod().getParameters().get( i ).getSimpleName() +
259                                               " of type " + parameterType + " is not assignable from target type " +
260                                               targetParameterType + " of parameter " + element.getSimpleName() +
261                                               " in method " + eventHandler.getEventHandlerType().getQualifiedName() +
262                                               "." + target.getSimpleName() + ".",
263                                               eventHandler.getMethod() );
264          }
265        }
266      }
267    }
268
269    descriptor.setEventHandlers( eventHandlers );
270  }

272  @Nonnull
273  private EventHandlerDescriptor createEventHandlerDescriptor( @Nonnull final ComponentDescriptor descriptor,
274                                                               @Nonnull final ExecutableElement method )
275  {
276    final String name = deriveEventHandlerName( method );
277    final TypeElement eventHandlerType = getEventHandlerType( method );
278    final ExecutableType methodType =
279      (ExecutableType) processingEnv.getTypeUtils().asMemberOf( descriptor.getDeclaredType(), method );
280    final List<ExecutableElement> eventHandlerMethods =
281      ProcessorUtil.getMethods( eventHandlerType, processingEnv.getTypeUtils() ).stream().
282        filter( m11 -> m11.getModifiers().contains( Modifier.ABSTRACT ) ).
283        collect( Collectors.toList() );
284    if ( eventHandlerMethods.isEmpty() )
285    {
286      throw new ReactProcessorException( "Method annotated with @EventHandler specified type " +
287                                         eventHandlerType.getQualifiedName() + " that has no abstract method and " +
288                                         "thus is not a functional interface", method );
289    }
290    else if ( eventHandlerMethods.size() > 1 )
291    {
292      throw new ReactProcessorException( "Method annotated with @EventHandler specified type " +
293                                         eventHandlerType.getQualifiedName() + " that has more than 1 abstract " +
294                                         "method and thus is not a functional interface", method );
295    }
296
297    if ( descriptor.isArezComponent() )
298    {
299      final AnnotationMirror nonActionAnnotation = method.getAnnotationMirrors().stream().
300        filter( m -> m.getAnnotationType().toString().equals( "react4j.arez.NoAutoAction" ) ).
301        findAny().orElse( null );
302      if ( null == nonActionAnnotation )
303      {
304        final AnnotationMirror actionAnnotation = method.getAnnotationMirrors().stream().
305          filter( m -> m.getAnnotationType().toString().equals( "org.realityforge.arez.annotations.Action" ) ).
306          findAny().orElse( null );
307        if ( null != actionAnnotation )
308        {
309          throw new ReactProcessorException( "Method annotated with @EventHandler is also annotated with " +
310                                             "@org.realityforge.arez.annotations.Action but is not annotated with " +
311                                             "@react4j.arez.NoAutoAction which would stop react4j from also " +
312                                             "annotating the method with @Action. Please remove @Action or add " +
313                                             "@NoAutoAction annotation.", method );
314        }
315      }
316    }
317
318    return new EventHandlerDescriptor( name, method, methodType, eventHandlerType, eventHandlerMethods.get( 0 ) );
319  }

321  @Nonnull
322  private TypeElement getEventHandlerType( @Nonnull final ExecutableElement method )
323  {
324    final DeclaredType typeMirror =
325      ProcessorUtil.getTypeMirrorAnnotationParameter( processingEnv.getElementUtils(),
326                                                      method,
327                                                      Constants.EVENT_HANDLER_ANNOTATION_CLASSNAME,
328                                                      "value" );
329    assert null != typeMirror;
330    return (TypeElement) processingEnv.getTypeUtils().asElement( typeMirror );
331  }

333  @Nonnull
334  private String deriveEventHandlerName( @Nonnull final ExecutableElement method )
335    throws ReactProcessorException
336  {
337    final String name =
338      (String) ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
339                                                 method,
340                                                 Constants.EVENT_HANDLER_ANNOTATION_CLASSNAME,
341                                                 "name" ).getValue();
342
343    if ( ProcessorUtil.isSentinelName( name ) )
344    {
345      return method.getSimpleName().toString();
346    }
347    else
348    {
349      if ( name.isEmpty() || !ProcessorUtil.isJavaIdentifier( name ) )
350      {
351        throw new ReactProcessorException( "Method annotated with @EventHandler specified invalid name " + name,
352                                           method );
353      }
354      return name;
355    }
356  }

358  private void determineLifecycleMethods( @Nonnull final TypeElement typeElement,
359                                          @Nonnull final ComponentDescriptor descriptor )
360  {
361    /*
362     * Get the list of lifecycle methods that have been overridden by typeElement
363     * a parent class, or by a default method method implemented by typeElement or
364     * a parent class.
365     */
366    final Collection<ExecutableElement> lifecycleMethods = getComponentLifecycleMethods().values();
367    final Elements elementUtils = processingEnv.getElementUtils();
368    final Types typeUtils = processingEnv.getTypeUtils();
369    final TypeElement componentType = elementUtils.getTypeElement( Constants.COMPONENT_CLASSNAME );
370    final List<MethodDescriptor> overriddenLifecycleMethods =
371      // Get all methods on type parent classes, and default methods from interfaces
372      ProcessorUtil.getMethods( typeElement, processingEnv.getTypeUtils() ).stream()
373        // Only keep methods that override the lifecycle methods
374        .filter( m -> lifecycleMethods.stream().anyMatch( l -> elementUtils.overrides( m, l, typeElement ) ) )
375        //Remove those that come from the base classes
376        .filter( m -> m.getEnclosingElement() != componentType )
377        .map( m -> new MethodDescriptor( m, (ExecutableType) typeUtils.asMemberOf( descriptor.getDeclaredType(), m ) ) )
378        .collect( Collectors.toList() );
379
380    descriptor.setLifecycleMethods( overriddenLifecycleMethods );
381  }

383  private void determineRenderMethod( @Nonnull final TypeElement typeElement,
384                                      @Nonnull final ComponentDescriptor descriptor )
385  {
386    /*
387     * Get the render method that has been overridden by the typeElement, a parent class, or by a default
388     * method method implemented by the typeElement or a parent class.
389     */
390    final ExecutableElement renderMethod = getComponentRenderMethod();
391    final Elements elementUtils = processingEnv.getElementUtils();
392    final Types typeUtils = processingEnv.getTypeUtils();
393    final TypeElement componentType = elementUtils.getTypeElement( Constants.COMPONENT_CLASSNAME );
394    final MethodDescriptor overriddenRenderMethod =
395      // Get all methods on type parent classes, and default methods from interfaces
396      ProcessorUtil.getMethods( typeElement, processingEnv.getTypeUtils() ).stream()
397        // Only keep method if they override the render method
398        .filter( m -> elementUtils.overrides( m, renderMethod, typeElement ) )
399        //Remove those that come from the base classes
400        .filter( m -> m.getEnclosingElement() != componentType )
401        .map( m -> new MethodDescriptor( m, (ExecutableType) typeUtils.asMemberOf( descriptor.getDeclaredType(), m ) ) )
402        .findAny().
403        orElse( null );
404
405    if ( null == overriddenRenderMethod )
406    {
407      throw new ReactProcessorException( "The react component does not override any render methods.", typeElement );
408    }
409    else
410    {
411      descriptor.setRenderMethod( overriddenRenderMethod );
412    }
413  }

415  @Nonnull
416  private String deriveComponentName( @Nonnull final TypeElement typeElement )
417  {
418    final String name =
419      (String) ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
420                                                 typeElement,
421                                                 Constants.REACT_COMPONENT_ANNOTATION_CLASSNAME,
422                                                 "name" ).getValue();
423
424    if ( ProcessorUtil.isSentinelName( name ) )
425    {
426      return typeElement.getSimpleName().toString();
427    }
428    else
429    {
430      if ( name.isEmpty() || !ProcessorUtil.isJavaIdentifier( name ) )
431      {
432        throw new ReactProcessorException( "The @ReactComponent specified an invalid name. Name should be follow " +
433                                           "the rules of a java identifier.", typeElement );
434      }
435      return name;
436    }
437  }

439  /**
440   * Return the set of lifecycle methods as defined on the base component class.
441   * Used to compare against those on type class to see if they override.
442   */
443  @Nonnull
444  private HashMap<String, ExecutableElement> getComponentLifecycleMethods()
445  {
446    if ( _componentLifecycleMethods.isEmpty() )
447    {
448      final TypeElement componentType = processingEnv.getElementUtils().getTypeElement( Constants.COMPONENT_CLASSNAME );
449      for ( final ExecutableElement method : ProcessorUtil.getMethods( componentType, processingEnv.getTypeUtils() ) )
450      {
451        final String methodName = method.getSimpleName().toString();
452        if ( LIFECYCLE_METHODS.contains( methodName ) )
453        {
454          _componentLifecycleMethods.put( methodName, method );
455        }
456      }
457
458    }
459    return _componentLifecycleMethods;
460  }

462  /**
463   * Return the set of render methods as defined on the base component class.
464   * Used to compare against those on type class to see if they override.
465   */
466  @Nonnull
467  private ExecutableElement getComponentRenderMethod()
468  {
469    if ( null == _componentRenderMethod )
470    {
471      final TypeElement componentType = processingEnv.getElementUtils().getTypeElement( Constants.COMPONENT_CLASSNAME );
472      for ( final ExecutableElement method : ProcessorUtil.getMethods( componentType, processingEnv.getTypeUtils() ) )
473      {
474        final String methodName = method.getSimpleName().toString();
475        if ( "render".equals( methodName ) )
476        {
477          _componentRenderMethod = method;
478          break;
479        }
480      }
481    }
482    return _componentRenderMethod;
483  }

485  private void determineComponentType( @Nonnull final ComponentDescriptor descriptor,
486                                       @Nonnull final TypeElement typeElement )
487  {
488    final TypeElement componentType = processingEnv.getElementUtils().getTypeElement( Constants.COMPONENT_CLASSNAME );
489    final TypeMirror rawComponentType = processingEnv.getTypeUtils().erasure( componentType.asType() );
490
491    /*
492     * Arez need not be on the classpath in which case this will return a null value to arezComponentType.
493     * Our code should just gracefully handle this and perform none of the arez specific checks or generation.
494     */
495    @Nullable
496    final TypeElement arezComponentType =
497      processingEnv.getElementUtils().getTypeElement( "react4j.arez.ReactArezComponent" );
498    @Nullable
499    final TypeMirror rawArezComponentType =
500      null == arezComponentType ? null : processingEnv.getTypeUtils().erasure( arezComponentType.asType() );
501
502    final TypeMirror type = descriptor.getDeclaredType();
503
504    final boolean isComponent = processingEnv.getTypeUtils().isSubtype( type, rawComponentType );
505    final boolean isArezComponent =
506      null != rawArezComponentType && processingEnv.getTypeUtils().isSubtype( type, rawArezComponentType );
507
508    if ( !isComponent )
509    {
510      throw new ReactProcessorException( "@ReactComponent target must be a subclass of react4j.core.Component",
511                                         typeElement );
512    }
513    else if ( isArezComponent )
514    {
515      final AnnotationMirror arezAnnotation = typeElement.getAnnotationMirrors().stream().
516        filter( m -> m.getAnnotationType().toString().equals( "org.realityforge.arez.annotations.ArezComponent" ) ).
517        findAny().orElse( null );
518      if ( null != arezAnnotation )
519      {
520        throw new ReactProcessorException( "@ReactComponent target extends react4j.arez.ReactArezComponent and should " +
521                                           "not be annotated with org.realityforge.arez.annotations.ArezComponent as " +
522                                           "React4j will add annotation", typeElement );
523      }
524    }
525
526    final boolean runArezScheduler =
527      ProcessorUtil.getMethods( typeElement, processingEnv.getTypeUtils() ).
528        stream().anyMatch( this::hasAutorunAnnotation );
529
530    final boolean needsInjection = isInjectionRequired( typeElement );
531    final boolean isDaggerPresent = needsInjection && isDaggerRequired( typeElement );
532
533    descriptor.setNeedsInjection( needsInjection );
534    descriptor.setNeedsDaggerIntegration( isDaggerPresent );
535    descriptor.setArezComponent( isArezComponent );
536    descriptor.setRunArezScheduler( runArezScheduler );
537  }

539  private boolean isInjectionRequired( @Nonnull final TypeElement typeElement )
540  {
541    final VariableElement injectParameter = (VariableElement)
542      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
543                                        typeElement,
544                                        Constants.REACT_COMPONENT_ANNOTATION_CLASSNAME,
545                                        "inject" ).getValue();
546    switch ( injectParameter.getSimpleName().toString() )
547    {
548      case "TRUE":
549        return true;
550      case "FALSE":
551        return false;
552      default:
553        return ProcessorUtil.getFieldElements( typeElement ).stream().anyMatch( this::hasInjectAnnotation ) ||
554               ProcessorUtil.getMethods( typeElement, processingEnv.getTypeUtils() ).
555                 stream().anyMatch( this::hasInjectAnnotation );
556    }
557  }
558
559  private boolean isDaggerRequired( @Nonnull final TypeElement typeElement )
560  {
561    final VariableElement injectParameter = (VariableElement)
562      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
563                                        typeElement,
564                                        Constants.REACT_COMPONENT_ANNOTATION_CLASSNAME,
565                                        "dagger" ).getValue();
566    switch ( injectParameter.getSimpleName().toString() )
567    {
568      case "TRUE":
569        return true;
570      case "FALSE":
571        return false;
572      default:
573        return null != processingEnv.getElementUtils().getTypeElement( Constants.DAGGER_MODULE_CLASSNAME );
574    }
575  }

587  private void determinePropsAndStateTypes( @Nonnull final ComponentDescriptor descriptor )
588  {
589    final TypeElement componentType = processingEnv.getElementUtils().getTypeElement( Constants.COMPONENT_CLASSNAME );
590    final List<? extends TypeParameterElement> typeParameters = componentType.getTypeParameters();
591    assert 3 == typeParameters.size();
592
593    final TypeParameterElement propsTypeParameter = typeParameters.get( 0 );
594    assert propsTypeParameter.getSimpleName().toString().equals( "P" );
595    final TypeElement propsType = resolveToElement( descriptor, propsTypeParameter );
596    descriptor.setPropsType( propsType );
597
598    final TypeParameterElement stateTypeParameter = typeParameters.get( 1 );
599    assert stateTypeParameter.getSimpleName().toString().equals( "S" );
600    final TypeElement stateType = resolveToElement( descriptor, stateTypeParameter );
601    descriptor.setStateType( stateType );
602
603    final TypeParameterElement contextTypeParameter = typeParameters.get( 2 );
604    assert contextTypeParameter.getSimpleName().toString().equals( "C" );
605    final TypeElement contextType = resolveToElement( descriptor, contextTypeParameter );
606
607    final Map<String, TypeMirror> contextTypeFields =
608      ProcessorUtil.getFields( contextType, processingEnv.getTypeUtils() );
609
610    descriptor.setContextType( contextType, contextTypeFields );
611  }

613  private void determineChildContextTypes( @Nonnull final ComponentDescriptor descriptor )
614  {
615    final MethodDescriptor getChildContext = descriptor.getLifecycleMethods().stream().
616      filter( m -> m.getMethod().getSimpleName().toString().equals( "getChildContext" ) ).findFirst().orElse( null );
617    if ( null != getChildContext )
618    {
619      final DeclaredType returnType = (DeclaredType) getChildContext.getMethodType().getReturnType();
620      final Map<String, TypeMirror> childContextTypeFields =
621        ProcessorUtil.getFields( (TypeElement) returnType.asElement(), processingEnv.getTypeUtils() );
622      descriptor.setChildContextTypeFields( childContextTypeFields );
623    }
624  }

626  @Nonnull
627  private TypeElement resolveToElement( @Nonnull final ComponentDescriptor descriptor,
628                                        @Nonnull final TypeParameterElement typeParameter )
629  {
630    final TypeMirror propsType =
631      processingEnv.getTypeUtils().asMemberOf( descriptor.getDeclaredType(), typeParameter );
