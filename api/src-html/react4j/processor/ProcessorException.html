001package react4j.processor;
002
003import java.util.Objects;
004import javax.annotation.Nonnull;
005import javax.lang.model.element.Element;
006
007public final class ProcessorException
008  extends RuntimeException
009{
010  @Nonnull
011  private final Element _element;
012
013  public ProcessorException( @Nonnull final String message, @Nonnull final Element element )
014  {
015    super( message );
016    _element = Objects.requireNonNull( element );
017  }
018
019  @Nonnull
020  public Element getElement()
021  {
022    return _element;
023  }
024}
