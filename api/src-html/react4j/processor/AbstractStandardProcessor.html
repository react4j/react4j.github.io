001package react4j.processor;
002
003import com.google.auto.common.SuperficialValidation;
004import java.io.IOException;
005import java.io.PrintWriter;
006import java.io.StringWriter;
007import java.util.ArrayList;
008import java.util.Collection;
009import java.util.HashSet;
010import java.util.List;
011import java.util.Map;
012import java.util.Set;
013import java.util.stream.Collectors;
014import javax.annotation.Nonnull;
015import javax.annotation.Nullable;
016import javax.annotation.processing.AbstractProcessor;
017import javax.annotation.processing.RoundEnvironment;
018import javax.lang.model.element.Element;
019import javax.lang.model.element.ExecutableElement;
020import javax.lang.model.element.PackageElement;
021import javax.lang.model.element.TypeElement;
022import javax.lang.model.element.VariableElement;
023import static javax.tools.Diagnostic.Kind.*;
024
025public abstract class AbstractStandardProcessor
026  extends AbstractProcessor
027{
028  @Nonnull
029  private Set<TypeElement> _deferred = new HashSet<>();
030  private int _invalidTypeCount;
031
032  @Override
033  public boolean process( @Nonnull final Set<? extends TypeElement> annotations, @Nonnull final RoundEnvironment env )
034  {
035    final TypeElement annotation = processingEnv.getElementUtils().getTypeElement( getRootAnnotationClassname() );
036    @SuppressWarnings( "unchecked" )
037    final Set<TypeElement> elements = (Set<TypeElement>) env.getElementsAnnotatedWith( annotation );
038
039    final Map<String, String> options = processingEnv.getOptions();
040    final String deferUnresolvedValue = options.get( getOptionPrefix() + ".defer.unresolved" );
041    final boolean deferUnresolved = null == deferUnresolvedValue || "true".equals( deferUnresolvedValue );
042
043    if ( deferUnresolved )
044    {
045      final Collection<TypeElement> elementsToProcess = getElementsToProcess( elements );
046      processElements( env, elementsToProcess );
047      if ( env.getRootElements().isEmpty() && !_deferred.isEmpty() )
048      {
049        _deferred.forEach( e -> processingErrorMessage( env, e ) );
050        _deferred.clear();
051      }
052    }
053    else
054    {
055      processElements( env, new ArrayList<>( elements ) );
056    }
057    if ( env.processingOver() )
058    {
059      if ( 0 != _invalidTypeCount )
060      {
061        processingEnv
062          .getMessager()
063          .printMessage( ERROR, getClass().getSimpleName() + " failed to process " + _invalidTypeCount +
064                                " types. See earlier warnings for further details." );
065      }
066      _invalidTypeCount = 0;
067    }
068    return true;
069  }
070
071  @Nonnull
072  protected abstract String getRootAnnotationClassname();
073
074  @Nonnull
075  protected abstract String getIssueTrackerURL();
076
077  @Nonnull
078  protected abstract String getOptionPrefix();
079
080  private void processingErrorMessage( @Nonnull final RoundEnvironment env, @Nonnull final TypeElement target )
081  {
082    reportError( env,
083                 getClass().getSimpleName() + " unable to process " + target.getQualifiedName() +
084                 " because not all of its dependencies could be resolved. Check for " +
085                 "compilation errors or a circular dependency with generated code.",
086                 target );
087  }
088
089  private void reportError( @Nonnull final RoundEnvironment env,
090                            @Nonnull final String message,
091                            @Nullable final Element element )
092  {
093    final String deferErrorsValue = processingEnv.getOptions().get( getOptionPrefix() + ".defer.errors" );
094    final boolean deferErrors = null == deferErrorsValue || "true".equals( deferErrorsValue );
095    _invalidTypeCount++;
096    if ( !deferErrors || env.errorRaised() || env.processingOver() )
097    {
098      processingEnv.getMessager().printMessage( ERROR, message, element );
099    }
100    else
101    {
102      processingEnv.getMessager().printMessage( MANDATORY_WARNING, message, element );
103    }
104  }
105
106  private void processElements( @Nonnull final RoundEnvironment env, @Nonnull final Collection<TypeElement> elements )
107  {
108    for ( final TypeElement element : elements )
109    {
110      try
111      {
112        process( element );
113      }
114      catch ( final IOException ioe )
115      {
116        reportError( env, ioe.getMessage(), element );
117      }
118      catch ( final ProcessorException e )
119      {
120        final Element errorLocation = e.getElement();
121        final Element outerElement = getOuterElement( errorLocation );
122        if ( !env.getRootElements().contains( outerElement ) )
123        {
124          final String location;
125          if ( errorLocation instanceof ExecutableElement )
126          {
127            final ExecutableElement executableElement = (ExecutableElement) errorLocation;
128            final TypeElement typeElement = (TypeElement) executableElement.getEnclosingElement();
129            location = typeElement.getQualifiedName() + "." + executableElement.getSimpleName();
130          }
131          else if ( errorLocation instanceof VariableElement )
132          {
133            final VariableElement variableElement = (VariableElement) errorLocation;
134            final TypeElement typeElement = (TypeElement) variableElement.getEnclosingElement();
135            location = typeElement.getQualifiedName() + "." + variableElement.getSimpleName();
136          }
137          else
138          {
139            assert errorLocation instanceof TypeElement;
140            final TypeElement typeElement = (TypeElement) errorLocation;
141            location = typeElement.getQualifiedName().toString();
142          }
143
144          final StringWriter sw = new StringWriter();
145          processingEnv.getElementUtils().printElements( sw, errorLocation );
146          sw.flush();
147
148          final String message =
149            "An error was generated processing the element " + element.getSimpleName() +
150            " but the error was triggered by code not currently being compiled but inherited or " +
151            "implemented by the element and may not be highlighted by your tooling or IDE. The " +
152            "error occurred at " + location + " and may look like:\n" + sw.toString();
153
154          reportError( env, e.getMessage(), element );
155          reportError( env, message, null );
156        }
157        reportError( env, e.getMessage(), e.getElement() );
158      }
159      catch ( final Throwable e )
160      {
161        final StringWriter sw = new StringWriter();
162        e.printStackTrace( new PrintWriter( sw ) );
163        sw.flush();
164
165        final String message =
166          "Unexpected error running the " + getClass().getName() + " processor. This has " +
167          "resulted in a failure to process the code and has left the compiler in an invalid " +
168          "state. Please report the failure to the developers so that it can be fixed.\n" +
169          " Report the error at: " + getIssueTrackerURL() + "\n" +
170          "\n\n" +
171          sw.toString();
172        reportError( env, message, element );
173      }
174    }
175  }
176
177  @Nonnull
178  private Collection<TypeElement> getElementsToProcess( @Nonnull final Collection<TypeElement> elements )
179  {
180    final List<TypeElement> deferred = _deferred
181      .stream()
182      .map( e -> processingEnv.getElementUtils().getTypeElement( e.getQualifiedName() ) )
183      .collect( Collectors.toList() );
184    _deferred = new HashSet<>();
185
186    final List<TypeElement> elementsToProcess = new ArrayList<>();
187    collectElementsToProcess( elements, elementsToProcess );
188    collectElementsToProcess( deferred, elementsToProcess );
189    return elementsToProcess;
190  }
191
192  private void collectElementsToProcess( @Nonnull final Collection<TypeElement> elements,
193                                         @Nonnull final List<TypeElement> elementsToProcess )
194  {
195    for ( final TypeElement element : elements )
196    {
197      if ( SuperficialValidation.validateElement( element ) )
198      {
199        elementsToProcess.add( element );
200      }
201      else
202      {
203        _deferred.add( element );
204      }
205    }
206  }
207
208  /**
209   * Return the outer enclosing element.
210   * This is either the top-level class, interface, enum, etc within a package.
211   * This helps identify the top level compilation units.
212   */
213  @Nonnull
214  private Element getOuterElement( @Nonnull final Element element )
215  {
216    Element result = element;
217    while ( !( result.getEnclosingElement() instanceof PackageElement ) )
218    {
219      result = result.getEnclosingElement();
220    }
221    return result;
222  }
223
224  protected abstract void process( @Nonnull final TypeElement element )
225    throws IOException, ProcessorException;
226}
