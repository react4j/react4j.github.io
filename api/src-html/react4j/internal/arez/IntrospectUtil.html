001package react4j.internal.arez;
002
003import arez.Arez;
004import arez.Observer;
005import arez.spy.ObservableValueInfo;
006import arez.spy.ObserverInfo;
007import grim.annotations.OmitType;
008import java.util.stream.Stream;
009import javax.annotation.Nonnull;
010import javax.annotation.Nullable;
011import jsinterop.base.Js;
012import jsinterop.base.JsPropertyMap;
013
014/**
015 * Utilities for introspecting the Arez components and runtime.
016 */
017@OmitType( unless = "react4j.store_debug_data_as_state" )
018public final class IntrospectUtil
019{
020  private IntrospectUtil()
021  {
022  }
023
024  /**
025   * Return the value for specified observable.
026   * Exceptions are caught and types are converted to strings using {@link java.lang.Object#toString()}
027   *
028   * @param observableInfo the observable.
029   * @return the value as a string.
030   */
031  @SuppressWarnings( "UnnecessaryUnboxing" )
032  @Nullable
033  public static Object getValue( @Nonnull final ObservableValueInfo observableInfo )
034  {
035    try
036    {
037      if ( Arez.arePropertyIntrospectorsEnabled() && observableInfo.hasAccessor() )
038      {
039        // Consider unwrapping collections and potentially serializing Arez entities so they are presented correctly in DevTools
040        final Object value = observableInfo.getValue();
041        if ( null == value )
042        {
043          return null;
044        }
045        else if ( value instanceof Enum )
046        {
047          return ( (Enum) value ).name();
048        }
049        else if ( value instanceof Integer )
050        {
051          return Js.asAny( ( (Integer) value ).intValue() );
052        }
053        else if ( value instanceof Boolean )
054        {
055          return Js.asAny( ( (Boolean) value ).booleanValue() );
056        }
057        else if ( value instanceof Long )
058        {
059          return Js.asAny( ( (Long) value ).doubleValue() );
060        }
061        else if ( value instanceof Float )
062        {
063          return Js.asAny( ( (Float) value ).doubleValue() );
064        }
065        else if ( value instanceof Short )
066        {
067          return Js.asAny( ( (Short) value ).intValue() );
068        }
069        else if ( value instanceof Byte )
070        {
071          return Js.asAny( ( (Byte) value ).intValue() );
072        }
073        else if ( value instanceof Character )
074        {
075          return value.toString();
076        }
077        else if ( value instanceof Stream )
078        {
079          // Streams are new instances every time so render them as strings to avoid infinite loops.
080          return "<Stream>";
081        }
082        else
083        {
084          return value;
085        }
086      }
087      else
088      {
089        return "?";
090      }
091    }
092    catch ( final Throwable throwable )
093    {
094      return throwable;
095    }
096  }
097
098  /**
099   * For the specified observer, collect all dependencies and record them in data to be emitted as debug data.
100   *
101   * @param observer the observer.
102   * @param data     the target in which to place debug data.
103   */
104  public static void collectDependencyDebugData( @Nonnull final Observer observer,
105                                                 @Nonnull final JsPropertyMap<Object> data )
106  {
107    final ObserverInfo observerInfo = observer.getContext().getSpy().asObserverInfo( observer );
108    observerInfo.getDependencies().forEach( d -> data.set( d.getName(), getValue( d ) ) );
109  }
110}
