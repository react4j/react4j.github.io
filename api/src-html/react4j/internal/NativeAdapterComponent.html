001package react4j.internal;
002
003import javax.annotation.Nullable;
004import jsinterop.annotations.JsConstructor;
005import jsinterop.base.JsPropertyMap;
006import react4j.Component;
007import react4j.ReactNode;
008
009/**
010 * This class provides a base class that designed to forward all lifecycle methods to a target component.
011 * The target component is of type {@link Component}.
012 *
013 * @param <I> the type of the native component.
014 */
015public abstract class NativeAdapterComponent<I extends Component>
016  extends NativeComponent
017{
018  /**
019   * The target component that all lifecycle methods are forwarded to.
020   */
021  private final I _component;
022
023  /**
024   * Create a component that designed to delegate to a target component.
025   *
026   * @param props the initial props.
027   */
028  @JsConstructor
029  protected NativeAdapterComponent( @Nullable final JsPropertyMap<Object> props )
030  {
031    super( props );
032    _component = createComponent();
033    _component.bindComponent( this );
034
035    performPostConstruct();
036  }
037
038  /**
039   * Template method that actually creates the target component.
040   *
041   * @return a new instance of the target component.
042   */
043  protected abstract I createComponent();
044
045  /**
046   * Initialize the target component.
047   */
048  private void performPostConstruct()
049  {
050    component().performPostConstruct();
051  }
052
053  /**
054   * Return the non-native component associated with this component.
055   *
056   * @return the component.
057   */
058  protected final I component()
059  {
060    return _component;
061  }
062
063  /**
064   * Call render on the target component.
065   *
066   * @return the output of rendering.
067   * @see Component#performRender()
068   */
069  @Nullable
070  public final ReactNode render()
071  {
072    return component().performRender();
073  }
074}
