001package react4j;
002
003import elemental2.core.JsObject;
004import java.util.Objects;
005import javaemul.internal.annotations.DoNotAutobox;
006import javax.annotation.Nonnull;
007import javax.annotation.Nullable;
008import jsinterop.annotations.JsOverlay;
009import jsinterop.annotations.JsPackage;
010import jsinterop.annotations.JsProperty;
011import jsinterop.annotations.JsType;
012import jsinterop.base.Js;
013import jsinterop.base.JsPropertyMap;
014import react4j.internal.ViewConstructorFunction;
015import static org.realityforge.braincheck.Guards.*;
016
017/**
018 * Element represents either a view or a host component.
019 */
020@SuppressWarnings( "unused" )
021@JsType( isNative = true, name = "Object", namespace = JsPackage.GLOBAL )
022public class ReactElement
023  implements ReactNode
024{
025  @JsProperty( name = "$$typeof" )
026  private Object typeof;
027  private Object type;
028  private String key;
029  private Object ref;
030  @JsProperty( name = "props" )
031  private JsPropertyMap<Object> inputs;
032  // The view responsible for creating this element.
033  // can be null if create happens outside of a render method (i.e. at the top level).
034  @Nullable
035  private Object _owner;
036
037  /**
038   * Complete the element.
039   * If {@link React#shouldFreezeInputs()} returns true this method will freeze the inputs and the
040   * element, otherwise this method is a no-op. This method should be called before returning the element
041   * to the react runtime.
042   */
043  @JsOverlay
044  public final void complete()
045  {
046    if ( React.shouldFreezeInputs() )
047    {
048      JsObject.freeze( this );
049      JsObject.freeze( inputs );
050    }
051  }
052
053  @JsOverlay
054  public final ReactElement dup()
055  {
056    final ReactElement element = createRawNode( typeof, type );
057    element.key = key;
058    element.ref = ref;
059    element.inputs = JsPropertyMap.of();
060    // Need to use an unchecked cast here otherwise it is cast to a java object array breaks assign
061    JsObject.assign( element.inputs, Js.uncheckedCast( this.inputs ) );
062    return element;
063  }
064
065  @JsOverlay
066  @Nonnull
067  public static ReactElement createViewElement( @Nonnull final ViewConstructorFunction type )
068  {
069    final ReactElement element = create( type );
070    element.inputs = JsPropertyMap.of();
071    element.key = null;
072    element.ref = null;
073    return element;
074  }
075
076  @JsOverlay
077  @Nonnull
078  static ReactElement createContextElement( @Nonnull final Object type )
079  {
080    final ReactElement element = create( type );
081    element.inputs = JsPropertyMap.of();
082    element.key = null;
083    element.ref = null;
084    return element;
085  }
086
087  @JsOverlay
088  @Nonnull
089  private static ReactElement create( @Nonnull final Object type )
090  {
091    return createRawNode( React.Element, type );
092  }
093
094  @JsOverlay
095  @Nonnull
096  private static ReactElement createRawNode( @Nonnull final Object typeof, @Nonnull final Object type )
097  {
098    final ReactElement element = new ReactElement();
099    element.typeof = Objects.requireNonNull( typeof );
100    element.type = Objects.requireNonNull( type );
101    element._owner = React.currentOwner();
102    return element;
103  }
104
105  @JsOverlay
106  @Nonnull
107  private static ReactElement createRawElement( @Nonnull final Object type,
108                                                @Nullable final String key,
109                                                @Nullable final Object ref,
110                                                @Nonnull final JsPropertyMap<Object> inputs )
111  {
112    final ReactElement element = create( type );
113    element.key = key;
114    element.ref = ref;
115    element.inputs = Objects.requireNonNull( inputs );
116
117    element.complete();
118    return element;
119  }
120
121  @JsOverlay
122  @Nonnull
123  public static ReactElement createFragment( @Nullable final String key, @Nonnull final ReactNode... children )
124  {
125    final ReactElement element = createRawNode( React.Element, React.Fragment );
126    element.key = key;
127    element.ref = null;
128    element.inputs = JsPropertyMap.of( "children", Objects.requireNonNull( children ) );
129
130    element.complete();
131    return element;
132  }
133
134  /**
135   * Create a StrictMode component with the specified children.
136   *
137   * @param children the child nodes.
138   * @return a new React.StrictMode component.
139   */
140  @JsOverlay
141  @Nonnull
142  public static ReactNode createStrictMode( @Nonnull final ReactNode... children )
143  {
144    return ReactElement.createRawElement( React.StrictMode,
145                                          null,
146                                          null,
147                                          JsPropertyMap.of( "children", Objects.requireNonNull( children ) ) );
148  }
149
150  @JsOverlay
151  @Nonnull
152  public static ReactElement createSuspense( @Nullable final String key,
153                                             @Nullable final ReactNode fallback,
154                                             final int maxTimeToFallback,
155                                             @Nonnull final ReactNode... children )
156  {
157    final ReactElement element = createRawNode( React.Element, React.Suspense );
158    element.key = key;
159    element.ref = null;
160    element.inputs = JsPropertyMap.of( "children", Objects.requireNonNull( children ),
161                                       "fallback", fallback,
162                                       "ms", maxTimeToFallback );
163
164    element.complete();
165    return element;
166  }
167
168  @JsOverlay
169  @Nonnull
170  public static ReactElement createHostElement( @Nonnull final String type,
171                                                @Nullable final String key,
172                                                @Nullable final Object ref,
173                                                @Nonnull final JsPropertyMap<Object> inputs )
174  {
175    return createRawElement( type, key, ref, inputs );
176  }
177
178  @JsOverlay
179  @Nullable
180  public final String key()
181  {
182    return key;
183  }
184
185  @JsOverlay
186  public final void setKey( @Nullable final String key )
187  {
188    if ( React.shouldCheckInvariants() && React.shouldFreezeInputs() )
189    {
190      invariant( () -> !JsObject.isFrozen( this ),
191                 () -> "Attempting to modify key of ReactElement after it has been frozen" );
192    }
193    this.key = key;
194  }
195
196  @JsOverlay
197  @Nonnull
198  public final JsPropertyMap<Object> inputs()
199  {
200    return inputs;
201  }
202
203  @JsOverlay
204  @Nonnull
205  public final ReactElement input( @Nonnull final String key, @DoNotAutobox final Object value )
206  {
207    inputs.set( key, value );
208    return this;
209  }
210
211  @JsOverlay
212  protected final void setInputs( @Nonnull final JsPropertyMap<Object> inputs )
213  {
214    this.inputs = inputs;
215  }
216}
