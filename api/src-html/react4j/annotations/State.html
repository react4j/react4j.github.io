001package react4j.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007
008/**
009 * Annotation used to annotate a pair of methods to access and mutate reacts underlying state object.
010 * The annotation need only be applied to one of the methods; either the getter or setter.
011 *
012 * <p>The methods that are annotated with this annotation must also comply with the following constraints:</p>
013 * <ul>
014 * <li>Must not be annotated with any other react annotation</li>
015 * <li>Must be an abstract instance method</li>
016 * <li>Must not throw exceptions</li>
017 * <li>Must be accessible from the same package as the class annotated by {@link ReactComponent}</li>
018 * </ul>
019 * <p>The getter must also comply with the following constraints:</p>
020 * <ul>
021 * <li>Must have 0 parameters</li>
022 * <li>Must return a value</li>
023 * <li>The type of the return value must be identical to the type of setters parameter</li>
024 * </ul>
025 * <p>The setter must also comply with the following constraints:</p>
026 * <ul>
027 * <li>Must have 1 parameter</li>
028 * <li>Must not return a value</li>
029 * <li>The type of the parameter must be identical to the type of the getters return value</li>
030 * </ul>
031 */
032@Documented
033@Target( ElementType.METHOD )
034public @interface State
035{
036  /**
037   * Return the name of the state value.
038   * This is the key that is used to access the value in the underlying state object.
039   * If the underlying method conforms to java beans accessor conventions (i.e. starts with "is" and is a boolean
040   * or starts with "get") then the name is the same as the java bean convention dictates, otherwise the name of
041   * the method is used as the default value for the state value.
042   *
043   * @return the name of the state value.
044   */
045  @Nonnull
046  String name() default "<default>";
047}
