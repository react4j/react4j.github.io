001package react4j.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007
008/**
009 * Annotation used to specify a React component.
010 */
011@Documented
012@Target( ElementType.TYPE )
013public @interface ReactComponent
014{
015  /**
016   * Return the name of the component.
017   * The value defaults to the simple name name of the class. If the value is specified, the
018   * value must conform to the requirements of a java identifier. It should also be unique
019   * across the suite of components used within an application but this is not strictly
020   * required as the name is only used for development purposes. (i.e. This is the name
021   * that is used within the React DevTools).
022   *
023   * @return the name of the component.
024   */
025  @Nonnull
026  String name() default "<default>";
027
028  /**
029   * Indicate whether an @Inject annotation should be added to constructor of generated class.
030   * {@link Feature#ENABLE} will force the addition of an @Inject annotation, {@link Feature#DISABLE}
031   * will result in no @Inject annotation and {@link Feature#AUTODETECT} will add an @Inject
032   * if any fields or methods in the react4j component or any parent type has an @Inject annotation.
033   *
034   * @return enum controlling present of Inject annotation on constructor.
035   */
036  Feature inject() default Feature.AUTODETECT;
037
038  /**
039   * Indicate whether a dagger sub-component and module is created for component.
040   * Dagger is detected by searching for "dagger.Module" class in the same processing environment that
041   * is compiling the react4j component.
042   *
043   * @return enum controlling whether dagger artifacts are generated.
044   */
045  Feature dagger() default Feature.AUTODETECT;
046
047  /**
048   * React components that are arez enabled will generate an error if they observe no arez dependencies and this parameter is false.
049   *
050   * @return true to avoid arez components generating invariant failures if they observe no reactive components.
051   */
052  boolean allowNoArezDeps() default false;
053}
