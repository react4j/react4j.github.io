001package react4j.annotations;
002
003import arez.annotations.Observe;
004import java.lang.annotation.Documented;
005import java.lang.annotation.ElementType;
006import java.lang.annotation.Target;
007import java.util.Objects;
008import javax.annotation.Nonnull;
009
010/**
011 * Annotation used to annotate a React component.
012 */
013@Documented
014@Target( ElementType.TYPE )
015@StingProvider( "[FlatEnclosingName][SimpleName]Factory" )
016public @interface ReactComponent
017{
018  /**
019   * Enum indicating type of component.
020   */
021  enum Type
022  {
023    /**
024     * The component can store state in fields, can declared lifecycle methods and may invoke scheduleRender() method.
025     * The component will be scheduled to re-render if the components container re-renders and any props with
026     * {@link Prop#shouldUpdateOnChange()} set to <code>true</code> has changed value from last render. A prop
027     * is considered changed if when the new value and the old value are passed to
028     * {@link Objects#equals(Object, Object)} and that method returns <code>false</code>. A component will also
029     * be scheduled to re-render if the component invokes {@code Component.scheduleRender()}.
030     */
031    STATEFUL,
032    /**
033     * The component is a {@link #STATEFUL} component <b>and</b> will re-render the component if any Arez
034     * dependencies change. The framework uses the {@link Observe} annotation to track Arez dependencies.
035     */
036    TRACKING,
037    /**
038     * The component is a {@link #TRACKING} component but it will not generate an invariant failure if a
039     * render results in zero Arez dependencies.
040     */
041    MAYBE_TRACKING
042  }
043
044  /**
045   * Return the name of the component.
046   * The value defaults to the simple name name of the class. If the value is specified, the
047   * value must conform to the requirements of a java identifier. It should also be unique
048   * across the suite of components used within an application but this is not strictly
049   * required as the name is only used for development purposes. (i.e. This is the name
050   * that is used within the React DevTools).
051   *
052   * @return the name of the component.
053   */
054  @Nonnull
055  String name() default "<default>";
056
057  /**
058   * Enum indicating the capabilities of the component.
059   * See the {@link Type} enum for further details.
060   *
061   * @return an enum indicating the capabilities of the component.
062   * @see Type
063   */
064  @Nonnull
065  Type type() default Type.STATEFUL;
066
067  /**
068   * Enum controlling whether javax.inject integration is generated for the component.
069   * The generated class is the same name as the component with the "Factory" suffix.
070   * React4j only supports constructor injection and thus this MUST NOT be set to {@link Feature#ENABLE}
071   * unless there are constructor parameters on the component. If the value is set to
072   * {@link Feature#AUTODETECT} then the feature will be enabled if the component has
073   * constructor parameters and the {@code javax.inject.Inject} class is present when
074   * compiling the component.
075   *
076   * @return an enum controlling whether javax.inject integration is generated for the component.
077   */
078  Feature inject() default Feature.AUTODETECT;
079
080  /**
081   * Enum controlling whether Sting injection integration is generated for the component.
082   * The generated class is the same name as the component with the "Factory" suffix.
083   * React4j only supports constructor injection and thus this MUST NOT be set to {@link Feature#ENABLE}
084   * unless there are constructor parameters on the component. If the value is set to
085   * {@link Feature#AUTODETECT} then the feature will be enabled if the component has
086   * constructor parameters and the {@code sting.Injectable} class is present when compiling
087   * the component.
088   *
089   * @return an enum controlling whether sting integration is generated for the component.
090   */
091  Feature sting() default Feature.AUTODETECT;
092}
