001package react4j.annotations;
002
003import arez.annotations.Observe;
004import java.lang.annotation.Documented;
005import java.lang.annotation.ElementType;
006import java.lang.annotation.Target;
007import java.util.Objects;
008import javax.annotation.Nonnull;
009
010/**
011 * Annotation used to annotate a React component.
012 */
013@Documented
014@Target( ElementType.TYPE )
015public @interface ReactComponent
016{
017  /**
018   * Enum indicating type of component.
019   */
020  enum Type
021  {
022    /**
023     * The component can store state in fields, can declared lifecycle methods and may invoke scheduleRender() method.
024     * The component will be scheduled to re-render if the components container re-renders and any props with
025     * {@link Prop#shouldUpdateOnChange()} set to <code>true</code> has changed value from last render. A prop
026     * is considered changed if when the new value and the old value are passed to
027     * {@link Objects#equals(Object, Object)} and that method returns <code>false</code>. A component will also
028     * be scheduled to re-render if the component invokes {@code Component.scheduleRender()}.
029     */
030    STATEFUL,
031    /**
032     * The component is a {@link #STATEFUL} component <b>and</b> will re-render the component if any Arez
033     * dependencies change. The framework uses the {@link Observe} annotation to track Arez dependencies.
034     */
035    TRACKING,
036    /**
037     * The component is a {@link #TRACKING} component but it will not generate an invariant failure if a
038     * render results in zero Arez dependencies.
039     */
040    MAYBE_TRACKING
041  }
042
043  /**
044   * Return the name of the component.
045   * The value defaults to the simple name name of the class. If the value is specified, the
046   * value must conform to the requirements of a java identifier. It should also be unique
047   * across the suite of components used within an application but this is not strictly
048   * required as the name is only used for development purposes. (i.e. This is the name
049   * that is used within the React DevTools).
050   *
051   * @return the name of the component.
052   */
053  @Nonnull
054  String name() default "<default>";
055
056  /**
057   * Indicate whether the generated component class is expected to be created and injected by the injection framework.
058   * {@link Feature#ENABLE} will force injection framework integration, {@link Feature#DISABLE}
059   * will force no injection framework integration and {@link Feature#AUTODETECT} will enable injection framework
060   * integration  if any fields or methods in the component or any parent type has an javax.inject.Inject
061   * annotation.
062   *
063   * @return enum controlling injection framework integration.
064   */
065  Feature inject() default Feature.AUTODETECT;
066
067  /**
068   * Enum indicating the capabilities of the component.
069   * See the {@link Type} enum for further details.
070   *
071   * @return an enum indicating the capabilities of the component.
072   * @see Type
073   */
074  @Nonnull
075  Type type() default Type.STATEFUL;
076}
