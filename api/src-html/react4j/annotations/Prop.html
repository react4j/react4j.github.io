001package react4j.annotations;
002
003import arez.annotations.ArezComponent;
004import arez.annotations.Memoize;
005import arez.annotations.Observable;
006import arez.annotations.Observe;
007import java.lang.annotation.Documented;
008import java.lang.annotation.ElementType;
009import java.lang.annotation.Target;
010import javax.annotation.Nonnull;
011import react4j.Keyed;
012
013/**
014 * Annotation used to specify an abstract method that returns a prop.
015 * The property is extracted from Reacts underlying props object.
016 *
017 * <p>The method that is annotated with this annotation must also comply with the following constraints:</p>
018 * <ul>
019 * <li>Must not be annotated with any other react annotation</li>
020 * <li>Must have 0 parameters</li>
021 * <li>Must return a value</li>
022 * <li>Must be an abstract instance method</li>
023 * <li>Must not throw exceptions</li>
024 * <li>Must be accessible from the same package as the class annotated by {@link ReactComponent}</li>
025 * </ul>
026 */
027@Documented
028@Target( ElementType.METHOD )
029public @interface Prop
030{
031  /**
032   * Return the name of the prop.
033   * If the underlying method conforms to java beans accessor conventions (i.e. starts with "is" and is a boolean
034   * or starts with "get") then the name is the same as the java bean convention dictates, otherwise the name of
035   * the method is used as the default value for the prop.
036   *
037   * @return the name of the prop.
038   */
039  @Nonnull
040  String name() default "<default>";
041
042  /**
043   * Return enum indicating whether prop should be when component is constructed.
044   * This influences validation when enabled and how the Builder class is created.
045   * If set to {@link Feature#ENABLE} then the user MUST supply the prop and the builder will require the user
046   * to specify the value. If set to {@link Feature#DISABLE} then the user can optionally supply the prop.
047   * If set to {@link Feature#AUTODETECT} then the annotation processor will treat it as {@link Feature#DISABLE}
048   * if there is a corresponding {@link PropDefault} for the prop, otherwise it will be treated as
049   * {@link Feature#ENABLE}.
050   *
051   * @return the flag indicating whether the prop needs to be supplied.
052   */
053  Feature require() default Feature.AUTODETECT;
054
055  /**
056   * Indicate whether changing the value of the prop should schedule a re-render.
057   * If set to {@link Feature#AUTODETECT} then the annotation processor will treat it as
058   * {@link Feature#ENABLE} unless {@link #immutable()} is set to <code>true</code>.
059   *
060   * @return the enum indicating whether prop should trigger a render if it is modified.
061   */
062  Feature shouldUpdateOnChange() default Feature.AUTODETECT;
063
064  /**
065   * Indicate whether the prop should be annotated by {@link Observable}.
066   *
067   * <p>If set to {@link Feature#AUTODETECT} then the prop will be observable if and only if:</p>
068   * <ul>
069   * <li>{@link #shouldUpdateOnChange()} is not set to {@link Feature#DISABLE}.</li>
070   * <li>the component has at least one method annotated with {@link Memoize} or {@link Observe}.</li>
071   * </ul>
072   *
073   * @return the enum indicating whether prop is observable.
074   */
075  Feature observable() default Feature.AUTODETECT;
076
077  /**
078   * Return an enum indicating whether the component should check whether the value of the prop is disposed
079   * prior to rendering. If the value is disposed then the render method will exit early and return null.
080   * If this parameter is set to {@link Feature#AUTODETECT} then the annotation processor will inspect the
081   * type of the prop and treat it as {@link Feature#ENABLE} if the type is annotated with the {@link ArezComponent}
082   * annotation or the {@link arez.annotations.ActAsComponent} annotation.
083   *
084   * @return an enum indicating whether the component should check whether the value of the prop is disposed prior to rendering.
085   */
086  Feature disposable() default Feature.AUTODETECT;
087
088  /**
089   * True if the prop is not expected to change after initial value is set. If the value of the prop does change
090   * then it is expected that the react component will be unmounted and a new component created. This is implemented
091   * by synthesizing a key for the component every time the component that is derived from this prop. To enable this
092   * the annotation processor must be able to identify the type of the prop so that a key can be synthesized. The
093   * following types are supported by the annotation processor;
094   *
095   * <ul>
096   * <li>primitive types (i.e. boolean, short etc) and their corresponding boxed types (i.e. {@link java.lang.Boolean}, {@link java.lang.Short} etc).</li>
097   * <li>the {@link java.lang.String} type</li>
098   * <li>any class that is annotated with {@link ArezComponent} where the {@link ArezComponent#requireId()} parameter does not resolve to {@link arez.annotations.Feature#DISABLE}</li>
099   * <li>any class that implements {@link Keyed}</li>
100   * </ul>
101   *
102   * <p>In the future, the annotation processor may include additional allowable types such as those that implement
103   * {@link arez.component.Identifiable} directly or other primitive types within the runtime library. These other
104   * types will be added if demand is established, othewise an additional hook will be added to allow users to
105   * customize key generation using a static method on the component. </p>
106   *
107   * @return true if changing the prop recreates the component.
108   */
109  boolean immutable() default false;
110}
