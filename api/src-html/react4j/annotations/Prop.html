001package react4j.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007import jsinterop.base.JsPropertyMap;
008
009/**
010 * Annotation used to specify an abstract method that returns a prop.
011 * The property is extracted from Reacts underlying props object.
012 *
013 * <p>The method that is annotated with <code>@Prop</code> must also comply with the following constraints:</p>
014 * <ul>
015 * <li>Must not be annotated with any other react annotation</li>
016 * <li>Must have 0 parameters</li>
017 * <li>Must return a value</li>
018 * <li>Must be an abstract instance method</li>
019 * <li>Must not throw exceptions</li>
020 * <li>Must be accessible from the same package as the class annotated by {@link ReactComponent}</li>
021 * </ul>
022 */
023@Documented
024@Target( ElementType.METHOD )
025public @interface Prop
026{
027  /**
028   * Return the name of the prop.
029   * If the underlying method conforms to java beans accessor conventions (i.e. starts with "is" and is a boolean
030   * or starts with "get") then the name is the same as the java bean convention dictates, otherwise the name of
031   * the method is used as the default value for the prop.
032   *
033   * @return the name of the prop.
034   */
035  @Nonnull
036  String name() default "<default>";
037
038  /**
039   * Return enum indicating whether prop should be when component is constructed.
040   * This influences validation when enabled and how the Builder class is created.
041   * If set to {@link Feature#ENABLE} then the user MUST supply the prop and the builder will require the user
042   * to specify the value. If set to {@link Feature#DISABLE} then the user can optionally supply the prop.
043   * If set to {@link Feature#AUTODETECT} then the annotation processor will treat it as {@link Feature#DISABLE}
044   * if there is a corresponding {@link PropDefault} for the prop, otherwise it will be treated as
045   * {@link Feature#ENABLE}.
046   *
047   * @return the flag indicating whether the prop needs to be supplied.
048   */
049  Feature require() default Feature.AUTODETECT;
050
051  /**
052   * Return an enum indicating whether prop should trigger a change if it updated.
053   * This setting is used when the annotation processors is responsible for implementing
054   * the {@link react4j.Component#shouldComponentUpdate(JsPropertyMap, JsPropertyMap)} lifecycle
055   * method. If set to {@link Feature#ENABLE} then the code will check the prop when implementing
056   * <code>shouldComponentUpdate</code>. If set to {@link Feature#DISABLE} then the prop will be ignored
057   * when implementing <code>shouldComponentUpdate</code>. If set to {@link Feature#AUTODETECT} then the
058   * annotation processor will treat it as {@link Feature#ENABLE}.
059   *
060   * @return the enum indicating whether prop should trigger a change if it updated.
061   */
062  Feature shouldUpdateOnChange() default Feature.AUTODETECT;
063
064  /**
065   * Return an enum indicating whether prop is disposable and should be checked before rendering component.
066   * If a prop is disposable then the class will override <code>ReactArezComponent.anyPropsDisposed()</code>
067   * and check whether the prop is disposed prior to attempting to render the component. If any props
068   * are disposed then rendering will be aborted and the render method will return null.
069   * If set to {@link Feature#ENABLE} then the code will add an isDisposed() check while if set to
070   * {@link Feature#DISABLE} no such check will be generated. If set to {@link Feature#AUTODETECT} then the
071   * annotation processor will inspect the type of the prop and treat it as {@link Feature#ENABLE} if the
072   * type is annotated with the <code>arez.annotations.ArezComponent</code> annotation and the host component is
073   * an Arez component. It is invalid to specify {@link Feature#ENABLE} unless the host component is an Arez
074   * component.
075   *
076   * @return the enum indicating whether prop is disposable and should be checked before rendering component.
077   */
078  Feature disposable() default Feature.AUTODETECT;
079}
