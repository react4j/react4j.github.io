001package react4j.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007
008/**
009 * Annotation used to specify an abstract method that returns a prop.
010 * The property is extracted from Reacts underlying props object.
011 *
012 * <p>The method that is annotated with this annotation must also comply with the following constraints:</p>
013 * <ul>
014 * <li>Must not be annotated with any other react annotation</li>
015 * <li>Must have 0 parameters</li>
016 * <li>Must return a value</li>
017 * <li>Must be an abstract instance method</li>
018 * <li>Must not throw exceptions</li>
019 * <li>Must be accessible from the same package as the class annotated by {@link ReactComponent}</li>
020 * </ul>
021 */
022@Documented
023@Target( ElementType.METHOD )
024public @interface Prop
025{
026  /**
027   * Return the name of the prop.
028   * If the underlying method conforms to java beans accessor conventions (i.e. starts with "is" and is a boolean
029   * or starts with "get") then the name is the same as the java bean convention dictates, otherwise the name of
030   * the method is used as the default value for the prop.
031   *
032   * @return the name of the prop.
033   */
034  @Nonnull
035  String name() default "<default>";
036
037  /**
038   * Return enum indicating whether prop should be when component is constructed.
039   * This influences validation when enabled and how the Builder class is created.
040   * If set to {@link Feature#ENABLE} then the user MUST supply the prop and the builder will require the user
041   * to specify the value. If set to {@link Feature#DISABLE} then the user can optionally supply the prop.
042   * If set to {@link Feature#AUTODETECT} then the annotation processor will treat it as {@link Feature#DISABLE}
043   * if there is a corresponding {@link PropDefault} for the prop, otherwise it will be treated as
044   * {@link Feature#ENABLE}.
045   *
046   * @return the flag indicating whether the prop needs to be supplied.
047   */
048  Feature require() default Feature.AUTODETECT;
049
050  /**
051   * Return an enum indicating whether prop should trigger a change if it updated.
052   * If set to {@link Feature#ENABLE} then the code will check the prop when implementing
053   * <code>shouldComponentUpdate</code> while setting the value to {@link Feature#DISABLE} will result in
054   * the value of the prop being ignored when determining whether a prop change should cause a re-render.
055   * If set to {@link Feature#AUTODETECT} then the annotation processor will treat it as {@link Feature#ENABLE}.
056   *
057   * @return the enum indicating whether prop should trigger a render if it is modified.
058   */
059  Feature shouldUpdateOnChange() default Feature.AUTODETECT;
060
061  /**
062   * Indicate whether the prop is an Arez observable.
063   * If set to {@link Feature#ENABLE} the generated class will make the prop observable and report changes
064   * in the prop when the prop changes. The component MUST be a subclass of <code>ReactArezComponent</code>.
065   * If set to {@link Feature#DISABLE} then the prop will not be observable. If set to {@link Feature#AUTODETECT}
066   * then the prop will be observable if the component is a subclass of <code>ReactArezComponent</code>, the value
067   * of {@link #shouldUpdateOnChange()} is not {@link Feature#DISABLE} and the component has a method
068   * annotated with <code>arez.annotations.Computed</code>, <code>arez.annotations.Memoize</code> or
069   * <code>arez.annotations.Observe</code>.
070   *
071   * @return the enum indicating whether prop is observable.
072   */
073  Feature observable() default Feature.AUTODETECT;
074
075  /**
076   * Return an enum indicating whether prop is disposable and should be checked before rendering component.
077   * If a prop is disposable then the class will override <code>ReactArezComponent.anyPropsDisposed()</code>
078   * and check whether the prop is disposed prior to attempting to render the component. If any props
079   * are disposed then rendering will be aborted and the render method will return null.
080   * If set to {@link Feature#ENABLE} then the code will add an isDisposed() check while if set to
081   * {@link Feature#DISABLE} no such check will be generated. If set to {@link Feature#AUTODETECT} then the
082   * annotation processor will inspect the type of the prop and treat it as {@link Feature#ENABLE} if the
083   * type is annotated with the <code>arez.annotations.ArezComponent</code> annotation and the host component is
084   * an Arez component. It is invalid to specify {@link Feature#ENABLE} unless the host component is an Arez
085   * component.
086   *
087   * @return the enum indicating whether prop is disposable and should be checked before rendering component.
088   */
089  Feature disposable() default Feature.AUTODETECT;
090}
