001package react4j.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007
008/**
009 * Annotation used to specify a React component.
010 * The class must be a concrete subclass of react4j.core.SideComponent with a single constructor that
011 * takes the underlying react component.
012 */
013@Documented
014@Target( ElementType.TYPE )
015public @interface ReactComponent
016{
017  /**
018   * Enum to control when injectible elements should be present.
019   */
020  enum Feature
021  {
022    TRUE, FALSE, IF_DETECTED
023  }
024
025  /**
026   * Return the name of the component.
027   * The value defaults to the simple name name of the class. If the value is specified, the
028   * value must conform to the requirements of a java identifier. It should also be unique
029   * across the suite of components used within an application but this is not strictly
030   * required as the name is only used for development purposes. (i.e. This is the name
031   * that is used within the React DevTools).
032   *
033   * @return the name of the component.
034   */
035  @Nonnull
036  String name() default "<default>";
037
038  /**
039   * Indicate whether an @Inject annotation should be added to constructor of generated class.
040   * {@link Feature#TRUE} will force the addition of an @Inject annotation, {@link Feature#FALSE}
041   * will result in no @Inject annotation and {@link Feature#IF_DETECTED} will add an @Inject
042   * if any fields or methods in the react4j component or any parent type has an @Inject annotation.
043   *
044   * @return enum controlling present of Inject annotation on constructor.
045   */
046  Feature inject() default Feature.IF_DETECTED;
047
048  /**
049   * Indicate whether a dagger sub-component and module is created for component.
050   * Dagger is detected by searching for "dagger.Module" class in the same processing environment that
051   * is compiling the react4j component.
052   *
053   * @return enum controlling whether dagger artifacts are generated.
054   */
055  Feature dagger() default Feature.IF_DETECTED;
056}
