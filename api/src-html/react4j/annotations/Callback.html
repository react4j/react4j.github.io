001package react4j.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007import jsinterop.annotations.JsFunction;
008
009/**
010 * Annotation used to specify a callback passed to react components.
011 * This annotation used to generate a callback accessor method in the
012 * helper class. The callback accessor method will return the same callback
013 * instance each time it is called, making this ideal for use in "PureComponent"
014 * style components. If the callback is annotated with <code>@JsFunction</code>
015 * then the code gives the callback a user friendly name for debugging. The
016 * code for specifying the friendly name will be optimized away when
017 * ReactConfig.enableComponentNames() returns false.
018 *
019 * <p>Using this annotation and the corresponding helper method is entirely
020 * optional but does provide some value when building components that you
021 * want to easily interact with from the React DevTools or building a component
022 * that compares props values to determine whether it needs to re-render.</p>
023 *
024 * <p>The method that is annotated with <code>@Callback</code> must also comply with the following constraints:</p>
025 * <ul>
026 * <li>Must not be annotated with any other react annotation</li>
027 * <li>Must not be an abstract instance method</li>
028 * <li>Must not throw exceptions</li>
029 * <li>Must be accessible from the same package as the class annotated by {@link ReactComponent}</li>
030 * <li>Must have 0 parameters or the same set of parameters as the callback type</li>
031 * <li>Must return a value of the same type as the callback type if any</li>
032 * </ul>
033 */
034@Documented
035@Target( ElementType.METHOD )
036public @interface Callback
037{
038  /**
039   * Return the name of the callback.
040   * The value defaults to the name of the method. The value must conform to the
041   * requirements of a java identifier and be unique across the component.
042   *
043   * @return the name of the callback.
044   */
045  @Nonnull
046  String name() default "<default>";
047
048  /**
049   * The type of the callback.
050   * The type should be an interface and the interface should be annotated with
051   * {@link jsinterop.annotations.JsFunction} to ensure that the method can be
052   * given a useful debug name.
053   *
054   * @return the type of the callback.
055   */
056  Class value() default Procedure.class;
057
058  /**
059   * The flag indicates whether the annotation processor should manage the calling context.
060   * if set to {@link Feature#ENABLE} then caller context will be setup. In practical terms
061   * this means that subclasses of <code>ReactArezComponent</code> will have the callback
062   * annotated with an <code>arez.annotations.Action</code> annotation. if set to {@link Feature#DISABLE}
063   * then the code that invokes the callback is responsible for setting up the calling context. This is
064   * typically used when you are passing a "render" prop into a component that invokes the callback to
065   * render react4j.ReactNode. The {@link Feature#AUTODETECT} value will detect whether the component needs
066   * calling context setup and do so if necessary. At the moment this means that subclasses of
067   * <code>ReactArezComponent</code> are enabled.
068   *
069   * @return the flag to control calling context.
070   */
071  Feature initCallbackContext() default Feature.AUTODETECT;
072
073  /**
074   * Interface for performing an action that does not return a value.
075   */
076  @FunctionalInterface
077  @JsFunction
078  interface Procedure
079  {
080    /**
081     * Perform an action.
082     */
083    void call();
084  }
085}
