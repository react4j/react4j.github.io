001package react4j;
002
003import javax.annotation.Nullable;
004
005/**
006 * The base java class that mirrors the react component.
007 */
008public abstract class Component
009{
010  /**
011   * Render the component.
012   * See the <a href="https://reactjs.org/docs/react-component.html#render">React Component documentation</a> for more details.
013   *
014   * @return the result of rendering.
015   */
016  @Nullable
017  protected abstract ReactNode render();
018}
