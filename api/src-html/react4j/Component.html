001package react4j;
002
003import java.util.Objects;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import react4j.internal.NativeComponent;
007import static org.realityforge.braincheck.Guards.*;
008
009/**
010 * The base java class that mirrors the react component.
011 */
012public abstract class Component
013{
014  @Nullable
015  private NativeComponent _nativeComponent;
016
017  /**
018   * Bind the native react component to this component.
019   * This should not be called by user code and is instead invoked by the generated component class.
020   *
021   * @param nativeComponent the native react component.
022   */
023  protected final void bindComponent( @Nonnull final NativeComponent nativeComponent )
024  {
025    if ( React.shouldCheckInvariants() )
026    {
027      invariant( () -> null == _nativeComponent,
028                 () -> "Invoked bindComponent() on " + this + " but component has already been bound." );
029    }
030    _nativeComponent = Objects.requireNonNull( nativeComponent );
031  }
032
033  /**
034   * Return the native react component.
035   *
036   * @return the native react component.
037   */
038  @Nonnull
039  protected NativeComponent component()
040  {
041    if ( React.shouldCheckInvariants() )
042    {
043      invariant( () -> null != _nativeComponent,
044                 () -> "Invoked component() on " + this + " before a component has been bound." );
045    }
046    assert null != _nativeComponent;
047    return _nativeComponent;
048  }
049
050  /**
051   * Render the component.
052   * See the <a href="https://reactjs.org/docs/react-component.html#render">React Component documentation</a> for more details.
053   *
054   * @return the result of rendering.
055   */
056  @Nullable
057  protected abstract ReactNode render();
058}
