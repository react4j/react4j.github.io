001package react4j;
002
003import arez.Arez;
004import arez.annotations.ComponentIdRef;
005import arez.annotations.ComponentNameRef;
006import elemental2.core.JsObject;
007import grim.annotations.OmitSymbol;
008import java.util.Objects;
009import javax.annotation.Nonnull;
010import javax.annotation.Nullable;
011import jsinterop.base.Any;
012import jsinterop.base.Js;
013import jsinterop.base.JsPropertyMap;
014import react4j.internal.NativeComponent;
015import static org.realityforge.braincheck.Guards.*;
016
017/**
018 * The base java class that mirrors the react component.
019 */
020public abstract class Component
021{
022  @Nullable
023  private NativeComponent _nativeComponent;
024  /**
025   * If the last render resulted in state update to record new debug state then this will be true.
026   * It guards against multiple renders of a single component where rendering is just updating
027   * react state to expose debug data. This should only be true if {@link React#shouldStoreDebugDataAsState()}
028   * returns true.
029   */
030  private boolean _scheduledDebugStateUpdate;
031
032  /**
033   * Bind the native react component to this component.
034   * This should not be called by user code and is instead invoked by the generated component class.
035   *
036   * @param nativeComponent the native react component.
037   */
038  protected final void bindComponent( @Nonnull final NativeComponent nativeComponent )
039  {
040    if ( React.shouldCheckInvariants() )
041    {
042      invariant( () -> null == _nativeComponent,
043                 () -> "Invoked bindComponent() on " + this + " but component has already been bound." );
044    }
045    _nativeComponent = Objects.requireNonNull( nativeComponent );
046  }
047
048  /**
049   * Return the native react component.
050   *
051   * @return the native react component.
052   */
053  @Nonnull
054  private NativeComponent component()
055  {
056    if ( React.shouldCheckInvariants() )
057    {
058      invariant( () -> null != _nativeComponent,
059                 () -> "Invoked component() on " + this + " before a component has been bound." );
060    }
061    assert null != _nativeComponent;
062    return _nativeComponent;
063  }
064
065  /**
066   * Return the component props from the native component.
067   * This may be null if no props were supplied.
068   *
069   * @return the component state.
070   */
071  protected final JsPropertyMap<Object> props()
072  {
073    return component().props();
074  }
075
076  /**
077   * Schedule this component for re-rendering skipping the <code>shouldComponentUpdate()</code> lifecycle method.
078   * This is equivalent to calling {@link #scheduleRender(boolean)} passing a <code>true</code> value.
079   *
080   * @see #scheduleRender(boolean)
081   */
082  protected final void scheduleRender()
083  {
084    component().forceUpdate();
085  }
086
087  /**
088   * Schedule this component for re-rendering.
089   * The component re-renders when props change but calling this method is another way to schedule the
090   * component to be re-rendered. When this method is called the <code>shouldComponentUpdate()</code>
091   * lifecycle method will be skipped if the force parameter is true as calling this method is equivalent to
092   * calling forceUpdate() on the native react component. See the <a href="https://reactjs.org/docs/react-component.html#forceupdate">React Component
093   * documentation</a> for more details.
094   *
095   * @param force if true then the component will skip the <code>shouldComponentUpdate()</code> lifecycle method.
096   */
097  protected final void scheduleRender( final boolean force )
098  {
099    if ( force )
100    {
101      component().forceUpdate();
102    }
103    else
104    {
105      component().setState( JsPropertyMap.of() );
106    }
107  }
108
109  /**
110   * Render the component.
111   * See the <a href="https://reactjs.org/docs/react-component.html#render">React Component documentation</a> for more details.
112   *
113   * @return the result of rendering.
114   */
115  @Nullable
116  protected abstract ReactNode render();
117
118  /**
119   * Store debug data on the state object of the native component.
120   * This is only done if {@link React#shouldStoreDebugDataAsState()} returns true and is primarily
121   * done to make it easy to debug the component from within React DevTools.
122   */
123  @OmitSymbol( unless = "react4j.store_debug_data_as_state" )
124  protected final void storeDebugDataAsState()
125  {
126    if ( React.shouldStoreDebugDataAsState() )
127    {
128      if ( _scheduledDebugStateUpdate )
129      {
130        _scheduledDebugStateUpdate = false;
131      }
132      else
133      {
134        final JsPropertyMap<Object> newState = JsPropertyMap.of();
135        // Present component id as state. Useful to track when instance ids change.
136        newState.set( "Arez.id", getComponentId() );
137        if ( Arez.areNamesEnabled() )
138        {
139          newState.set( "Arez.name", getComponentName() );
140        }
141        populateDebugData( newState );
142
143        final JsPropertyMap<Object> state = component().state();
144        final JsPropertyMap<Object> currentState = null == state ? null : Js.asPropertyMap( state );
145        /*
146         * To determine whether we need to do a state update we do compare each key and value and make sure
147         * they match. In some cases keys can be removed (i.e. a dependency is no longer observed) but as state
148         * updates in react are merges, we need to implement this by putting undefined values into the state.
149         */
150        if ( null != currentState )
151        {
152          for ( final String key : JsObject.keys( Js.uncheckedCast( currentState ) ) )
153          {
154            if ( !newState.has( key ) )
155            {
156              newState.set( key, Js.undefined() );
157            }
158          }
159
160          boolean newStateHasChanges = false;
161          for ( final String key : JsObject.keys( Js.uncheckedCast( newState ) ) )
162          {
163            final Any newValue = currentState.getAsAny( key );
164            final Any existingValue = newState.getAsAny( key );
165            if ( !Objects.equals( newValue, existingValue ) )
166            {
167              newStateHasChanges = true;
168              break;
169            }
170          }
171          if ( !newStateHasChanges )
172          {
173            return;
174          }
175        }
176        component().setState( Js.cast( JsObject.freeze( newState ) ) );
177        /*
178         * Force an update so do not go through shouldComponentUpdate() as that would be wasted cycles.
179         */
180        component().forceUpdate();
181        _scheduledDebugStateUpdate = true;
182      }
183    }
184  }
185
186  /**
187   * Return the unique identifier of component according to Arez.
188   *
189   * @return the unique identifier of component according to Arez.
190   */
191  @ComponentIdRef
192  protected abstract int getComponentId();
193
194  /**
195   * Return the name of the component according to Arez.
196   *
197   * @return the name of the component according to Arez.
198   */
199  @ComponentNameRef
200  protected abstract String getComponentName();
201
202  /**
203   * Populate the state parameter with any data that is useful when debugging the component.
204   *
205   * @param data the property map to populate with debug data.
206   */
207  @OmitSymbol( unless = "react4j.store_debug_data_as_state" )
208  protected void populateDebugData( @Nonnull final JsPropertyMap<Object> data )
209  {
210  }
211}
