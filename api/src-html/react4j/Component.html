001package react4j;
002
003import elemental2.core.JsError;
004import elemental2.core.JsObject;
005import java.util.Objects;
006import javax.annotation.Nonnull;
007import javax.annotation.Nullable;
008import jsinterop.base.Any;
009import jsinterop.base.Js;
010import jsinterop.base.JsPropertyMap;
011import react4j.internal.NativeComponent;
012import static org.realityforge.braincheck.Guards.*;
013
014/**
015 * The base java class that mirrors the react component.
016 */
017public abstract class Component
018{
019  @Nullable
020  private NativeComponent _nativeComponent;
021  /**
022   * If the last render resulted in state update to record new debug state then this will be true.
023   * It guards against multiple renders of a single component where rendering is just updating
024   * react state to expose debug data. This should only be true if {@link React#shouldStoreDebugDataAsState()}
025   * returns true.
026   */
027  private boolean _scheduledDebugStateUpdate;
028
029  public final void bindComponent( @Nonnull final NativeComponent nativeComponent )
030  {
031    _nativeComponent = Objects.requireNonNull( nativeComponent );
032  }
033
034  /**
035   * Return the native react component.
036   */
037  @Nonnull
038  private NativeComponent component()
039  {
040    if ( React.shouldCheckInvariants() )
041    {
042      invariant( () -> null != _nativeComponent,
043                 () -> "Invoked component() on " + this + " before a component has been bound." );
044    }
045    assert null != _nativeComponent;
046    return _nativeComponent;
047  }
048
049  /**
050   * Return the component props from the native component.
051   * This may be null if no props were supplied.
052   *
053   * @return the component state.
054   */
055  protected final JsPropertyMap<Object> props()
056  {
057    return component().props();
058  }
059
060  /**
061   * Schedule this component for re-rendering.
062   * The component re-renders when props change but calling this method is another way to schedule the
063   * component to be re-rendered. When this method is called the <code>shouldComponentUpdate()</code>
064   * lifecycle method will be skipped if the force parameter is true as calling this method is equivalent to
065   * calling forceUpdate() on the native react component. See the <a href="https://reactjs.org/docs/react-component.html#forceupdate">React Component
066   * documentation</a> for more details.
067   *
068   * @param force if true then the component will skip the <code>shouldComponentUpdate()</code> lifecycle method.
069   */
070  protected final void scheduleRender( final boolean force )
071  {
072    if ( force )
073    {
074      component().forceUpdate();
075    }
076    else
077    {
078      component().setState( JsPropertyMap.of() );
079    }
080  }
081
082  /**
083   * Render the component.
084   * See the <a href="https://reactjs.org/docs/react-component.html#render">React Component documentation</a> for more details.
085   *
086   * @return the result of rendering.
087   */
088  @Nullable
089  protected abstract ReactNode render();
090
091  @Nullable
092  public final ReactNode performRender()
093  {
094    return render();
095  }
096
097  /**
098   * Wrapper method that delegates to the {@link #postConstruct()} method.
099   * This method exists to give middleware a mechanism to hook into component lifecycle step.
100   */
101  public void performPostConstruct()
102  {
103    final JsPropertyMap<Object> props = props();
104    if ( React.shouldValidatePropValues() && null != props )
105    {
106      validatePropValues( props );
107    }
108    postConstruct();
109  }
110
111  /**
112   * This method is invoked after the component is constructed and bound to a native react component.
113   * This is a good place to perform initialization. It is called before {@link #render()}, therefore
114   * setting state in this method will not trigger a re-rendering. This replaces the
115   * <a href="https://reactjs.org/docs/react-component.html#componentwillmount">componentWillMount</a>
116   * lifecycle method from react as well as the code that appears in constructors in native React ES6
117   * components.
118   */
119  protected void postConstruct()
120  {
121  }
122
123  /**
124   * The componentDidCatch() method works like a JavaScript catch {} block, but for components.
125   * Only class components can be error boundaries. In practice, most of the time you’ll want to
126   * declare an error boundary component once and use it throughout your application.
127   *
128   * <p>Note that error boundaries only catch errors in the components below them in the tree. An
129   * error boundary can’t catch an error within itself. If an error boundary fails trying to render
130   * the error message, the error will propagate to the closest error boundary above it. This, too,
131   * is similar to how catch {} block works in JavaScript.</p>
132   *
133   * @param error the error that has been thrown.
134   * @param info  information about component stack during thrown error.
135   */
136  protected void componentDidCatch( @Nonnull final JsError error, @Nonnull final ReactErrorInfo info )
137  {
138  }
139
140  /**
141   * Perform validation on props supplied to the component.
142   *
143   * @param props the props of the component.
144   */
145  protected void validatePropValues( @Nonnull final JsPropertyMap<Object> props )
146  {
147  }
148
149  /**
150   * Store debug data on the state object of the native component.
151   * This is only done if {@link React#shouldStoreDebugDataAsState()} returns true and is primarily
152   * done to make it easy to debug the component from within React DevTools.
153   */
154  protected final void storeDebugDataAsState()
155  {
156    if ( React.shouldStoreDebugDataAsState() )
157    {
158      if ( _scheduledDebugStateUpdate )
159      {
160        _scheduledDebugStateUpdate = false;
161      }
162      else
163      {
164        final JsPropertyMap<Object> newState = JsPropertyMap.of();
165        populateDebugData( newState );
166
167        final JsPropertyMap<Object> state = component().state();
168        final JsPropertyMap<Object> currentState = null == state ? null : Js.asPropertyMap( state );
169        /*
170         * To determine whether we need to do a state update we do compare each key and value and make sure
171         * they match. In some cases keys can be removed (i.e. a dependency is no longer observed) but as state
172         * updates in react are merges, we need to implement this by putting undefined values into the state.
173         */
174        if ( null != currentState )
175        {
176          for ( final String key : JsObject.keys( Js.uncheckedCast( currentState ) ) )
177          {
178            if ( !newState.has( key ) )
179            {
180              newState.set( key, Js.undefined() );
181            }
182          }
183
184          boolean newStateHasChanges = false;
185          for ( final String key : JsObject.keys( Js.uncheckedCast( newState ) ) )
186          {
187            final Any newValue = currentState.getAny( key );
188            final Any existingValue = newState.getAny( key );
189            if ( !Objects.equals( newValue, existingValue ) )
190            {
191              newStateHasChanges = true;
192              break;
193            }
194          }
195          if ( !newStateHasChanges )
196          {
197            return;
198          }
199        }
200        component().setState( Js.cast( JsObject.freeze( newState ) ) );
201        /*
202         * Force an update so do not go through shouldComponentUpdate() as that would be wasted cycles.
203         */
204        component().forceUpdate();
205        _scheduledDebugStateUpdate = true;
206      }
207    }
208  }
209
210  /**
211   * Populate the state parameter with any data that is useful when debugging the component.
212   *
213   * @param state the property map to populate with debug data.
214   */
215  protected void populateDebugData( @Nonnull final JsPropertyMap<Object> state )
216  {
217  }
218}
