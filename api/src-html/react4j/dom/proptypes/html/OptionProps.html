001package react4j.dom.proptypes.html;
002
003import javax.annotation.Nonnull;
004import jsinterop.annotations.JsOverlay;
005import jsinterop.annotations.JsPackage;
006import jsinterop.annotations.JsType;
007
008/**
009 * Props for option elements. Refer to http://www.w3schools.com/tags/tag_option.asp
010 */
011@JsType( isNative = true, namespace = JsPackage.GLOBAL, name = "Object" )
012public class OptionProps
013  extends HtmlGlobalFields<OptionProps>
014{
015  @JsOverlay
016  @Nonnull
017  public final OptionProps disabled( boolean b )
018  {
019    setDisabled( b );
020    return self();
021  }
022
023  @JsOverlay
024  @Nonnull
025  public final OptionProps label( String s )
026  {
027    setLabel( s );
028    return self();
029  }
030
031  @JsOverlay
032  @Nonnull
033  public final OptionProps selected( boolean b )
034  {
035    setSelected( b );
036    return self();
037  }
038
039  @JsOverlay
040  @Nonnull
041  public final OptionProps value( String s )
042  {
043    setValue( s );
044    return self();
045  }
046}
