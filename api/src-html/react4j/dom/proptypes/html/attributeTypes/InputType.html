001package react4j.dom.proptypes.html.attributeTypes;
002
003import javax.annotation.Nonnull;
004
005/**
006 * An enumeration specifying the type of input control to render.
007 *
008 * @see <a href="https://developer.mozilla.org/en-US/docs/Web/HTML/Element/input#%3Cinput%3E_types"><code>&lt;input&gt;</code> type - MDN</a>
009 */
010public final class InputType
011{
012  /**
013   * A push button with no default behavior displaying the value of the value attribute, empty by default.
014   */
015  @Nonnull
016  public static final String button = "button";
017  /**
018   * A check box allowing single values to be selected/deselected.
019   */
020  @Nonnull
021  public static final String checkbox = "checkbox";
022  /**
023   * A control for specifying a color; opening a color picker when active in supporting browsers.
024   */
025  @Nonnull
026  public static final String color = "color";
027  /**
028   * A control for entering a date (year, month, and day, with no time). Opens a date picker or numeric wheels for year, month, day when active in supporting browsers.
029   */
030  @Nonnull
031  public static final String date = "date";
032  /**
033   * A control for entering a date and time, with no time zone. Opens a date picker or numeric wheels for date- and time-components when active in supporting browsers.
034   */
035  @Nonnull
036  public static final String datetime_local = "datetime-local";
037  /**
038   * A field for editing an email address. Looks like a text input, but has validation parameters and relevant keyboard in supporting browsers and devices with dynamic keyboards.
039   */
040  @Nonnull
041  public static final String email = "email";
042  /**
043   * A control that lets the user select a file. Use the accept attribute to define the types of files that the control can select.
044   */
045  @Nonnull
046  public static final String file = "file";
047  /**
048   * A control that is not displayed but whose value is submitted to the server.
049   */
050  @Nonnull
051  public static final String hidden = "hidden";
052  /**
053   * A graphical submit button. Displays an image defined by the src attribute. The alt attribute displays if the image src is missing.
054   */
055  @Nonnull
056  public static final String image = "image";
057  /**
058   * A control for entering a month and year, with no time zone.
059   */
060  @Nonnull
061  public static final String month = "month";
062  /**
063   * A control for entering a number. Displays a spinner and adds default validation when supported.
064   * Displays a numeric keypad in some devices with dynamic keypads.
065   */
066  @Nonnull
067  public static final String number = "number";
068  /**
069   * A single-line text field whose value is obscured. Will alert user if site is not secure.
070   */
071  @Nonnull
072  public static final String password = "password";
073  /**
074   * A radio button, allowing a single value to be selected out of multiple choices with the same name value.
075   */
076  @Nonnull
077  public static final String radio = "radio";
078  /**
079   * A control for entering a number whose exact value is not important.
080   * Displays as a range widget defaulting to the middle value. Used in conjunction min and max to define
081   * the range of acceptable values.
082   */
083  @Nonnull
084  public static final String range = "range";
085  /**
086   * A button that resets the contents of the form to default values. Not recommended.
087   */
088  @Nonnull
089  public static final String reset = "reset";
090  /**
091   * A single-line text field for entering search strings. Line-breaks are automatically
092   * removed from the input value. May include a delete icon in supporting browsers that
093   * can be used to clear the field. Displays a search icon instead of enter key on some
094   * devices with dynamic keypads.
095   */
096  @Nonnull
097  public static final String search = "search";
098  /**
099   * A button that submits the form.
100   */
101  @Nonnull
102  public static final String submit = "submit";
103  /**
104   * A control for entering a telephone number.
105   * Displays a telephone keypad in some devices with dynamic keypads.
106   */
107  @Nonnull
108  public static final String tel = "tel";
109  /**
110   * The default value. A single-line text field. Line-breaks are automatically removed from the input value.
111   */
112  @Nonnull
113  public static final String text = "text";
114  /**
115   * A control for entering a time value with no time zone.
116   */
117  @Nonnull
118  public static final String time = "time";
119  /**
120   * A field for entering a URL. Looks like a text input, but has validation parameters and relevant keyboard in supporting browsers and devices with dynamic keyboards.
121   */
122  @Nonnull
123  public static final String url = "url";
124  /**
125   * A control for entering a date consisting of a week-year number and a week number with no time zone.
126   */
127  @Nonnull
128  public static final String week = "week";
129
130  private InputType()
131  {
132  }
133}
