001package react4j.dom.proptypes.html.attributeTypes;
002
003import javax.annotation.Nonnull;
004
005/**
006 * An enumeration specifying the HTTP method to submit the form with.
007 */
008public final class FormMethod
009{
010  /**
011   * The POST method; form data sent as the request body.
012   */
013  @Nonnull
014  public static final String post = "post";
015  /**
016   * The GET method; form data appended to the action URL with a ? separator. Use this method when the form has no side-effects.
017   */
018  @Nonnull
019  public static final String get = "get";
020  /**
021   * When the form is inside a &lt;dialog&gt;, closes the dialog on submission.
022   */
023  @Nonnull
024  public static final String dialog = "dialog";
025
026  private FormMethod()
027  {
028  }
029}
