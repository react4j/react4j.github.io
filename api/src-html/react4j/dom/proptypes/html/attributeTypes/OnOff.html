001package react4j.dom.proptypes.html.attributeTypes;
002
003import javax.annotation.Nonnull;
004
005/**
006 * An enumeration describing the whether elements can have their values automatically completed by the browser.
007 */
008public final class OnOff
009{
010  /**
011   * The browser may not automatically complete entries.
012   */
013  @Nonnull
014  public static final String off = "off";
015  /**
016   * The browser may automatically complete entries.
017   */
018  @Nonnull
019  public static final String on = "on";
020
021  private OnOff()
022  {
023  }
024}
