001package react4j.dom.proptypes.html.attributeTypes;
002
003import javax.annotation.Nonnull;
004
005/**
006 * An enumerated value to provide a hint to the browser about what the author thinks will lead to the best user experience.
007 */
008public final class AudioPreload
009{
010  /**
011   * Indicates that the audio should not be preloaded.
012   */
013  @Nonnull
014  public static final String none = "none";
015  /**
016   * Indicates that only audio metadata (e.g. length) is fetched.
017   */
018  @Nonnull
019  public static final String metadata = "metadata";
020  /**
021   * Indicates that the whole audio file can be downloaded, even if the user is not expected to use it.
022   */
023  @Nonnull
024  public static final String auto = "auto";
025
026  private AudioPreload()
027  {
028  }
029}
