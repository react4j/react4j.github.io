002
003public enum Target
004{

006  self( "_self" ),
007  parent( "_parent" ),
008  top( "_top" );
009
010  private String val;
011
012  Target( String val )
013  {
014    this.val = val;
015  }
016
017  String getVal()
018  {
019    return val;
020  }
021}
