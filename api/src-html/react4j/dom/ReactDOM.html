001package react4j.dom;
002
003import elemental2.dom.Element;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import jsinterop.annotations.JsMethod;
007import jsinterop.annotations.JsOverlay;
008import jsinterop.annotations.JsPackage;
009import jsinterop.annotations.JsType;
010import react4j.Procedure;
011import react4j.React;
012import react4j.ReactConfig;
013import react4j.ReactNode;
014import react4j.dom.proptypes.html.HtmlGlobalFields;
015import static org.realityforge.braincheck.Guards.*;
016
017/**
018 * Core interface into React DOM library.
019 */
020@JsType( isNative = true, namespace = JsPackage.GLOBAL )
021public class ReactDOM
022{
023  private ReactDOM()
024  {
025  }
026
027  /**
028   * Portals provide a first-class way to render children into a DOM node that exists outside the DOM hierarchy
029   * of the parent component.
030   *
031   * <p>Even though a portal can be anywhere in the DOM tree, it behaves like a normal React child in every
032   * other way. Features like context work exactly the same regardless of whether the child is a portal, as
033   * the portal still exists in the React tree regardless of position in the DOM tree.</p>
034   *
035   * <p>This includes event bubbling. An event fired from inside a portal will propagate to ancestors in
036   * the containing React tree, even if those elements are not ancestors in the DOM tree.</p>
037   *
038   * @param children  the react node to render.
039   * @param container the DOM element to render into.
040   * @return the new portal.
041   */
042  public static native ReactPortal createPortal( @Nonnull ReactNode children, @Nonnull Element container );
043
044  /**
045   * Render a React element into the DOM in the supplied container.
046   *
047   * <p>If the React element was previously rendered into container, this will perform an update on it and only
048   * mutate the DOM as necessary to reflect the latest React element.</p>
049   *
050   * <p>If the optional callback is provided, it will be executed after the component is rendered or updated.</p>
051   *
052   * @param node      the react node to render.
053   * @param container the DOM element to render into.
054   * @param onUpdate  the callback invoked when rendering is complete.
055   * @return a reference to the created React Component, DOM Node, Portal or null (stateless components).
056   */
057  @Nullable
058  @JsOverlay
059  public static Object render( @Nonnull final ReactNode node,
060                               @Nonnull final Element container,
061                               @Nullable final Procedure onUpdate )
062  {
063    if ( ReactConfig.shouldCheckInvariants() )
064    {
065      invariant( () -> React.isValidElement( node ), () -> "ReactDOM.render passed a non ReactElement" );
066    }
067    return render0( node, container, onUpdate );
068  }
069
070  /**
071   * Internal native method for rendering component./
072   *
073   * @param node      the react node to render.
074   * @param container the DOM element to render into.
075   * @param onUpdate  the callback invoked when rendering is complete.
076   * @return a reference to the created React Component, DOM Node, Portal or null (stateless components).
077   */
078  @JsMethod( name = "render" )
079  private static native Object render0( @Nonnull ReactNode node,
080                                        @Nonnull Element container,
081                                        @Nullable Procedure onUpdate );
082
083  /**
084   * Render a React element into the DOM in the supplied container.
085   *
086   * <p>If the React element was previously rendered into container, this will perform an update on it and only
087   * mutate the DOM as necessary to reflect the latest React element.</p>
088   *
089   * @param node      the react node to render.
090   * @param container the DOM element to render into.
091   * @return a reference to the created React Component, DOM Node, Portal or null (stateless components).
092   * @see #render(ReactNode, Element, Procedure)
093   */
094  @Nullable
095  @JsOverlay
096  public static Object render( @Nonnull ReactNode node, @Nonnull Element container )
097  {
098    return render( node, container, null );
099  }
100
101  /**
102   * Remove a mounted React component from the DOM and clean up its event handlers and state. If
103   * no component was mounted in the container, calling this function does nothing.
104   *
105   * @param container the DOM container containing the react component to unmount
106   * @return true if a component was unmounted and false if there was no component to unmount.
107   */
108  public static native boolean unmountComponentAtNode( @Nonnull Element container );
109
110  /**
111   * Create and return a new ReactElement of the given type.
112   *
113   * @param type A HTML tag name (eg. 'div', 'span', etc)
114   * @return the created ReactNode
115   */
116  @Nonnull
117  @JsOverlay
118  public static ReactNode createElement( @Nonnull final String type )
119  {
120    return createElement( type, null );
121  }
122
123  /**
124   * Create and return a new ReactElement of the given type.
125   *
126   * @param <P>   The type of the props. It should match the associated tag name.
127   * @param type  A HTML tag name (eg. 'div', 'span', etc)
128   * @param props The props to pass to the element.
129   * @return the created ReactNode
130   */
131  @Nonnull
132  @JsOverlay
133  public static <P extends HtmlGlobalFields> ReactNode createElement( @Nonnull String type, @Nullable P props )
134  {
135    return React.createElement( type, props );
136  }
137
138  /**
139   * Create and return a new ReactElement of the given type with specified children.
140   *
141   * @param <P>      The type of the props. It should match the associated tag name.
142   * @param type     A HTML tag name (eg. 'div', 'span', etc)
143   * @param props    The props to pass to the element.
144   * @param children The child elements.
145   * @return the created DOMElement
146   */
147  @Nonnull
148  @JsOverlay
149  public static <P extends HtmlGlobalFields> ReactNode createElement( @Nonnull String type,
150                                                                      @Nullable P props,
151                                                                      @Nullable ReactNode... children )
152  {
153    return React.createElement( type, props, children );
154  }
155
156  /**
157   * Batch all state updates within the action.
158   * This is currently an unstable API within the React 16, mostly because it is only useful when called
159   * outside an event handler (i.e. from network code) and because it is likely to be enabled by default
160   * in a later version of React.
161   *
162   * @param action the action where all state updates are batched.
163   */
164  @JsOverlay
165  public static void batchedUpdates( @Nonnull Procedure action )
166  {
167    unstable_batchedUpdates( action );
168  }
169
170  /**
171   * The native method with the unstable prefix.
172   *
173   * @param action the action where all state updates are batched.
174   */
175  @JsMethod
176  private static native void unstable_batchedUpdates( @Nonnull Procedure action );
177}
