001package react4j.dom;
002
003import elemental2.dom.Element;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import jsinterop.annotations.JsFunction;
007import jsinterop.annotations.JsMethod;
008import jsinterop.annotations.JsOverlay;
009import jsinterop.annotations.JsPackage;
010import jsinterop.annotations.JsType;
011import react4j.ReactNode;
012
013/**
014 * Core interface into React DOM library.
015 */
016@JsType( isNative = true, namespace = JsPackage.GLOBAL )
017public class ReactDOM
018{
019  private ReactDOM()
020  {
021  }
022
023  /**
024   * Interface for performing an action inside batch.
025   */
026  @FunctionalInterface
027  @JsFunction
028  public interface BatchedUpdatesFn
029  {
030    /**
031     * Perform action while batching react changes.
032     *
033     * @throws Throwable if an error occurred.
034     */
035    void call()
036      throws Throwable;
037  }
038
039  /**
040   * Interface for performing an action on render complete.
041   */
042  @FunctionalInterface
043  @JsFunction
044  public interface RenderCallbackFn
045  {
046    /**
047     * Perform action on render complete.
048     */
049    void call();
050  }
051
052  @JsOverlay
053  @Nonnull
054  public static ReactRoot createRoot( @Nonnull final Element container )
055  {
056    return unstable_createRoot( container );
057  }
058
059  @Nonnull
060  private static native ReactRoot unstable_createRoot( @Nonnull Element container );
061
062  /**
063   * Portals provide a first-class way to render children into a DOM node that exists outside the DOM hierarchy
064   * of the parent component.
065   *
066   * <p>Even though a portal can be anywhere in the DOM tree, it behaves like a normal React child in every
067   * other way. Features like context work exactly the same regardless of whether the child is a portal, as
068   * the portal still exists in the React tree regardless of position in the DOM tree.</p>
069   *
070   * <p>This includes event bubbling. An event fired from inside a portal will propagate to ancestors in
071   * the containing React tree, even if those elements are not ancestors in the DOM tree.</p>
072   *
073   * @param children  the react node to render.
074   * @param container the DOM element to render into.
075   * @return the new portal.
076   */
077  public static native ReactPortal createPortal( @Nonnull ReactNode children, @Nonnull Element container );
078
079  /**
080   * Render a React element into the DOM in the supplied container.
081   *
082   * <p>If the React element was previously rendered into container, this will perform an update on it and only
083   * mutate the DOM as necessary to reflect the latest React element.</p>
084   *
085   * <p>If the optional callback is provided, it will be executed after the component is rendered or updated.</p>
086   *
087   * @param node      the react node to render.
088   * @param container the DOM element to render into.
089   * @param onUpdate  the callback invoked when rendering is complete.
090   * @return a reference to the created React Component, DOM Node, Portal or null (stateless components).
091   */
092  @Nullable
093  @JsMethod
094  public static native Object render( @Nonnull ReactNode node,
095                                      @Nonnull Element container,
096                                      @Nullable RenderCallbackFn onUpdate );
097
098  /**
099   * Render a React element into the DOM in the supplied container.
100   *
101   * <p>If the React element was previously rendered into container, this will perform an update on it and only
102   * mutate the DOM as necessary to reflect the latest React element.</p>
103   *
104   * @param node      the react node to render.
105   * @param container the DOM element to render into.
106   * @return a reference to the created React Component, DOM Node, Portal or null (stateless components).
107   * @see #render(ReactNode, Element, RenderCallbackFn)
108   */
109  @Nullable
110  @JsOverlay
111  public static Object render( @Nonnull ReactNode node, @Nonnull Element container )
112  {
113    return render( node, container, null );
114  }
115
116  /**
117   * Remove a mounted React component from the DOM and clean up its event handlers and state. If
118   * no component was mounted in the container, calling this function does nothing.
119   *
120   * @param container the DOM container containing the react component to unmount
121   * @return true if a component was unmounted and false if there was no component to unmount.
122   */
123  public static native boolean unmountComponentAtNode( @Nonnull Element container );
124
125  /**
126   * Batch all state updates within the action.
127   * This is currently an unstable API within the React 16, mostly because it is only useful when called
128   * outside an event handler (i.e. from network code) and because it is likely to be enabled by default
129   * in a later version of React.
130   *
131   * @param action the action where all state updates are batched.
132   */
133  @JsOverlay
134  public static void batchedUpdates( @Nonnull final BatchedUpdatesFn action )
135  {
136    unstable_batchedUpdates( action );
137  }
138
139  /**
140   * The native method with the unstable prefix.
141   *
142   * @param action the action where all state updates are batched.
143   */
144  @JsMethod
145  private static native void unstable_batchedUpdates( @Nonnull BatchedUpdatesFn action );
146}
