248  /**
249   * Create a ReactElement of either component or host type.
250   */
251  @JsOverlay
252  @Nonnull
253  private static ReactElement _createElement( @Nonnull final String type,
254                                              @Nullable final JsPropertyMap<Object> props,
255                                              @Nullable final ReactNode[] children )
256  {
257    final JsPropertyMap<Object> actual = JsPropertyMap.of();
258    String key = null;
259    Object ref = null;
260    if ( null != props )
261    {
262      key = props.has( PropNames.KEY_PROP_NAME ) ? Js.asString( props.get( PropNames.KEY_PROP_NAME ) ) : null;
263      ref = props.has( PropNames.REF_PROP_NAME ) ? props.get( PropNames.REF_PROP_NAME ) : null;
264      props.forEach( p -> {
265        // In future we can probably remove this check when/if components are creating elements directly
266        // and can thus guarantee that these keys are not part of props.
267        if ( !p.equals( PropNames.KEY_PROP_NAME ) && !p.equals( PropNames.REF_PROP_NAME ) )
268        {
269          actual.set( p, props.get( p ) );
270        }
271      } );
272    }
273    // In a future iteration if we re-write the reconciler we should probably ensure that shape of ReactElement
274    // is always consistent and is either null or always an array element.
275    if ( null != children && children.length > 0 )
276    {
277      if ( 1 == children.length )
278      {
279        actual.set( PropNames.CHILDREN_PROP_NAME, children[ 0 ] );
280      }
281      else
282      {
283        actual.set( PropNames.CHILDREN_PROP_NAME, children );
284      }
285    }
286    return ReactElement.createHostElement( type, key, ref, actual );
287  }
