001package react4j;
002
003import elemental2.core.JsArray;
004import java.util.List;
005import java.util.function.IntFunction;
006import java.util.stream.Stream;
007import javax.annotation.Nonnull;
008import javax.annotation.Nullable;
009import jsinterop.annotations.JsMethod;
010import jsinterop.annotations.JsOverlay;
011import jsinterop.annotations.JsPackage;
012import jsinterop.annotations.JsProperty;
013import jsinterop.annotations.JsType;
014import jsinterop.base.Js;
015import jsinterop.base.JsPropertyMap;
016import static org.realityforge.braincheck.Guards.*;
017
018/**
019 * Native interface to native runtime for creating component.
020 */
021@JsType( isNative = true, namespace = JsPackage.GLOBAL )
022public final class React
023{
024  private React()
025  {
026  }
027
028  /**
029   * The magic component constructor function for Fragments.
030   */
031  @SuppressWarnings( "unused" )
032  @JsProperty( name = "Fragment" )
033  private static ComponentConstructorFunction Fragment;
034
035  /**
036   * The magic component constructor function for StrictMode components.
037   */
038  @SuppressWarnings( "unused" )
039  @JsProperty( name = "StrictMode" )
040  private static ComponentConstructorFunction StrictMode;
041
042  /**
043   * Return true if the specified node is a ReactElement.
044   *
045   * @param node the node to test.
046   * @return true if the specified node is a ReactElement.
047   */
048  public static native boolean isValidElement( @Nonnull ReactNode node );
049
050  /**
051   * Create a ReactElement for the specified React component with no props or children.
052   *
053   * @param type the constructor function for the native React component.
054   * @return a new ReactElement.
055   */
056  @JsOverlay
057  public static ReactNode createElement( @Nonnull final ComponentConstructorFunction type )
058  {
059    return createElement( type, null );
060  }
061
062  /**
063   * Create a ReactElement for the specified React component with no children.
064   *
065   * @param type  the constructor function for the native React component.
066   * @param props the props to pass to the component.
067   * @return a new ReactElement.
068   */
069  @JsOverlay
070  public static ReactNode createElement( @Nonnull final ComponentConstructorFunction type,
071                                         @Nullable final JsPropertyMap<Object> props )
072  {
073    // Need to pass through undefined to react otherwise the debugger tool displays
074    // children as null rather than omitting children
075    return createElement( type, props, Js.<ReactNode>uncheckedCast( Js.undefined() ) );
076  }
077
078  /**
079   * Create a ReactElement for the specified React component.
080   *
081   * @param type  the constructor function for the native React component.
082   * @param props the props to pass to the component.
083   * @param child the child of the react component.
084   * @return a new ReactElement.
085   */
086  public static native ReactNode createElement( @Nonnull ComponentConstructorFunction type,
087                                                @Nullable JsPropertyMap<Object> props,
088                                                @Nullable ReactNode child );
089
090  /**
091   * Create a ReactElement for the specified React component.
092   *
093   * @param type     the constructor function for the native React component.
094   * @param props    the props to pass to the component.
095   * @param children the children of the react component.
096   * @return a new ReactElement.
097   */
098  @JsOverlay
099  public static ReactNode createElement( @Nonnull final ComponentConstructorFunction type,
100                                         @Nullable final JsPropertyMap<Object> props,
101                                         @Nonnull final JsArray<ReactNode> children )
102  {
103    return createElement( type, props, Js.<ReactNode[]>cast( children ) );
104  }
105
106  /**
107   * Create a ReactElement for the specified React component.
108   *
109   * @param type     the constructor function for the native React component.
110   * @param props    the props to pass to the component.
111   * @param children the children of the react component.
112   * @return a new ReactElement.
113   */
114  public static native ReactNode createElement( @Nonnull ComponentConstructorFunction type,
115                                                @Nullable JsPropertyMap<Object> props,
116                                                @Nonnull ReactNode... children );
117  /**
118   * Create and return a new ReactElement of the given type.
119   *
120   * @param type  A HTML tag name (eg. 'div', 'span', etc)
121   * @param props The props to pass to the element.
122   * @return the created ReactNode
123   */
124  @Nonnull
125  public static native ReactNode createElement( @Nonnull String type, @Nullable Object props );
126
127  /**
128   * Create and return a new ReactElement of the given type with specified children.
129   *
130   * @param type     A HTML tag name (eg. 'div', 'span', etc)
131   * @param props    The props to pass to the element.
132   * @param children The child elements.
133   * @return the created DOMElement
134   */
135  @Nonnull
136  @JsMethod
137  public static native ReactNode createElement( @Nonnull String type, @Nullable Object props, @Nullable ReactNode... children );
138
139  /**
140   * Create a StrictMode component with the specified children.
141   *
142   * @param children the child nodes.
143   * @return a new React.Fragment object.
144   */
145  @JsOverlay
146  public static ReactNode createStrictMode( @Nonnull final ReactNode... children )
147  {
148    return createElement( StrictMode, null, children );
149  }
150
151  /**
152   * Create a Fragment with the specified children.
153   *
154   * @param children the child nodes.
155   * @return a new React.Fragment object.
156   */
157  @JsOverlay
158  public static ReactNode createFragment( @Nonnull final ReactNode... children )
159  {
160    return createElement( Fragment, null, children );
161  }
162
163  /**
164   * Create a Fragment with the specified children.
165   *
166   * @param children the child nodes.
167   * @return a new React.Fragment object.
168   * @see #createElement(ComponentConstructorFunction, JsPropertyMap, ReactNode...)
169   */
170  @JsOverlay
171  public static ReactNode createFragment( @Nonnull final JsArray<ReactNode> children )
172  {
173    return createFragment( Js.<ReactNode[]>cast( children ) );
174  }
175
176  /**
177   * Create a Fragment with the specified children.
178   *
179   * @param children the child nodes.
180   * @return a new React.Fragment object.
181   * @see #createElement(ComponentConstructorFunction, JsPropertyMap, ReactNode...)
182   */
183  @JsOverlay
184  public static ReactNode createFragment( @Nonnull final List<? extends ReactNode> children )
185  {
186    return createFragment( children.stream() );
187  }
188
189  /**
190   * Create a Fragment with the specified children.
191   *
192   * @param children the child nodes.
193   * @return a new React.Fragment object.
194   * @see #createElement(ComponentConstructorFunction, JsPropertyMap, ReactNode...)
195   */
196  @JsOverlay
197  public static ReactNode createFragment( @Nonnull final Stream<? extends ReactNode> children )
198  {
199    /*
200     * The GWT compiler does not handle method refs in this context. Not sure why
201     */
202    @SuppressWarnings( "Convert2MethodRef" )
203    final IntFunction<ReactNode[]> intFunction = v -> new ReactNode[ v ];
204    return createFragment( children.toArray( intFunction ) );
205  }
206
207  /**
208   * Creates a context with specified default value.
209   *
210   * @param <T>          the type of the context.
211   * @param defaultValue the default value.
212   * @return the context.
213   */
214  @Nonnull
215  public static native <T> Context<T> createContext( T defaultValue );
216
217  /**
218   * Clone and return a new ReactElement using element as the starting point. The resulting
219   * element will have the original element's props with the new props merged in shallowly.
220   * New children will replace existing children. Unlike React.addons.cloneWithProps, key and
221   * ref from the original element will be preserved. There is no special behavior for merging
222   * any props (unlike cloneWithProps).
223   *
224   * @param element the element to clone
225   * @param props   the props to merge
226   * @return the cloned element
227   */
228  @JsOverlay
229  public static ReactNode cloneElement( @Nonnull final ReactNode element, @Nullable JsPropertyMap<Object> props )
230  {
231    if ( ReactConfig.shouldCheckInvariants() )
232    {
233      invariant( () -> isValidElement( element ), () -> "React.cloneElement() passed a non ReactElement" );
234    }
235    return cloneElement0( element, props );
236  }
237
238  /**
239   * Internal method that delegates to reacts cloneElement method.
240   *
241   * @param element the element to clone
242   * @param props   the props to merge
243   * @return the cloned element
244   */
245  @JsMethod( name = "cloneElement" )
246  private static native ReactNode cloneElement0( @Nonnull ReactNode element, @Nullable JsPropertyMap<Object> props );
