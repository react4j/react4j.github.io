001package react4j.core;
002
003import elemental2.core.JsError;
004import java.util.Objects;
005import javax.annotation.Nonnull;
006import javax.annotation.Nullable;
007import jsinterop.annotations.JsFunction;
008import jsinterop.base.Js;
009import jsinterop.base.JsPropertyMap;
010import static org.realityforge.braincheck.Guards.*;
011
012/**
013 * The base java class that mirrors the react component.
014 */
015public abstract class Component
016{
017  /**
018   * Callback function for updating state.
019   * Useful if the state update is based on current state.
020   */
021  @JsFunction
022  @FunctionalInterface
023  public interface SetStateCallback
024  {
025    /**
026     * Callback used to update state.
027     * Result is merged into existing state.
028     * Return null to abort state update.
029     *
030     * @param previousState the preiovus state.
031     * @param currentProps  the current props.
032     * @return the state to shallow merge or null to abort state update.
033     */
034    @Nullable
035    JsPropertyMap<Object> onSetState( @Nullable JsPropertyMap<Object> previousState,
036                                      @Nullable JsPropertyMap<Object> currentProps );
037  }
038
039  @Nonnull
040  private ComponentPhase _phase = ComponentPhase.INITIALIZING;
041  @Nonnull
042  private LifecycleMethod _lifecycleMethod = LifecycleMethod.UNKNOWN;
043  @Nullable
044  private NativeComponent _nativeComponent;
045
046  /**
047   * Set the phase of the component. Only used for invariant checking.
048   */
049  final void setPhase( @Nonnull final ComponentPhase phase )
050  {
051    invariant( ReactConfig::checkComponentStateInvariants,
052               () -> "Component.setComponentPhase() invoked on " + this +
053                     " when ReactConfig.checkComponentStateInvariants() is false" );
054    _phase = Objects.requireNonNull( phase );
055  }
056
057  /**
058   * Set the current lifecycle method of the component. Only used for invariant checking.
059   */
060  final void setLifecycleMethod( @Nonnull final LifecycleMethod lifecycleMethod )
061  {
062    invariant( ReactConfig::checkComponentStateInvariants,
063               () -> "Component.setLifecycleMethod() invoked on " + this +
064                     " when ReactConfig.checkComponentStateInvariants() is false" );
065    _lifecycleMethod = Objects.requireNonNull( lifecycleMethod );
066  }
067
068  final void bindComponent( @Nonnull final NativeComponent nativeComponent )
069  {
070    _nativeComponent = Objects.requireNonNull( nativeComponent );
071  }
072
073  /**
074   * Set the initial state of the component.
075   * This should only be invoked when the component is initializing.
076   * Calling this at any other time is an error.
077   *
078   * @param state the state.
079   */
080  protected final void setInitialState( @Nonnull final JsPropertyMap<Object> state )
081  {
082    if ( ReactConfig.checkComponentStateInvariants() )
083    {
084      apiInvariant( () -> ComponentPhase.INITIALIZING == _phase,
085                    () -> "Attempted to invoke setInitialState on " + this + " when component is " +
086                          "not in INITIALIZING phase but in phase " + _phase + " and state " + _lifecycleMethod );
087    }
088    component().setInitialState( state );
089  }
090
091  /**
092   * Return the native react component.
093   */
094  @Nonnull
095  private NativeComponent component()
096  {
097    invariant( () -> null != _nativeComponent,
098               () -> "Invoked component() on " + this + " before a component has been bound." );
099    assert null != _nativeComponent;
100    return _nativeComponent;
101  }
102
103  /**
104   * Return the component state from the native component.
105   * This may be null if initial state was never set.
106   *
107   * @return the component state.
108   */
109  protected final JsPropertyMap<Object> state()
110  {
111    return component().state();
112  }
113
114  /**
115   * Return the component props from the native component.
116   * This may be null if no props were supplied.
117   *
118   * @return the component state.
119   */
120  protected final JsPropertyMap<Object> props()
121  {
122    return component().props();
123  }
124
125  /**
126   * Return the key associated with the component if any.
127   * The key is used by the react reconcilliation process as a means to identify components so that they
128   * can be moved rather than recreated or reconfigured. Typically, it should not be used by the component
129   * itself except for generating keys of sub-components. This value may not be specified in which case null
130   * will be returned.
131   *
132   * @return the key if specified.
133   */
134  @Nullable
135  protected final String getKey()
136  {
137    return Js.asPropertyMap( props() ).getAny( "key" ).asString();
138  }
139
140  /**
141   * Schedule a shallow merge of supplied state into current state.
142   * This will trigger an update cycle and is the primary method you
143   * use to trigger UI updates from event handlers and server request callbacks.
144   *
145   * @param state the object literal representing state.
146   */
147  protected final void scheduleStateUpdate( @Nonnull final JsPropertyMap<Object> state )
148  {
149    scheduleStateUpdate( ( p, s ) -> state );
150  }
151
152  /**
153   * Schedule a shallow merge of supplied state into current state.
154   * This will trigger an update cycle and is the primary method you
155   * use to trigger UI updates from event handlers and server request callbacks.
156   *
157   * @param state                 the object literal representing state.
158   * @param onStateUpdateComplete a callback that will be invoked after state has been updated.
159   */
160  protected final void scheduleStateUpdate( @Nonnull final JsPropertyMap<Object> state,
161                                            @Nullable final Procedure onStateUpdateComplete )
162  {
163    scheduleStateUpdate( ( p, s ) -> state, onStateUpdateComplete );
164  }
165
166  /**
167   * Schedule a shallow merge of supplied state into current state.
168   * This will trigger an update cycle and is the primary method you
169   * use to trigger UI updates from event handlers and server request callbacks.
170   *
171   * @param callback the callback that will be invoked to update state.
172   */
173  protected final void scheduleStateUpdate( @Nonnull final SetStateCallback callback )
174  {
175    scheduleStateUpdate( callback, null );
176  }
177
178  /**
179   * Schedule a shallow merge of supplied state into current state.
180   * This will trigger an update cycle and is the primary method you
181   * use to trigger UI updates from event handlers and server request callbacks.
182   *
183   * @param callback              the callback that will be invoked to update state.
184   * @param onStateUpdateComplete a callback that will be invoked after state has been updated.
185   */
186  protected void scheduleStateUpdate( @Nonnull final SetStateCallback callback,
187                                      @Nullable final Procedure onStateUpdateComplete )
188  {
189    invariantsSetState();
190    component().setState( callback, onStateUpdateComplete );
191  }
192
193  /**
194   * Check invariants that should be true when invoking setState()
195   */
196  private void invariantsSetState()
197  {
198    if ( ReactConfig.checkComponentStateInvariants() )
199    {
200      apiInvariant( () -> LifecycleMethod.RENDER != _lifecycleMethod,
201                    () -> "Incorrectly invoked scheduleStateUpdate() on " + this + " in scope of render()." );
202      apiInvariant( () -> ComponentPhase.UNMOUNTING != _phase,
203                    () -> "Incorrectly invoked scheduleStateUpdate() on " + this + " when component is " +
204                          "unmounting or has unmounted." );
205    }
206  }
207
208  /**
209   * Schedule this component for re-rendering.
210   * The component re-renders when state or props change but calling this method is another way to
211   * schedule the component to be re-rendered.
212   *
213   * <p>If the force parameter is true then the {@link #shouldComponentUpdate(JsPropertyMap, JsPropertyMap)} will be skipped
214   * and it is equivalent to calling forceUpdate() on the native react component. See the
215   * <a href="https://reactjs.org/docs/react-component.html#forceupdate">React Component documentation</a> for more
216   * details.</p>
217   *
218   * <p>If the force parameter is true then the {@link #shouldComponentUpdate(JsPropertyMap, JsPropertyMap)} will be
219   * invoked. This is equivalent to calling setState({}) on the native react component.</p>
220   *
221   * @param force true to skip shouldComponentUpdate during re-render, false otherwise.
222   */
223  protected final void scheduleRender( final boolean force )
224  {
225    if ( ReactConfig.checkComponentStateInvariants() )
226    {
227      apiInvariant( () -> ComponentPhase.UNMOUNTING != _phase,
228                    () -> "Incorrectly invoked scheduleRender() on " + this + " when component is " +
229                          "unmounting or has unmounted." );
230    }
231    if ( force )
232    {
233      component().forceUpdate();
234    }
235    else
236    {
237      // This schedules a re-render but will not skip shouldComponentUpdate
238      scheduleStateUpdate( JsPropertyMap.of() );
239    }
240  }
241
242  /**
243   * Render the component.
244   * See the <a href="https://reactjs.org/docs/react-component.html#render">React Component documentation</a> for more details.
245   *
246   * @return the result of rendering.
247   */
248  @Nullable
249  protected abstract ReactNode render();
250
251  /**
252   * Wrapper method for rendering the component.
253   * By default it just delegates to the {@link #render()} method.
254   * This method exists to give middleware a mechanism of wrapping calls to render
255   *
256   * @return the result of rendering.
257   */
258  @Nullable
259  protected ReactNode performRender()
260  {
261    return render();
262  }
263
264  /**
265   * Wrapper method for constructing the component.
266   * By default it just delegates to the {@link #postConstruct()} method.
267   * This method exists to give middleware a mechanism to hook into component construction lifecycle.
268   */
269  protected void performPostConstruct()
270  {
271    postConstruct();
272  }
273
274  /**
275   * This method is invoked after the component is constructed and bound to a native react component.
276   * This is a good place to perform initialization. It is called before {@link #render()}, therefore
277   * setting state in this method will not trigger a re-rendering. This replaces the
278   * <a href="https://reactjs.org/docs/react-component.html#componentwillmount">componentWillMount</a>
279   * lifecycle method from react as well as the code that appears in constructors in native React ES6
280   * components.
281   */
282  protected void postConstruct()
283  {
284  }
285
286  /**
287   * This method is invoked after a component is attached to the DOM.
288   * Initialization that requires DOM nodes should go here.
289   * Setting state in this method will trigger a re-rendering.
290   * See the <a href="https://reactjs.org/docs/react-component.html#componentdidmount">React Component documentation</a> for more details.
291   */
292  protected void componentDidMount()
293  {
294  }
295
296  /**
297   * This method is invoked immediately after updating occurs.
298   * If you need to interact with the DOM after the component has been updated.
299   * See the <a href="https://reactjs.org/docs/react-component.html#componentdidupdate">React Component documentation</a> for more details.
300   *
301   * @param prevProps the props before the component was updated.
302   * @param prevState the state before the component was updated.
303   */
304  protected void componentDidUpdate( @Nullable final JsPropertyMap<Object> prevProps,
305                                     @Nullable final JsPropertyMap<Object> prevState )
306  {
307  }
308
309  /**
310   * This method is invoked before a mounted component receives new props.
311   * If you need to update the state in response to prop changes (for example, to reset it), you
312   * may compare the {@link #props()} and supplied nextProps and perform state transitions using
313   * {@link #scheduleStateUpdate(JsPropertyMap)} in this method.
314   * See the <a href="https://reactjs.org/docs/react-component.html#componentwillreceiveprops">React Component documentation</a> for more details.
315   *
316   * <p>Note that React may call this method even if the props have not changed, so make sure to
317   * compare the current and next values if you only want to handle changes. This may occur when the
318   * parent component causes your component to re-render.</p>
319   *
320   * <p>React doesn't call this method with initial props during mounting. It only calls this method
321   * if some of component's props may update. Calling {@link #scheduleStateUpdate(JsPropertyMap)}
322   * generally doesn't trigger this method.</p>
323   *
324   * @param nextProps the new properties of the component.
325   */
326  protected void componentWillReceiveProps( @Nonnull final JsPropertyMap<Object> nextProps )
327  {
328  }
329
330  /**
331   * This method is invoked immediately before a component is unmounted and destroyed.
332   * Perform any necessary cleanup in this method, such as invalidating timers, canceling network requests, or cleaning up
333   * any DOM elements that were created in {@link #componentDidMount()}
334   * See the <a href="https://reactjs.org/docs/react-component.html#componentwillunmount">React Component documentation</a> for more details.
335   */
336  protected void componentWillUnmount()
337  {
338  }
339
340  /**
341   * The componentDidCatch() method works like a JavaScript catch {} block, but for components.
342   * Only class components can be error boundaries. In practice, most of the time you’ll want to
343   * declare an error boundary component once and use it throughout your application.
344   *
345   * <p>Note that error boundaries only catch errors in the components below them in the tree. An
346   * error boundary can’t catch an error within itself. If an error boundary fails trying to render
347   * the error message, the error will propagate to the closest error boundary above it. This, too,
348   * is similar to how catch {} block works in JavaScript.</p>
349   *
350   * @param error the error that has been thrown.
351   * @param info  information about component stack during thrown error.
352   */
353  protected void componentDidCatch( @Nonnull final JsError error, @Nonnull final ReactErrorInfo info )
354  {
355  }
356
357  /**
358   * Use this method to let React know if a component's output is not affected
359   * by the current change in state or props. The default behavior is to re-render on
360   * every state change, and in the vast majority of cases you should rely on the default behavior.
361   * See the <a href="https://reactjs.org/docs/react-component.html#shouldcomponentupdate">React Component documentation</a> for more details.
362   *
363   * <p>This method is invoked before rendering when new props or state are being received.
364   * This method is not called for the initial render or when {@link #scheduleRender(boolean)}} is called
365   * with the force parameter set to true.</p>
366   *
367   * <p>Returning false does not prevent child components from re-rendering when their state changes.</p>
368   *
369   * <p>If this method returns false, then {@link #render()}, and {@link #componentDidUpdate(JsPropertyMap, JsPropertyMap)}
370   * will not be invoked. In the future React may treat this method  as a hint rather than a strict directive, and
371   * returning false may still result in a re-rendering of the component.</p>
372   *
373   * @param nextProps the new properties of the component.
374   * @param nextState the new state of the component.
375   * @return true if the component should be updated.
376   */
377  protected boolean shouldComponentUpdate( @Nullable final JsPropertyMap<Object> nextProps,
378                                           @Nullable final JsPropertyMap<Object> nextState )
379  {
380    return true;
381  }
