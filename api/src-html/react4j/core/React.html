001package react4j.core;
002
003import javax.annotation.Nonnull;
004import javax.annotation.Nullable;
005import jsinterop.annotations.JsOverlay;
006import jsinterop.annotations.JsPackage;
007import jsinterop.annotations.JsType;
008import jsinterop.base.Js;
009import jsinterop.base.JsPropertyMap;
010
011/**
012 * Native interface to native runtime for creating component.
013 */
014@JsType( isNative = true, namespace = JsPackage.GLOBAL )
015public final class React
016{
017  private React()
018  {
019  }
020
021  /**
022   * Create a ReactElement for the specified React component with no props or children.
023   *
024   * @param type the constructor function for the native React component.
025   * @return a new ReactElement.
026   */
027  @JsOverlay
028  public static ReactElement<ComponentConstructorFunction> createElement( @Nonnull final ComponentConstructorFunction type )
029  {
030    return createElement( type, null );
031  }
032
033  /**
034   * Create a ReactElement for the specified React component with no children.
035   *
036   * @param type  the constructor function for the native React component.
037   * @param props the props to pass to the component.
038   * @return a new ReactElement.
039   */
040  @JsOverlay
041  public static ReactElement<ComponentConstructorFunction> createElement( @Nonnull final ComponentConstructorFunction type,
042                                                                          @Nullable final JsPropertyMap<Object> props )
043  {
044    // Need to pass through undefined to react otherwise the debugger tool displays
045    // children as null rather than omitting children
046    return createElement( type, props, Js.uncheckedCast( Js.undefined() ) );
047  }
048
049  /**
050   * Create a ReactElement for the specified React component.
051   *
052   * @param type  the constructor function for the native React component.
053   * @param props the props to pass to the component.
054   * @param child the child/children of the react component.
055   * @return a new ReactElement.
056   */
057  public static native ReactElement<ComponentConstructorFunction> createElement( @Nonnull ComponentConstructorFunction type,
058                                                                                 @Nullable JsPropertyMap<Object> props,
059                                                                                 @Nullable ReactNode child );
060
061  /**
062   * Clone and return a new ReactElement using element as the starting point. The resulting
063   * element will have the original element's props with the new props merged in shallowly.
064   * New children will replace existing children. Unlike React.addons.cloneWithProps, key and
065   * ref from the original element will be preserved. There is no special behavior for merging
066   * any props (unlike cloneWithProps).
067   *
068   * @param <T>     the type of the component.
069   * @param element the element to clone
070   * @param props   the props to merge
071   * @return the cloned element
072   */
073  public static native <T>
074  ReactElement<T> cloneElement( @Nonnull ReactElement<T> element, @Nullable JsPropertyMap<Object> props );
075}
