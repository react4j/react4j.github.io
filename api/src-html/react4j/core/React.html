001package react4j.core;
002
003import javax.annotation.Nonnull;
004import javax.annotation.Nullable;
005import jsinterop.annotations.JsOverlay;
006import jsinterop.annotations.JsPackage;
007import jsinterop.annotations.JsType;
008import jsinterop.base.Js;
009
010/**
011 * Native interface to native runtime for creating component.
012 */
013@JsType( isNative = true, namespace = JsPackage.GLOBAL )
014public final class React
015{
016  private React()
017  {
018  }
019
020  /**
021   * Create a ReactElement for the specified React component with no props or children.
022   *
023   * @param <P>  the type of the component props.
024   * @param <C>  the type of the component context.
025   * @param type the constructor function for the native React component.
026   * @return a new ReactElement.
027   */
028  @JsOverlay
029  public static <
030    P extends BaseProps,
031    C extends BaseContext
032    >
033  ReactElement<P, ComponentConstructorFunction<P, C>> createElement( @Nonnull final ComponentConstructorFunction<P, C> type )
034  {
035    return createElement( type, null );
036  }
037
038  /**
039   * Create a ReactElement for the specified React component with no children.
040   *
041   * @param <P>   the type of the component props.
042   * @param <C>   the type of the component context.
043   * @param type  the constructor function for the native React component.
044   * @param props the props to pass to the component.
045   * @return a new ReactElement.
046   */
047  @JsOverlay
048  public static <
049    P extends BaseProps,
050    C extends BaseContext
051    >
052  ReactElement<P, ComponentConstructorFunction<P, C>> createElement( @Nonnull final ComponentConstructorFunction<P, C> type,
053                                                                     @Nullable final P props )
054  {
055    // Need to pass through undefined to react otherwise the debugger tool displays
056    // children as null rather than omitting children
057    return createElement( type, props, Js.uncheckedCast( Js.undefined() ) );
058  }
059
060  /**
061   * Create a ReactElement for the specified React component.
062   *
063   * @param <P>   the type of the component props.
064   * @param <C>   the type of the component context.
065   * @param type  the constructor function for the native React component.
066   * @param props the props to pass to the component.
067   * @param child the child/children of the react component.
068   * @return a new ReactElement.
069   */
070  public static native <
071    P extends BaseProps,
072    C extends BaseContext
073    >
074  ReactElement<P, ComponentConstructorFunction<P, C>> createElement( @Nonnull ComponentConstructorFunction<P, C> type,
075                                                                     @Nullable P props,
076                                                                     @Nullable ReactNode child );
077
078  /**
079   * Clone and return a new ReactElement using element as the starting point. The resulting
080   * element will have the original element's props with the new props merged in shallowly.
081   * New children will replace existing children. Unlike React.addons.cloneWithProps, key and
082   * ref from the original element will be preserved. There is no special behavior for merging
083   * any props (unlike cloneWithProps).
084   *
085   * @param <P>     the type of the component props.
086   * @param <T>     the type of the component.
087   * @param element the element to clone
088   * @param props   the props to merge
089   * @return the cloned element
090   */
091  public static native <P extends BaseProps, T>
092  ReactElement<P, T> cloneElement( @Nonnull ReactElement<P, T> element, @Nullable P props );
093}
