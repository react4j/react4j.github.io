001package react4j.core;
002
003import elemental2.core.JsArray;
004import java.util.List;
005import java.util.function.IntFunction;
006import java.util.stream.Stream;
007import javax.annotation.Nonnull;
008import javax.annotation.Nullable;
009import jsinterop.annotations.JsMethod;
010import jsinterop.annotations.JsOverlay;
011import jsinterop.annotations.JsPackage;
012import jsinterop.annotations.JsProperty;
013import jsinterop.annotations.JsType;
014import jsinterop.base.Js;
015import jsinterop.base.JsPropertyMap;
016import static org.realityforge.braincheck.Guards.*;
017
018/**
019 * Native interface to native runtime for creating component.
020 */
021@JsType( isNative = true, namespace = JsPackage.GLOBAL )
022public final class React
023{
024  private React()
025  {
026  }
027
028  /**
029   * The magic component constructor function for Fragments.
030   */
031  @SuppressWarnings( "unused" )
032  @JsProperty( name = "Fragment" )
033  private static ComponentConstructorFunction Fragment;
034
035  /**
036   * The magic component constructor function for StrictMode components.
037   */
038  @SuppressWarnings( "unused" )
039  @JsProperty( name = "StrictMode" )
040  private static ComponentConstructorFunction StrictMode;
041
042  /**
043   * Return true if the specified node is a ReactElement.
044   *
045   * @return true if the specified node is a ReactElement.
046   */
047  public static native boolean isValidElement( @Nonnull ReactNode node );
048
049  /**
050   * Create a ReactElement for the specified React component with no props or children.
051   *
052   * @param type the constructor function for the native React component.
053   * @return a new ReactElement.
054   */
055  @JsOverlay
056  public static ReactNode createElement( @Nonnull final ComponentConstructorFunction type )
057  {
058    return createElement( type, null );
059  }
060
061  /**
062   * Create a ReactElement for the specified React component with no children.
063   *
064   * @param type  the constructor function for the native React component.
065   * @param props the props to pass to the component.
066   * @return a new ReactElement.
067   */
068  @JsOverlay
069  public static ReactNode createElement( @Nonnull final ComponentConstructorFunction type,
070                                         @Nullable final JsPropertyMap<Object> props )
071  {
072    // Need to pass through undefined to react otherwise the debugger tool displays
073    // children as null rather than omitting children
074    return createElement( type, props, Js.<ReactNode>uncheckedCast( Js.undefined() ) );
075  }
076
077  /**
078   * Create a ReactElement for the specified React component.
079   *
080   * @param type  the constructor function for the native React component.
081   * @param props the props to pass to the component.
082   * @param child the child of the react component.
083   * @return a new ReactElement.
084   */
085  public static native ReactNode createElement( @Nonnull ComponentConstructorFunction type,
086                                                @Nullable JsPropertyMap<Object> props,
087                                                @Nullable ReactNode child );
088
089  /**
090   * Create a ReactElement for the specified React component.
091   *
092   * @param type     the constructor function for the native React component.
093   * @param props    the props to pass to the component.
094   * @param children the children of the react component.
095   * @return a new ReactElement.
096   */
097  @JsOverlay
098  public static ReactNode createElement( @Nonnull final ComponentConstructorFunction type,
099                                         @Nullable final JsPropertyMap<Object> props,
100                                         @Nonnull final JsArray<ReactNode> children )
101  {
102    return createElement( type, props, Js.<ReactNode[]>cast( children ) );
103  }
104
105  /**
106   * Create a ReactElement for the specified React component.
107   *
108   * @param type     the constructor function for the native React component.
109   * @param props    the props to pass to the component.
110   * @param children the children of the react component.
111   * @return a new ReactElement.
112   */
113  public static native ReactNode createElement( @Nonnull ComponentConstructorFunction type,
114                                                @Nullable JsPropertyMap<Object> props,
115                                                @Nonnull ReactNode... children );
116
117  /**
118   * Create a StrictMode component with the specified children.
119   *
120   * @param children the child nodes.
121   * @return a new React.Fragment object.
122   */
123  @JsOverlay
124  public static ReactNode createStrictMode( @Nonnull final ReactNode... children )
125  {
126    return createElement( StrictMode, null, children );
127  }
128
129  /**
130   * Create a Fragment with the specified children.
131   *
132   * @param children the child nodes.
133   * @return a new React.Fragment object.
134   */
135  @JsOverlay
136  public static ReactNode createFragment( @Nonnull final ReactNode... children )
137  {
138    return createElement( Fragment, null, children );
139  }
140
141  /**
142   * Create a Fragment with the specified children.
143   *
144   * @param children the child nodes.
145   * @return a new React.Fragment object.
146   * @see #createElement(ComponentConstructorFunction, JsPropertyMap, ReactNode...)
147   */
148  @JsOverlay
149  public static ReactNode createFragment( @Nonnull final JsArray<ReactNode> children )
150  {
151    return createFragment( Js.<ReactNode[]>cast( children ) );
152  }
153
154  /**
155   * Create a Fragment with the specified children.
156   *
157   * @param children the child nodes.
158   * @return a new React.Fragment object.
159   * @see #createElement(ComponentConstructorFunction, JsPropertyMap, ReactNode...)
160   */
161  @JsOverlay
162  public static ReactNode createFragment( @Nonnull final List<? extends ReactNode> children )
163  {
164    return createFragment( children.stream() );
165  }
166
167  /**
168   * Create a Fragment with the specified children.
169   *
170   * @param children the child nodes.
171   * @return a new React.Fragment object.
172   * @see #createElement(ComponentConstructorFunction, JsPropertyMap, ReactNode...)
173   */
174  @JsOverlay
175  public static ReactNode createFragment( @Nonnull final Stream<? extends ReactNode> children )
176  {
177    /*
178     * The GWT compiler does not handle method refs in this context. Not sure why
179     */
180    @SuppressWarnings( "Convert2MethodRef" )
181    final IntFunction<ReactNode[]> intFunction = v -> new ReactNode[ v ];
182    return createFragment( children.toArray( intFunction ) );
183  }
184
185  /**
186   * Creates a context with specified default value.
187   *
188   * @param <T>          the type of the context.
189   * @param defaultValue the default value.
190   * @return the context.
191   */
192  @Nonnull
193  public static native <T> Context<T> createContext( T defaultValue );
194
195  /**
196   * Clone and return a new ReactElement using element as the starting point. The resulting
197   * element will have the original element's props with the new props merged in shallowly.
198   * New children will replace existing children. Unlike React.addons.cloneWithProps, key and
199   * ref from the original element will be preserved. There is no special behavior for merging
200   * any props (unlike cloneWithProps).
201   *
202   * @param element the element to clone
203   * @param props   the props to merge
204   * @return the cloned element
205   */
206  @JsOverlay
207  public static ReactNode cloneElement( @Nonnull final ReactNode element, @Nullable JsPropertyMap<Object> props )
208  {
209    if ( ReactConfig.shouldCheckInvariants() )
210    {
211      invariant( () -> isValidElement( element ), () -> "React.cloneElement() passed a non ReactElement" );
212    }
213    return cloneElement0( element, props );
214  }
215
216  /**
217   * Internal method that delegates to reacts cloneElement method.
218   *
219   * @param element the element to clone
220   * @param props   the props to merge
221   * @return the cloned element
222   */
223  @JsMethod( name = "cloneElement" )
224  private static native ReactNode cloneElement0( @Nonnull ReactNode element, @Nullable JsPropertyMap<Object> props );
225}
