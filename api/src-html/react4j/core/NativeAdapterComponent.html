001package react4j.core;
002
003import elemental2.core.JsError;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import jsinterop.base.JsPropertyMap;
007
008/**
009 * This class provides a base class that designed to forward all lifecycle methods to a target component.
010 * The target component is of type {@link Component}. This class is also responsible for calling
011 * {@link Component#setLifecycleMethod(LifecycleMethod)} before and after each call to the target component
012 * if {@link ReactConfig#checkComponentStateInvariants()} returns true. This will make it possible for the
013 * target component to check that the application code is correctly interacting with the React component
014 * framework. In production builds it is expected that the method calls will be inlined and the checks will
015 * be optimized away, having no significant performance impact.
016 */
017public abstract class NativeAdapterComponent<
018  S extends BaseState,
019  I extends Component<S>
020  >
021  extends NativeComponent<S>
022{
023  /**
024   * The target component that all lifecycle methods are forwarded to.
025   */
026  private final I _component;
027
028  /**
029   * Create a component that designed to delegate to a target component.
030   *
031   * @param props the initial props.
032   */
033  protected NativeAdapterComponent( @Nullable final JsPropertyMap<Object> props )
034  {
035    super( props );
036    _component = createComponent();
037    _component.bindComponent( this );
038
039    performPostConstruct();
040  }
041
042  /**
043   * Template method that actually creates the target component.
044   *
045   * @return a new instance of the target component.
046   */
047  protected abstract I createComponent();
048
049  /**
050   * Initialize the target component.
051   */
052  private void performPostConstruct()
053  {
054    if ( ReactConfig.checkComponentStateInvariants() )
055    {
056      _component.setLifecycleMethod( LifecycleMethod.COMPONENT_POST_CONSTRUCT );
057      _component.setPhase( ComponentPhase.INITIALIZING );
058    }
059    try
060    {
061      _component.performPostConstruct();
062    }
063    finally
064    {
065      if ( ReactConfig.checkComponentStateInvariants() )
066      {
067        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
068        _component.setPhase( ComponentPhase.MOUNTING );
069      }
070    }
071  }
072
073  /**
074   * Call render on the target component.
075   *
076   * @return the output of rendering.
077   * @see Component#render()
078   */
079  @Nullable
080  public final ReactNode render()
081  {
082    if ( ReactConfig.checkComponentStateInvariants() )
083    {
084      _component.setLifecycleMethod( LifecycleMethod.RENDER );
085    }
086    try
087    {
088      return _component.performRender();
089    }
090    finally
091    {
092      if ( ReactConfig.checkComponentStateInvariants() )
093      {
094        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
095      }
096    }
097  }
098
099  /**
100   * Call componentDidMount on the target component.
101   * It is expected that the subclass will implement a public method componentDidMount() that
102   * delegates to this method to perform the work.
103   *
104   * @see Component#componentDidMount()
105   */
106  protected final void performComponentDidMount()
107  {
108    if ( ReactConfig.checkComponentStateInvariants() )
109    {
110      _component.setLifecycleMethod( LifecycleMethod.COMPONENT_DID_MOUNT );
111    }
112    try
113    {
114      _component.componentDidMount();
115    }
116    finally
117    {
118      if ( ReactConfig.checkComponentStateInvariants() )
119      {
120        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
121        _component.setPhase( ComponentPhase.UPDATING );
122      }
123    }
124  }
125
126  /**
127   * Call componentWillUnmount on the target component.
128   * It is expected that the subclass will implement a public method componentWillUnmount() that
129   * delegates to this method to perform the work.
130   *
131   * @see Component#componentWillUnmount()
132   */
133  protected final void performComponentWillUnmount()
134  {
135    if ( ReactConfig.checkComponentStateInvariants() )
136    {
137      _component.setPhase( ComponentPhase.UNMOUNTING );
138      _component.setLifecycleMethod( LifecycleMethod.COMPONENT_WILL_UNMOUNT );
139    }
140    try
141    {
142      _component.componentWillUnmount();
143    }
144    finally
145    {
146      if ( ReactConfig.checkComponentStateInvariants() )
147      {
148        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
149      }
150    }
151  }
152
153  /**
154   * Call componentWillUpdate on the target component.
155   * It is expected that the subclass will implement a public method componentWillUpdate() that
156   * delegates to this method to perform the work.
157   *
158   * @param nextProps the new properties of the component.
159   * @param nextState the state.
160   * @see Component#componentWillUpdate(JsPropertyMap, BaseState)
161   */
162  protected final void performComponentWillUpdate( @Nonnull final JsPropertyMap<Object> nextProps,
163                                                   @Nonnull final S nextState )
164  {
165    if ( ReactConfig.checkComponentStateInvariants() )
166    {
167      _component.setLifecycleMethod( LifecycleMethod.COMPONENT_WILL_UPDATE );
168    }
169    try
170    {
171      _component.componentWillUpdate( nextProps, nextState );
172    }
173    finally
174    {
175      if ( ReactConfig.checkComponentStateInvariants() )
176      {
177        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
178      }
179    }
180  }
181
182  /**
183   * Call shouldComponentUpdate on the target component.
184   * It is expected that the subclass will implement a public method shouldComponentUpdate() that
185   * delegates to this method to perform the work.
186   *
187   * @param nextProps the new properties of the component.
188   * @param nextState the new state of the component.
189   * @return true if the component should be updated.
190   * @see Component#shouldComponentUpdate(JsPropertyMap, BaseState)
191   */
192  protected final boolean performShouldComponentUpdate( @Nonnull final JsPropertyMap<Object> nextProps,
193                                                        @Nonnull final S nextState )
194  {
195    if ( ReactConfig.checkComponentStateInvariants() )
196    {
197      _component.setLifecycleMethod( LifecycleMethod.SHOULD_COMPONENT_UPDATE );
198    }
199    try
200    {
201      return _component.shouldComponentUpdate( nextProps, nextState );
202    }
203    finally
204    {
205      if ( ReactConfig.checkComponentStateInvariants() )
206      {
207        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
208      }
209    }
210  }
211
212  /**
213   * Call componentWillReceiveProps on the target component.
214   * It is expected that the subclass will implement a public method componentWillReceiveProps() that
215   * delegates to this method to perform the work.
216   *
217   * @param nextProps the new properties of the component.
218   * @see Component#componentWillReceiveProps(JsPropertyMap)
219   */
220  protected final void performComponentWillReceiveProps( @Nonnull final JsPropertyMap<Object> nextProps )
221  {
222    if ( ReactConfig.checkComponentStateInvariants() )
223    {
224      _component.setLifecycleMethod( LifecycleMethod.COMPONENT_WILL_RECEIVE_PROPS );
225    }
226    try
227    {
228      _component.componentWillReceiveProps( nextProps );
229    }
230    finally
231    {
232      if ( ReactConfig.checkComponentStateInvariants() )
233      {
234        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
235      }
236    }
237  }
238
239  /**
240   * Call componentDidUpdate on the target component.
241   * It is expected that the subclass will implement a public method componentDidUpdate() that
242   * delegates to this method to perform the work.
243   *
244   * @param nextProps the new properties of the component.
245   * @param nextState the new state of the component.
246   * @see Component#componentDidUpdate(JsPropertyMap, BaseState)
247   */
248  protected final void performComponentDidUpdate( @Nonnull final JsPropertyMap<Object> nextProps,
249                                                  @Nonnull final S nextState )
250  {
251    if ( ReactConfig.checkComponentStateInvariants() )
252    {
253      _component.setLifecycleMethod( LifecycleMethod.COMPONENT_DID_UPDATE );
254    }
255    try
256    {
257      _component.componentDidUpdate( nextProps, nextState );
258    }
259    finally
260    {
261      if ( ReactConfig.checkComponentStateInvariants() )
262      {
263        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
264      }
265    }
266  }
267
268  /**
269   * Call componentDidCatch on the target component.
270   * It is expected that the subclass will implement a public method componentDidCatch() that
271   * delegates to this method to perform the work.
272   *
273   * @param error the error that has been thrown.
274   * @param info  information about component stack during thrown error.
275   * @see Component#componentDidCatch(JsError, ReactErrorInfo)
276   */
277  protected final void performComponentDidCatch( @Nonnull final JsError error, @Nonnull final ReactErrorInfo info )
278  {
279    if ( ReactConfig.checkComponentStateInvariants() )
280    {
281      _component.setLifecycleMethod( LifecycleMethod.COMPONENT_DID_CATCH );
282    }
283    try
284    {
285      _component.componentDidCatch( error, info );
286    }
287    finally
288    {
289      if ( ReactConfig.checkComponentStateInvariants() )
290      {
291        _component.setLifecycleMethod( LifecycleMethod.UNKNOWN );
292      }
293    }
294  }
