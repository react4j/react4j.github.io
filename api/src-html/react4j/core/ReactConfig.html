001package react4j.core;
002
003/**
004 * Location of all compile time configuration settings for framework.
005 */
006public final class ReactConfig
007{
008  private static final boolean PRODUCTION_MODE =
009    System.getProperty( "react4j.environment", "production" ).equals( "production" );
010  private static boolean ENABLE_NAMES =
011    "true".equals( System.getProperty( "arez.enable_component_names", PRODUCTION_MODE ? "false" : "true" ) );
012  private static final boolean CHECK_COMPONENT_STATE_INVARIANTS =
013    "true".equals( System.getProperty( "react4j.check_component_state_invariants",
014                                       PRODUCTION_MODE ? "false" : "true" ) );
015
016  private ReactConfig()
017  {
018  }
019
020  /**
021   * Return true if in production mode.
022   * Production mode sets the default values for other compile time constants to the variant
023   * that assumes your code is correct and does not generate additional assert or debug statements.
024   * The individual configuration settings can still be specified to override this value.
025   *
026   * @return true if in production mode.
027   */
028  public static boolean isProductionMode()
029  {
030    return PRODUCTION_MODE;
031  }
032
033  /**
034   * Return true if components should have human readable names specified.
035   * Useful if you want to interact via DevTools or other tool chains.
036   *
037   * @return to enable human readable names for components.
038   */
039  public static boolean enableComponentNames()
040  {
041    return ENABLE_NAMES;
042  }
043
044  /**
045   * Return true if we should check that the user interacts with React component in a way compatible with the state.
046   * i.e. setInitialState should only be invoked at the start, setState should not be invoked while componentWillUpdate
047   * is running etc.
048   *
049   * @return true to enable invariant checking about how we interact with native react component.
050   */
051  public static boolean checkComponentStateInvariants()
052  {
053    return CHECK_COMPONENT_STATE_INVARIANTS;
054  }
055}
