001package react4j;
002
003import javax.annotation.Nonnull;
004import react4j.annotations.Prop;
005
006/**
007 * Interface implemented by objects so that they can be marked as {@link Prop#immutable()}
008 */
009public interface Keyed
010{
011  /**
012   * Return a string that will be the key or contribute to the key of a react component.
013   *
014   * @return the value used to form key.
015   */
016  @Nonnull
017  String getKey();
018}
