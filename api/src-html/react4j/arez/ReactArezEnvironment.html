001package react4j.arez;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Environment;
006import arez.Function;
007import arez.SafeFunction;
008import javax.annotation.Nonnull;
009import react4j.dom.ReactDOM;
010
011/**
012 * Utility class for configuring the Arez {@link Environment} suitable for React.
013 * The only difference is that all changes that occur as a result of Arez reactions are
014 * batched to minimize the number of changes propagated to react.
015 */
016public final class ReactArezEnvironment
017{
018  /**
019   * The underlying instance of the environment.
020   */
021  private static final Environment INSTANCE = new ReactEnvironment();
022
023  private ReactArezEnvironment()
024  {
025  }
026
027  /**
028   * Install the appropriate {@link Environment} into the current ArezContext.
029   */
030  public static void install()
031  {
032    install( Arez.context() );
033  }
034
035  /**
036   * Install the appropriate {@link Environment} into the specified ArezContext.
037   *
038   * @param context the arez context to configure environment.
039   */
040  public static void install( @Nonnull final ArezContext context )
041  {
042    context.setEnvironment( INSTANCE );
043  }
044
045  @SuppressWarnings( "unchecked" )
046  private static final class ReactEnvironment
047    implements Environment
048  {
049    /**
050     * {@inheritDoc}
051     */
052    @Override
053    public <T> T run( @Nonnull final SafeFunction<T> function )
054    {
055      final T[] result = (T[]) new Object[ 1 ];
056      ReactDOM.batchedUpdates( () -> result[ 0 ] = function.call() );
057      return result[ 0 ];
058    }
059
060    /**
061     * {@inheritDoc}
062     */
063    @Override
064    public <T> T run( @Nonnull final Function<T> function )
065      throws Throwable
066    {
067      final T[] result = (T[]) new Object[ 1 ];
068      ReactDOM.batchedUpdates( () -> result[ 0 ] = function.call() );
069      return result[ 0 ];
070    }
071  }
072}
