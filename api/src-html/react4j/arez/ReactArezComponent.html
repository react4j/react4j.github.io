008import arez.annotations.ComponentId;

010import arez.annotations.ObserverRef;
011import arez.annotations.OnDepsChanged;

013import arez.spy.ObservableInfo;
014import elemental2.core.JsObject;
015import java.util.List;
016import javax.annotation.Nonnull;
017import javax.annotation.Nullable;
018import javax.annotation.OverridingMethodsMustInvokeSuper;
019import jsinterop.base.Js;

027/**
028 * A base class for all Arez enabled components. This class makes the component
029 * rendering reactive and it will schedule a re-render any time any of the observable
030 * entities accessed within the scope of the render method are changed.
031 *
032 * <p>To achieve this goal, the props and state of the component are converted into
033 * observable properties. This of course means they must be accessed within the scope
034 * of an Arez transaction. (Typically this means it needs to be accessed within the
035 * scope of a {@link Action} annotated method or within the scope of the render method.</p>
036 */
037public abstract class ReactArezComponent
038  extends Component
039{
040  /**
041   * Key used to store the arez data in state.
042   */
043  private static final String AREZ_STATE_KEY = "arez";
044
045  private static int c_nextComponentId = 1;
046  private final int _arezComponentId;
047  private boolean _renderDepsChanged;
048
049  protected ReactArezComponent()
050  {
051    _arezComponentId = c_nextComponentId++;
052  }
053
054  /**
055   * Method invoked when props changes.
056   *
057   * @param nextProps the new properties of the component.
058   */
059  protected abstract void reportPropsChanged( @Nullable final JsPropertyMap<Object> nextProps );
060
061  /**
062   * {@inheritDoc}
063   */
064  @Override
065  protected final void scheduleStateUpdate( @Nonnull final SetStateCallback callback,
066                                            @Nullable final Procedure onStateUpdateComplete )
067  {
068    Guards.fail( () -> "Attempted to schedule state update on ReactArezComponent subclass. Use Arez @Observable or @Computed properties instead." );
069  }
070
071  /**
072   * After construction of the object. Schedule any autoruns attached to component.
073   */
074  @Override
075  protected final void performPostConstruct()
076  {
077    super.performPostConstruct();
078    triggerScheduler();
079  }
080
081  /**
082   * Template method overridden by annotation processor if there are autoruns to schedule.
083   */
084  protected void triggerScheduler()
085  {
086  }
087
088  /**
089   * Return true if the render dependencies have been marked as changed and component has yet to be re-rendered.
090   *
091   * @return true if render dependencies changed, false otherwise.
092   */
093  protected final boolean hasRenderDepsChanged()
094  {
095    return _renderDepsChanged;
096  }
097
098  /**
099   * Hook used by Arez to notify component that it needs to be re-rendered.
100   */
101  @OnDepsChanged
102  protected final void onRenderDepsChanged()
103  {
104    _renderDepsChanged = true;
105    scheduleRender( true );
106  }
107
108  /**
109   * Return the arez context that this component is associated with.
110   * The component is associated with the context that was active when it was created
111   * and can only react to observables associated with the same context.
112   *
113   * @return the arez context that this component is associated with.
114   */
115  @ContextRef
116  @Nonnull
117  protected abstract ArezContext getContext();
118
119  /**
120   * Return the unique identifier of component according to Arez.
121   * This method is invoked by the code generated by the Arez annotation processor.
122   *
123   * @return the unique identifier of component according to Arez.
124   */
125  @ComponentId
126  protected final int getArezComponentId()
127  {
128    return _arezComponentId;
129  }
130
131  /**
132   * Return the Observer associated with the render tracker method.
133   *
134   * @return the Observer associated with the render tracker method.
135   */
136  @ObserverRef
137  @Nonnull
138  protected abstract Observer getRenderObserver();
139
140  /**
141   * {@inheritDoc}
142   */
143  @Override
144  protected final ReactNode performRender()
145  {
146    /*
147     * Consider generating a more efficient check than `Disposable.isDisposed( this )`. We should be able to
148     * inline this variable. This could be done by adding @PreDispose hook to set a instance variable or somehow
149     * updating Arez to efficiently expose the variable (which may involve the generated subclass calling
150     * setDisposed() or something similar in this class).
151     */
152    return Disposable.isDisposed( this ) ? null : triggerRender();
153  }
154
155  /**
156   * {@inheritDoc}
157   */
158  @Track( name = "render" )
159  @Nullable
160  protected ReactNode triggerRender()
161  {
162    _renderDepsChanged = false;
163    return super.performRender();
164  }
165
166  /**
167   * {@inheritDoc}
168   */
169  @Override
170  @OverridingMethodsMustInvokeSuper
171  protected boolean shouldComponentUpdate( @Nullable final JsPropertyMap<Object> nextProps,
172                                           @Nullable final JsPropertyMap<Object> nextState )
173  {
174    if ( hasRenderDepsChanged() )
175    {
176      return true;
177    }
178    //noinspection SimplifiableIfStatement
179    if ( !Js.isTripleEqual( super.state(), nextState ) )
180    {
181      // If state is not identical then we need to re-render ...
182      // Previously we chose not to re-render if only AREZ_STATE_KEY that was updated but that
183      // meant deps in DevTools would not be update so now we just re-render anyway.
184      return true;
185    }
186    else
187    {
188      /*
189       * We just compare the props shallowly and avoid a re-render if the props have not changed.
190       */
191      final boolean modified = JsUtil.isObjectShallowModified( super.props(), nextProps );
192      if ( modified )
193      {
194        reportPropsChanged( nextProps );
195      }
196      return modified;
197    }
198  }
199
200  /**
201   * {@inheritDoc}
202   */
203  @Override
204  @OverridingMethodsMustInvokeSuper
205  protected void componentDidMount()
206  {
207    storeArezDataAsState();
208  }
209
210  /**
211   * {@inheritDoc}
212   */
213  @Override
214  @OverridingMethodsMustInvokeSuper
215  protected void componentDidUpdate( @Nullable final JsPropertyMap<Object> prevProps,
216                                     @Nullable final JsPropertyMap<Object> prevState )
217  {
218    storeArezDataAsState();
219  }
220
221  /**
222   * {@inheritDoc}
223   */
224  @Override
225  @OverridingMethodsMustInvokeSuper
226  protected void componentWillUnmount()
227  {
228    /*
229     * Dispose of all the arez resources. Necessary particularly for the render tracker that should
230     * not receive notifications of updates after the component has been unmounted.
231     */
232    Disposable.dispose( this );
233  }
234
235  /**
236   * Store arez data such as dependencies on the state of component.
237   * This is only done if {@link ReactArezConfig#shouldStoreArezDataAsState()} returns true and is primarily
238   * done to make it easy to debug from within React DevTools.
239   */
240  private void storeArezDataAsState()
241  {
242    if ( ReactArezConfig.shouldStoreArezDataAsState() && Arez.areSpiesEnabled() && !Disposable.isDisposed( this ) )
243    {
244      final Observer renderTracker = getRenderObserver();
245      final List<ObservableInfo> dependencies = getContext().getSpy().getDependencies( renderTracker );
246      final JsPropertyMap<Object> deps = JsPropertyMap.of();
247      dependencies.forEach( d -> deps.set( d.getName(), getValue( d ) ) );
248      final JsPropertyMap<Object> state = super.state();
249      final Object currentDepsData = null != state ? Js.asPropertyMap( state ).get( AREZ_STATE_KEY ) : null;
250      /*
251       * Do a shallow comparison against object and the deps. If either has changed then state needs to be updated.
252       * We skip deps on shallow comparison of data as it is always recreated anew.
253       */
254      if ( null == currentDepsData )
255      {
256        scheduleArezKeyUpdate( deps );
257      }
258      else
259      {
260        /*
261         * Deps are mappings to Info objects that can be garbage collected over time.
262         * So we just make sure the keys (which are the info objects names) match.
263         */
264        final String[] currentDeps = JsObject.keys( Js.uncheckedCast( currentDepsData ) );
265        final String[] newDeps = JsObject.keys( Js.uncheckedCast( deps ) );
266        if ( currentDeps.length != newDeps.length )
267        {
268          scheduleArezKeyUpdate( deps );
269        }
270        else
271        {
272          for ( int i = 0; i < currentDeps.length; i++ )
273          {
274            if ( !currentDeps[ i ].equals( newDeps[ i ] ) )
275            {
276              scheduleArezKeyUpdate( deps );
277            }
278          }
279        }
280      }
281    }
282  }
283
284  /**
285   * Return the value for specified observable.
286   * Exceptions are caught and types are converted to strings using {@link Object#toString()}
287   *
288   * @param observableInfo the observable.
289   * @return the value as a string.
290   */
291  @Nullable
292  private Object getValue( @Nonnull final ObservableInfo observableInfo )
293  {
294    try
295    {
296      //Consider unwrapping the boxed values and collections so they are presented correctly in DevTools
297      return Arez.arePropertyIntrospectorsEnabled() ? observableInfo.getValue() : "?";
298    }
299    catch ( final Throwable throwable )
300    {
301      return throwable;
302    }
303  }
304
305  /**
306   * Schedule state update the updates arez state.
307   * Makes sure the super class is invoked so reportStateChanged() is not invoked on State observable.
308   */
309  private void scheduleArezKeyUpdate( @Nonnull final JsPropertyMap<Object> data )
310  {
311    super.scheduleStateUpdate( ( p, s ) -> Js.cast( JsPropertyMap.of( AREZ_STATE_KEY, JsObject.freeze( data ) ) ),
312                               null );
313  }
