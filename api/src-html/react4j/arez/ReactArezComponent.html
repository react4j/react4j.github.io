027/**
028 * A base class for all Arez enabled components. This class makes the component
029 * rendering reactive and it will schedule a re-render any time any of the observable
030 * entities accessed within the scope of the render method are changed.
031 *
032 * <p>To achieve this goal, the props and state of the component are converted into
033 * observable properties. This of course means they must be accessed within the scope
034 * of an Arez transaction. (Typically this means it needs to be accessed within the
035 * scope of a {@link Action} annotated method or within the scope of the render method.</p>
036 */
037public abstract class ReactArezComponent
038  extends Component
039{
040  /**
041   * Key used to store the arez data in state.
042   */
043  private static final String AREZ_STATE_KEY = "arez";
044  private static int c_nextComponentId = 1;
045  private final int _arezComponentId;
046  private boolean _renderDepsChanged;
047  private boolean _unmounted;
048
049  protected ReactArezComponent()
050  {
051    _arezComponentId = c_nextComponentId++;
052  }
053
054  /**
055   * {@inheritDoc}
056   */
057  @Override
058  protected final void scheduleStateUpdate( @Nonnull final SetStateCallback callback,
059                                            @Nullable final Procedure onStateUpdateComplete )
060  {
061    fail( () -> "Attempted to schedule state update on ReactArezComponent subclass. Use Arez @Observable or @Computed properties instead." );
062  }
063
064  /**
065   * After construction of the object. Schedule any autoruns attached to component.
066   */
067  @Override
068  protected final void performPostConstruct()
069  {
070    super.performPostConstruct();
071    triggerScheduler();
072  }
073
074  /**
075   * Template method overridden by annotation processor if there are autoruns to schedule.
076   */
077  protected void triggerScheduler()
078  {
079  }
080
081  /**
082   * Return true if the render dependencies have been marked as changed and component has yet to be re-rendered.
083   *
084   * @return true if render dependencies changed, false otherwise.
085   */
086  protected final boolean hasRenderDepsChanged()
087  {
088    return _renderDepsChanged;
089  }
090
091  /**
092   * Hook used by Arez to notify component that it needs to be re-rendered.
093   */
094  @OnDepsChanged
095  protected final void onRenderDepsChanged()
096  {
097    if ( !_renderDepsChanged )
098    {
099      _renderDepsChanged = true;
100      if ( !_unmounted )
101      {
102        scheduleRender( true );
103      }
104    }
105  }
106
107  /**
108   * Return the arez context that this component is associated with.
109   * The component is associated with the context that was active when it was created
110   * and can only react to observables associated with the same context.
111   *
112   * @return the arez context that this component is associated with.
113   */
114  @ContextRef
115  @Nonnull
116  protected abstract ArezContext getContext();
117
118  /**
119   * Return the unique identifier of component according to Arez.
120   * This method is invoked by the code generated by the Arez annotation processor.
121   *
122   * @return the unique identifier of component according to Arez.
123   */
124  @ComponentId
125  protected final int getArezComponentId()
126  {
127    return _arezComponentId;
128  }
129
130  /**
131   * Return the Observer associated with the render tracker method.
132   *
133   * @return the Observer associated with the render tracker method.
134   */
135  @ObserverRef
136  @Nonnull
137  protected abstract Observer getRenderObserver();
138
139  /**
140   * {@inheritDoc}
141   */
142  @Override
143  protected final ReactNode performRender()
144  {
145    return Disposable.isDisposed( this ) ? null : trackRender();
146  }
147
148  /**
149   * Return true if any prop is an ArezComponent that has been disposed.
150   * This is used to guard against rendering a react component that has invalid props.
151   *
152   * @return true if any prop is an ArezComponent that has been disposed.
153   */
154  protected boolean anyPropsDisposed()
155  {
156    return false;
157  }
158
159  /**
160   * This method is the method enhanced by arez that performs render and tracks dependencies.
161   * This SHOULD NOT be merged with {@link #performRender()} as then the isDisposed check will be present
162   * in every instance of render method which can result in unnecessary code bloat.
163   *
164   * @return the result of rendering.
165   */
166  @Track( name = "render", priority = Priority.LOW, observeLowerPriorityDependencies = true )
167  @Nullable
168  protected ReactNode trackRender()
169  {
170    _renderDepsChanged = false;
171    if ( anyPropsDisposed() )
172    {
173      return null;
174    }
175    else
176    {
177      final ReactNode result = super.performRender();
178      if ( Arez.shouldCheckInvariants() && Arez.areSpiesEnabled() )
179      {
180        final List<ObservableValueInfo> dependencies =
181          getContext().getSpy().asObserverInfo( getRenderObserver() ).getDependencies();
182        invariant( () -> !dependencies.isEmpty(),
183                   () -> "ReactArezComponent render completed on '" + this + "' but the component does not " +
184                         "have any Arez dependencies. This component should extend react4j.Component instead." );
185      }
186      return result;
187    }
188  }
189
190  /**
191   * {@inheritDoc}
192   */
193  @SuppressWarnings( "SimplifiableIfStatement" )
194  @Override
195  protected boolean shouldComponentUpdate( @Nullable final JsPropertyMap<Object> nextProps,
196                                           @Nullable final JsPropertyMap<Object> nextState )
197  {
198    if ( hasRenderDepsChanged() )
199    {
200      return true;
201    }
202    else if ( ReactArezConfig.shouldStoreArezDataAsState() && !Js.isTripleEqual( super.state(), nextState ) )
203    {
204      // State is only updated when we store Arez data as state. The saving of arez data will actually
205      // trigger a re-render and we need to allow this otherwise the DevTools will not contain the
206      // updated state.
207      return true;
208    }
209    else
210    {
211      return shouldComponentUpdate( nextProps );
212    }
213  }
214
215  /**
216   * This method is overridden by the annotation processor. The method will return true if a prop has been updated
217   * and the prop has not set {@link Prop#shouldUpdateOnChange()} to false. Otherwise this method will return false.
218   *
219   * @param nextProps the new properties of the component.
220   * @return true if the component should be updated.
221   */
222  protected boolean shouldComponentUpdate( @Nullable final JsPropertyMap<Object> nextProps )
223  {
224    return false;
225  }
226
227  /**
228   * {@inheritDoc}
229   */
230  @Override
231  protected void componentDidMount()
232  {
233    storeArezDataAsState();
234  }
235
236  /**
237   * {@inheritDoc}
238   */
239  @Override
240  protected void componentDidUpdate( @Nullable final JsPropertyMap<Object> prevProps,
241                                     @Nullable final JsPropertyMap<Object> prevState )
242  {
243    storeArezDataAsState();
244  }
245
246  /**
247   * {@inheritDoc}
248   */
249  @Override
250  protected void componentWillUnmount()
251  {
252    /*
253     * This method is overridden as it forces the annotation processor to generate the native
254     * componentWillUnmount method. The native method is required so that performComponentWillUnmount()
255     * is invoked and correctly cleans up state.
256     */
257  }
258
259  /**
260   * {@inheritDoc}
261   */
262  protected final void performComponentWillUnmount()
263  {
264    final Disposable schedulerLock = getContext().pauseScheduler();
265    try
266    {
267      _unmounted = true;
268      super.performComponentWillUnmount();
269      /*
270       * Dispose of all the arez resources. Necessary particularly for the render tracker that should
271       * not receive notifications of updates after the component has been unmounted.
272       */
273      Disposable.dispose( this );
274    }
275    finally
276    {
277      schedulerLock.dispose();
278    }
279  }
280
281  /**
282   * Store arez data such as dependencies on the state of component.
283   * This is only done if {@link ReactArezConfig#shouldStoreArezDataAsState()} returns true and is primarily
284   * done to make it easy to debug from within React DevTools.
285   */
286  private void storeArezDataAsState()
287  {
288    if ( ReactArezConfig.shouldStoreArezDataAsState() && Arez.areSpiesEnabled() && !Disposable.isDisposed( this ) )
289    {
290      final List<ObservableValueInfo> dependencies =
291        getContext().getSpy().asObserverInfo( getRenderObserver() ).getDependencies();
292      final JsPropertyMap<Object> deps = JsPropertyMap.of();
293      dependencies.forEach( d -> deps.set( d.getName(), getValue( d ) ) );
294      final JsPropertyMap<Object> state = super.state();
295      final Object currentDepsData = null != state ? Js.asPropertyMap( state ).get( AREZ_STATE_KEY ) : null;
296      /*
297       * Do a shallow comparison against object and the deps. If either has changed then state needs to be updated.
298       * We skip deps on shallow comparison of data as it is always recreated anew.
299       */
300      if ( null == currentDepsData )
301      {
302        scheduleArezKeyUpdate( deps );
303      }
304      else
305      {
306        /*
307         * Deps are mappings to Info objects that can be garbage collected over time.
308         * So we just make sure the keys (which are the info objects names) match.
309         */
310        final String[] currentDeps = JsObject.keys( Js.uncheckedCast( currentDepsData ) );
311        final String[] newDeps = JsObject.keys( Js.uncheckedCast( deps ) );
312        if ( currentDeps.length != newDeps.length )
313        {
314          scheduleArezKeyUpdate( deps );
315        }
316        else
317        {
318          for ( int i = 0; i < currentDeps.length; i++ )
319          {
320            if ( !currentDeps[ i ].equals( newDeps[ i ] ) )
321            {
322              scheduleArezKeyUpdate( deps );
323            }
324          }
325        }
326      }
327    }
328  }
329
330  /**
331   * Return the value for specified observable.
332   * Exceptions are caught and types are converted to strings using {@link Object#toString()}
333   *
334   * @param observableInfo the observable.
335   * @return the value as a string.
336   */
337  @Nullable
338  private Object getValue( @Nonnull final ObservableValueInfo observableInfo )
339  {
340    try
341    {
342      //Consider unwrapping the boxed values and collections so they are presented correctly in DevTools
343      return Arez.arePropertyIntrospectorsEnabled() ? observableInfo.getValue() : "?";
344    }
345    catch ( final Throwable throwable )
346    {
347      return throwable;
348    }
349  }
350
351  /**
352   * Schedule state update the updates arez state.
353   * Makes sure the super class is invoked so reportStateChanged() is not invoked on State observable.
354   */
355  private void scheduleArezKeyUpdate( @Nonnull final JsPropertyMap<Object> data )
356  {
357    super.scheduleStateUpdate( ( p, s ) -> Js.cast( JsPropertyMap.of( AREZ_STATE_KEY, JsObject.freeze( data ) ) ),
358                               null );
359  }
