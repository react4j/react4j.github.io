001package react4j.arez;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Disposable;
006import arez.Observer;
007import arez.annotations.Action;
008import arez.annotations.ComponentIdRef;
009import arez.annotations.ComponentNameRef;
010import arez.annotations.ContextRef;
011import arez.annotations.ObserverRef;
012import arez.annotations.PreDispose;
013import arez.spy.ObservableValueInfo;
014import arez.spy.ObserverInfo;
015import elemental2.promise.Promise;
016import java.util.stream.Stream;
017import javax.annotation.Nonnull;
018import javax.annotation.Nullable;
019import jsinterop.base.Js;
020import jsinterop.base.JsPropertyMap;
021import react4j.Component;
022
023/**
024 * A base class for all Arez enabled components. This class makes the component
025 * rendering reactive and it will schedule a re-render any time any of the observable
026 * entities accessed within the scope of the render method are changed.
027 *
028 * <p>To achieve this goal, the props and state of the component are converted into
029 * observable properties. This of course means they must be accessed within the scope
030 * of an Arez transaction. (Typically this means it needs to be accessed within the
031 * scope of a {@link Action} annotated method or within the scope of the render method.</p>
032 */
033public abstract class ReactArezComponent
034  extends Component
035{
036  /**
037   * A non-null lock that will be released in the next micro-task which will schedule any renders required.
038   */
039  @Nullable
040  private static Disposable c_schedulerLock;
041  private boolean _renderDepsChanged;
042  private boolean _unmounted;
043
044  protected ReactArezComponent()
045  {
046  }
047
048  /**
049   * After construction of the object. Schedule any autoruns attached to component.
050   */
051  @Override
052  protected final void performPostConstruct()
053  {
054    super.performPostConstruct();
055    triggerScheduler();
056  }
057
058  /**
059   * Template method overridden by annotation processor if there are autoruns to schedule.
060   */
061  protected void triggerScheduler()
062  {
063  }
064
065  /**
066   * Return true if the render dependencies have been marked as changed and component has yet to be re-rendered.
067   *
068   * @return true if render dependencies changed, false otherwise.
069   */
070  protected final boolean hasRenderDepsChanged()
071  {
072    return _renderDepsChanged;
073  }
074
075  /**
076   * Helper method invoked when it has detected a dependency of the render method has changed.
077   *
078   * @param componentHasObservableProps flag indicating whether the component has any observable props.
079   */
080  protected final void onRenderDepsChange( final boolean componentHasObservableProps )
081  {
082    if ( !_renderDepsChanged )
083    {
084      _renderDepsChanged = true;
085      if ( !_unmounted )
086      {
087        scheduleRender( !componentHasObservableProps );
088      }
089    }
090  }
091
092  protected final void clearRenderDepsChanged()
093  {
094    _renderDepsChanged = false;
095  }
096
097  /**
098   * Return the arez context that this component is associated with.
099   * The component is associated with the context that was active when it was created
100   * and can only react to observables associated with the same context.
101   *
102   * @return the arez context that this component is associated with.
103   */
104  @ContextRef
105  @Nonnull
106  protected abstract ArezContext getContext();
107
108  /**
109   * Return the unique identifier of component according to Arez.
110   *
111   * @return the unique identifier of component according to Arez.
112   */
113  @ComponentIdRef
114  protected abstract int getArezComponentId();
115
116  /**
117   * Return the name of the component according to Arez.
118   *
119   * @return the name of the component according to Arez.
120   */
121  @ComponentNameRef
122  protected abstract String getArezComponentName();
123
124  /**
125   * Return the Observer associated with the render tracker method.
126   *
127   * @return the Observer associated with the render tracker method.
128   */
129  @ObserverRef
130  @Nonnull
131  protected abstract Observer getRenderObserver();
132
133  /**
134   * The first time an arez component is rendered it will lock the Arez scheduler and release
135   * the lock in the micro-task immediately following the task that prompted the render. If this
136   * is not done it is possible that Arez can re-trigger a component render when the scheduler is
137   * triggered after the tracked render completes but before the render method has returned to the
138   * react runtime. This results in error message from react as a setState()/forceRender() was invoked
139   * while still within a render() method.
140   *
141   * <p>NOTE: While render methods are read-only transactions, they can un-observe components with
142   * <code>disposeOnDeactivate=true</code> that would result in the component being disposed and
143   * triggering an update that would mark particular React Components/Observers as STALE and trigger
144   * a re-render of that component.</p>
145   */
146  protected final void pauseArezSchedulerUntilRenderLoopComplete()
147  {
148    if ( null == c_schedulerLock )
149    {
150      c_schedulerLock = getContext().pauseScheduler();
151      // Use a hack of an empty promise that immediately resolves to
152      // schedule the block immediately after this call stack pops.
153      Promise.resolve( (Object) null ).then( ignored -> {
154        c_schedulerLock.dispose();
155        c_schedulerLock = null;
156        return null;
157      } );
158    }
159  }
160
161  @PreDispose
162  protected void preDispose()
163  {
164    _unmounted = true;
165  }
166
167  /**
168   * {@inheritDoc}
169   */
170  @Override
171  protected final void populateDebugData( @Nonnull final JsPropertyMap<Object> newState )
172  {
173    // Present component id as state. Useful to track when instance ids change.
174    newState.set( "Arez.id", getArezComponentId() );
175    newState.set( "Arez.name", getArezComponentName() );
176
177    // Collect existing dependencies as state
178    final ObserverInfo observerInfo = getContext().getSpy().asObserverInfo( getRenderObserver() );
179    observerInfo.getDependencies().forEach( d -> newState.set( d.getName(), getValue( d ) ) );
180  }
181
182  /**
183   * Return the value for specified observable.
184   * Exceptions are caught and types are converted to strings using {@link Object#toString()}
185   *
186   * @param observableInfo the observable.
187   * @return the value as a string.
188   */
189  @SuppressWarnings( "UnnecessaryUnboxing" )
190  @Nullable
191  private Object getValue( @Nonnull final ObservableValueInfo observableInfo )
192  {
193    try
194    {
195      if ( Arez.arePropertyIntrospectorsEnabled() && observableInfo.hasAccessor() )
196      {
197        // Consider unwrapping collections and potentially serializing Arez entities so they are presented correctly in DevTools
198        final Object value = observableInfo.getValue();
199        if ( null == value )
200        {
201          return null;
202        }
203        else if ( value instanceof Enum )
204        {
205          return ( (Enum) value ).name();
206        }
207        else if ( value instanceof Integer )
208        {
209          return Js.asAny( ( (Integer) value ).intValue() );
210        }
211        else if ( value instanceof Boolean )
212        {
213          return Js.asAny( ( (Boolean) value ).booleanValue() );
214        }
215        else if ( value instanceof Long )
216        {
217          return Js.asAny( ( (Long) value ).doubleValue() );
218        }
219        else if ( value instanceof Float )
220        {
221          return Js.asAny( ( (Float) value ).doubleValue() );
222        }
223        else if ( value instanceof Short )
224        {
225          return Js.asAny( ( (Short) value ).intValue() );
226        }
227        else if ( value instanceof Byte )
228        {
229          return Js.asAny( ( (Byte) value ).intValue() );
230        }
231        else if ( value instanceof Character )
232        {
233          return value.toString();
234        }
235        else if ( value instanceof Stream )
236        {
237          // Streams are new instances every time so render them as strings to avoid infinite loops.
238          return "<Stream>";
239        }
240        else
241        {
242          return value;
243        }
244      }
245      else
246      {
247        return "?";
248      }
249    }
250    catch ( final Throwable throwable )
251    {
252      return throwable;
253    }
254  }
255}
