031/**
032 * A base class for all Arez enabled components. This class makes the component
033 * rendering reactive and it will schedule a re-render any time any of the observable
034 * entities accessed within the scope of the render method are changed.
035 *
036 * <p>To achieve this goal, the props and state of the component are converted into
037 * observable properties. This of course means they must be accessed within the scope
038 * of an Arez transaction. (Typically this means it needs to be accessed within the
039 * scope of a {@link Action} annotated method or within the scope of the render method.</p>
040 */
041public abstract class ReactArezComponent<P extends BaseProps, C extends BaseContext>
042  extends Component<P, BaseState, C>
043{
044  /**
045   * Key used to store the arez data in state.
046   */
047  private static final String AREZ_STATE_KEY = "arez";
048
049  private static int c_nextComponentId = 1;
050  private final int _arezComponentId;
051  private boolean _renderDepsChanged;
052
053  protected ReactArezComponent()
054  {
055    _arezComponentId = c_nextComponentId++;
056  }
057
058  /**
059   * {@inheritDoc}
060   */
061  @Observable( expectSetter = false )
062  @Override
063  protected P props()
064  {
065    return super.props();
066  }
067
068  /**
069   * Return the Observable used to make `props` observable.
070   *
071   * @return the Observable used to make `props` observable.
072   */
073  @ObservableRef
074  protected arez.Observable<P> getPropsObservable()
075  {
076    throw new IllegalStateException();
077  }
078
079  /**
080   * Action that invoked when props changes.
081   */
082  @Action
083  protected void reportPropsChanged()
084  {
085    getPropsObservable().reportChanged();
086  }
087
088  /**
089   * {@inheritDoc}
090   */
091  @Override
092  protected final void scheduleStateUpdate( @Nonnull final SetStateCallback<P, BaseState> callback,
093                                            @Nullable final Procedure onStateUpdateComplete )
094  {
095    Guards.fail( () -> "Attempted to schedule state update on ReactArezComponent subclass. Use Arez @Observable or @Computed properties instead." );
096  }
097
098  /**
099   * After construction of the object. Schedule any autoruns attached to component.
100   *
101   */
102  @Override
103  protected final void performComponentDidConstruct()
104  {
105    super.performComponentDidConstruct();
106    triggerScheduler();
107  }
108
109  /**
110   * Template method overridden by annotation processor if there are autoruns to schedule.
111   */
112  protected void triggerScheduler()
113  {
114  }
115
116  /**
117   * Return true if the render dependencies have been marked as changed and component has yet to be re-rendered.
118   *
119   * @return true if render dependencies changed, false otherwise.
120   */
121  protected final boolean hasRenderDepsChanged()
122  {
123    return _renderDepsChanged;
124  }
125
126  /**
127   * Hook used by Arez to notify component that it needs to be re-rendered.
128   */
129  @OnDepsChanged
130  protected final void onRenderDepsChanged()
131  {
132    _renderDepsChanged = true;
133    scheduleRender( true );
134  }
135
136  /**
137   * Return the arez context that this component is associated with.
138   * The component is associated with the context that was active when it was created
139   * and can only react to observables associated with the same context.
140   *
141   * @return the arez context that this component is associated with.
142   */
143  @ContextRef
144  @Nonnull
145  protected ArezContext getContext()
146  {
147    throw new IllegalStateException();
148  }
149
150  /**
151   * Return the unique identifier of component according to Arez.
152   * This method is invoked by the code generated by the Arez annotation processor.
153   *
154   * @return the unique identifier of component according to Arez.
155   */
156  @ComponentId
157  protected final int getArezComponentId()
158  {
159    return _arezComponentId;
160  }
161
162  /**
163   * Return the Observer associated with the render tracker method.
164   *
165   * @return the Observer associated with the render tracker method.
166   */
167  @ObserverRef
168  @Nonnull
169  protected Observer getRenderObserver()
170  {
171    throw new IllegalStateException();
172  }
173
174  /**
175   * {@inheritDoc}
176   */
177  @Track( name = "render" )
178  @Nullable
179  @Override
180  protected ReactNode performRender()
181  {
182    _renderDepsChanged = false;
183    return super.performRender();
184  }
185
186  /**
187   * {@inheritDoc}
188   */
189  @Override
190  protected boolean shouldComponentUpdate( @Nullable final P nextProps,
191                                           @Nullable final BaseState nextState,
192                                           @Nullable final C nextContext )
193  {
194    if ( hasRenderDepsChanged() )
195    {
196      return true;
197    }
198    //noinspection SimplifiableIfStatement
199    if ( !Js.isTripleEqual( super.state(), nextState ) )
200    {
201      // If state is not identical then we need to re-render ...
202      // Previously we chose not to re-render if only AREZ_STATE_KEY that was updated but that
203      // meant deps in DevTools would not be update so now we just re-render anyway.
204      return true;
205    }
206    else
207    {
208      /*
209       * We just compare the props shallowly and avoid a re-render if the props have not changed.
210       */
211      final boolean modified = JsUtil.isObjectShallowModified( super.props(), nextProps );
212      if ( modified )
213      {
214        reportPropsChanged();
215      }
216      return modified;
217    }
218  }
219
220  /**
221   * {@inheritDoc}
222   */
223  @Override
224  protected void componentDidMount()
225  {
226    storeArezDataAsState();
227  }
228
229  /**
230   * {@inheritDoc}
231   */
232  @Override
233  protected void componentDidUpdate( @Nullable final P prevProps, @Nullable final BaseState prevState )
234  {
235    storeArezDataAsState();
236  }
237
238  /**
239   * {@inheritDoc}
240   */
241  @Override
242  protected void componentWillUnmount()
243  {
244    /*
245     * Dispose of all the arez resources. Necessary particularly for the render tracker that should
246     * not receive notifications of updates after the component has been unmounted.
247     */
248    Disposable.dispose( this );
249  }
250
251  /**
252   * Store arez data such as dependencies on the state of component.
253   * This is only done if {@link ReactArezConfig#shouldStoreArezDataAsState()} returns true and is primarily
254   * done to make it easy to debug from within React DevTools.
255   */
256  private void storeArezDataAsState()
257  {
258    if ( ReactArezConfig.shouldStoreArezDataAsState() && Arez.areSpiesEnabled() )
259    {
260      final Observer renderTracker = getRenderObserver();
261      final List<ObservableInfo> dependencies = getContext().getSpy().getDependencies( renderTracker );
262      final JsPropertyMap<Object> deps = JsPropertyMap.of();
263      dependencies.forEach( d -> deps.set( d.getName(), d ) );
264      final JsPropertyMap<Object> data = JsPropertyMap.of();
265      data.set( "name", renderTracker.getName() );
266      data.set( "observer", renderTracker );
267      data.set( "deps", deps );
268      final BaseState state = super.state();
269      final Object currentArezData = null != state ? Js.asPropertyMap( state ).get( AREZ_STATE_KEY ) : null;
270      final Object currentDepsData = null != currentArezData ? Js.asPropertyMap( currentArezData ).get( "deps" ) : null;
271      /*
272       * Do a shallow comparison against object and the deps. If either has changed then state needs to be updated.
273       * We skip deps on shallow comparison of data as it is always recreated anew.
274       */
275      if ( JsUtil.isObjectShallowModified( currentArezData, data, "deps" ) )
276      {
277        scheduleArezKeyUpdate( data );
278      }
279      else if ( null == currentDepsData )
280      {
281        scheduleArezKeyUpdate( data );
282      }
283      else
284      {
285        /*
286         * Deps are mappings to Info objects that can be garbage collected over time.
287         * So we just make sure the keys (which are the info objects names) match.
288         */
289        final String[] currentDeps = JsObject.keys( Js.uncheckedCast( currentDepsData ) );
290        final String[] newDeps = JsObject.keys( Js.uncheckedCast( deps ) );
291        if ( currentDeps.length != newDeps.length )
292        {
293          scheduleArezKeyUpdate( data );
294        }
295        else
296        {
297          for ( int i = 0; i < currentDeps.length; i++ )
298          {
299            if ( !currentDeps[ i ].equals( newDeps[ i ] ) )
300            {
301              scheduleArezKeyUpdate( data );
302            }
303          }
304        }
305      }
306    }
307  }
308
309  /**
310   * Schedule state update the updates arez state.
311   * Makes sure the super class is invoked so reportStateChanged() is not invoked on State observable.
312   */
313  private void scheduleArezKeyUpdate( @Nonnull final JsPropertyMap<Object> data )
314  {
315    super.scheduleStateUpdate( ( p, s ) -> Js.cast( JsPropertyMap.of( AREZ_STATE_KEY, data ) ), null );
316  }
