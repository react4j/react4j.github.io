001package react4j.arez;

003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006
007/**
008 * Marks event handler so that it is not automatically annotated with Action.
009 * This is should only be present in subclasses of {@link ReactArezComponent} and should only be placed on
010 * methods annotated with react4j.annotations.EventHandler. This method stops the react4j processor from
011 * generating an {@link arez.annotations.Action} annotation on synthesized method. The developer
012 * is still free to add their own {@link arez.annotations.Action} annotation and this may be useful
013 * if you need fine-grain control of action parameters.
014 */
015@Documented
016@Target( ElementType.METHOD )
017public @interface NoAutoAction
018{
019}
