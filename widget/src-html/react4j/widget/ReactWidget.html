001package react4j.widget;
002
003import com.google.gwt.dom.client.Document;
004import com.google.gwt.dom.client.Element;
005import com.google.gwt.user.client.ui.Widget;
006import java.util.Objects;
007import javax.annotation.Nonnull;
008import jsinterop.base.Js;
009import org.realityforge.braincheck.BrainCheckConfig;
010import react4j.React;
011import react4j.ReactNode;
012import react4j.dom.ReactDOM;
013import static org.realityforge.braincheck.Guards.*;
014
015/**
016 * A GWT widget that renders a React4j element.
017 */
018public class ReactWidget
019  extends Widget
020{
021  /**
022   * The GWT Element representation.
023   */
024  private final Element _element;
025  /**
026   * The react element to render.
027   */
028  private final ReactNode _node;
029
030  /**
031   * Create the widget.
032   *
033   * @param node the react element to render.
034   */
035  public ReactWidget( @Nonnull final ReactNode node )
036  {
037    _element = Document.get().createDivElement();
038    _node = Objects.requireNonNull( node );
039    if ( BrainCheckConfig.checkApiInvariants() )
040    {
041      apiInvariant( () -> React.isValidElement( node ),
042                    () -> "A ReactNode that is not a valid element has been incorrectly passed to ReactWidget constructor" );
043    }
044    setElement( _element );
045  }
046
047  /**
048   * {@inheritDoc}
049   */
050  @Override
051  public void onAttach()
052  {
053    super.onAttach();
054    ReactDOM.render( _node, Js.uncheckedCast( _element ) );
055
056  }
057
058  /**
059   * {@inheritDoc}
060   */
061  @Override
062  public void onDetach()
063  {
064    ReactDOM.unmountComponentAtNode( Js.uncheckedCast( _element ) );
065    super.onDetach();
066  }
067}
